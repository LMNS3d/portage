24 // intersect one source polygon (possibly non-convex) with a
25 // triangular decomposition of a target polygon
26 
27 inline
28 std::vector<double>
29 intersect_polys_r2d(std::vector<Wonton::Point<2>> const & source_poly,
30  std::vector<Wonton::Point<2>> const & target_poly,
31  NumericTolerances_t num_tols) {
32 
33  std::vector<double> moments(3, 0);
34  bool src_convex = true;
35  bool trg_convex = true;
36 
37  const int POLY_ORDER = 1; // max degree of moments to calculate
38 
39  // Initialize source polygon
40 
41  const int size1 = source_poly.size();
42  const int size2 = target_poly.size();
43  if (!size1 || !size2)
44  return moments; // could allow top level code to avoid an 'if' statement
45 
46  std::vector<r2d_rvec2> verts1(size1);
47  for (int i = 0; i < size1; ++i) {
48  verts1[i].xy[0] = source_poly[i][0];
49  verts1[i].xy[1] = source_poly[i][1];
50  }
51 #ifdef DEBUG
52  double volume1 = 0.;
53  for (int i = 1; i < size1-1; ++i)
54  volume1 += r2d_orient(verts1[0], verts1[i], verts1[i+1]);
55  if (volume1 < 0.)
56  throw std::runtime_error("target_poly has negative volume");
57 #endif
58 
59  r2d_poly srcpoly_r2d;
60  r2d_init_poly(&srcpoly_r2d, &verts1[0], size1);
61 
62 #ifdef DEBUG
63  if (r2d_is_good(&srcpoly_r2d) == 0)
64  throw std::runtime_error("source_poly: invalid poly");
65 #endif
66 
67 
68  // Initialize target polygon and check for convexity
69 
70  std::vector<r2d_plane> faces(size2);
71  std::vector<r2d_rvec2> verts2(size2);
72  for (int i = 0; i < size2; ++i) {
73  verts2[i].xy[0] = target_poly[i][0];
74  verts2[i].xy[1] = target_poly[i][1];
75  }
76 
77  // check for convexity of target polygon
78  for (int i = 0; i < size2; ++i)
79  if (r2d_orient(verts2[i], verts2[(i+1)%size2],
80  verts2[(i+2)%size2]) < num_tols.polygon_convexity_eps)
81  trg_convex = false;
82 
83  // case 1: target_poly is convex
84  // can simply use faces of target_poly as clip planes
85  if (trg_convex) {
86  r2d_poly_faces_from_verts(&faces[0], &verts2[0], size2);
87 
88  // clip the first poly against the faces of the second
89  r2d_clip(&srcpoly_r2d, &faces[0], size2);
90 
91  // find the moments (up to quadratic order) of the clipped poly
92  r2d_real om[R2D_NUM_MOMENTS(POLY_ORDER)];
93  r2d_reduce(&srcpoly_r2d, om, POLY_ORDER);
94 
95  // Check that the returned volume is positive (if the volume is zero,
96  // i.e. abs(om[0]) < eps, then it can sometimes be slightly negative,
97  // like om[0] == -1.24811e-16.
98  if (om[0] < num_tols.minimal_intersection_volume)
99  throw std::runtime_error("Negative volume");
100 
101  // Copy moments:
102  for (int j = 0; j < 3; ++j)
103  moments[j] = om[j];
104 
105  } else { // case 2: target_poly is non-convex
106 
107  // check for convexity of source polygon
108  for (int i = 0; i < size1; ++i)
109  if (r2d_orient(verts1[i], verts1[(i+1)%size1],
110  verts1[(i+2)%size1]) < num_tols.polygon_convexity_eps)
111  src_convex = false;
112 
113  // if source polygon is convex while target polygon is non-convex,
114  // call the routine with the polygons reversed
115 
116  if (src_convex)
117  return intersect_polys_r2d(target_poly, source_poly, num_tols);
118  else {
119 
120  // Must divide target_poly into triangles for clipping. Choice
121  // of the central point is crucial. Try the centroid first -
122  // computed by the area weighted sum of centroids of any
123  // triangulation of the polygon
124  bool center_point_ok = true;
125  Wonton::Point<2> cen(0.0, 0.0);
126  r2d_rvec2 cenr2d;
127  cenr2d.xy[0] = 0.0; cenr2d.xy[1] = 0.0;
128  double area_sum = 0.0;
129  for (int i = 1; i < size2; ++i) {
130  double area = r2d_orient(verts2[0], verts2[i], verts2[(i+1)%size2]);
131  area_sum += area;
132  Wonton::Point<2> tricen =
133  (target_poly[0] + target_poly[i] + target_poly[(i+1)%size2])/3.0;
134  cen += area*tricen;
135  }
136  cen /= area_sum;
137  cenr2d.xy[0] = cen[0]; cenr2d.xy[1] = cen[1];
138 
139  for (int i = 0; i < size2; ++i)
140  if (r2d_orient(cenr2d, verts2[i], verts2[(i+1)%size2]) < 0.)
141  center_point_ok = false;
142 
143  if (!center_point_ok) {
144  // If the centroid is not ok, we have to find the center of
145  // the feasible set of the polygon. This means clipping the
146  // target_poly with its own face planes/lines. For a
147  // non-convex polygon, this will give a new polygon whose
148  // interior (See Garimella/Shashkov/Pavel paper on untangling)
149 
150  r2d_poly fspoly;
151  r2d_init_poly(&fspoly, &verts2[0], size2);
152 
153  r2d_poly_faces_from_verts(&faces[0], &verts2[0], size2);
154 
155  r2d_clip(&fspoly, &faces[0], size2);
156 
157  // If the resulting polygon is empty, we are out of luck
158  if (fspoly.nverts == 0)
159  std::runtime_error("Could not find a valid center point to triangulate non-convex polygon");
160 
161  // Have R2D compute first and second moments of polygon and
162  // get its centroid from that
163 
164  r2d_real fspoly_moments[R2D_NUM_MOMENTS(POLY_ORDER)];
165  r2d_reduce(&fspoly, fspoly_moments, POLY_ORDER);
166 
167  cen[0] = cenr2d.xy[0] = fspoly_moments[1]/fspoly_moments[0];
168  cen[1] = cenr2d.xy[1] = fspoly_moments[2]/fspoly_moments[0];
169 
170  // Even if the resulting feasible set polygon has vertices,
171  // maybe its degenerate. So we have to verify that its centroid
172  // indeed is a point that will give valid triangles when
173  // paired with the edges of the target polygon.
174 
175  for (int i = 0; i < size2; ++i)
176  if (r2d_orient(cenr2d, verts2[i], verts2[(i+1)%size2]) < 0.)
177  center_point_ok = false;
178  }
179 
180  // If we still don't have a good center point, we are out of luck
181  if (!center_point_ok)
182  std::runtime_error("Could not find a valid center point to triangulate non-convex polygon");
183 
184 
185  for (int i = 0; i < size2; ++i) {
186  verts2[0].xy[0] = cen[0];
187  verts2[0].xy[1] = cen[1];
188  verts2[1].xy[0] = target_poly[i][0];
189  verts2[1].xy[1] = target_poly[i][1];
190  verts2[2].xy[0] = target_poly[(i+1)%size2][0];
191  verts2[2].xy[1] = target_poly[(i+1)%size2][1];
192 
193  r2d_poly_faces_from_verts(&faces[0], &verts2[0], 3);
194 
195  r2d_poly srcpoly_r2d_copy = srcpoly_r2d;
196 
197  // clip the first poly against the faces of the second
198  r2d_clip(&srcpoly_r2d_copy, &faces[0], 3);
199 
200  // find the moments (up to quadratic order) of the clipped poly
201  // Only do this check in Debug mode:
202 #ifdef DEBUG
203  if (R2D_NUM_MOMENTS(POLY_ORDER) != 3)
204  throw std::runtime_error("Invalid number of moments");
205 #endif
206  r2d_real om[R2D_NUM_MOMENTS(POLY_ORDER)];
207  r2d_reduce(&srcpoly_r2d_copy, om, POLY_ORDER);
208 
209  // Check that the returned volume is positive (if the volume is zero,
210  // i.e. abs(om[0]) < eps, then it can sometimes be slightly negative,
211  // like om[0] == -1.24811e-16.
212  if (om[0] < num_tols.minimal_intersection_volume)
213  throw std::runtime_error("Negative volume for triangle of polygon");
214 
215  // Accumulate moments:
216  for (int j = 0; j < 3; ++j)
217  moments[j] += om[j];
218  } // for i
219  } // if (src_convex) ... else ...
220  } // if convex {} else {}
221 
222  return moments;
223 }
