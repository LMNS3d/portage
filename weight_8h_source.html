245 double kernel(const Kernel kern, double x) {
246  double result;
247  switch (kern) {
248  case B4:{result = b4(x); break;}

268 double elliptic(Wonton::Point<dim> x, Wonton::Point<dim> y, array<double,dim> &h) {
269  double distance = 0.0, result;
270  for (size_t i=0; i<dim; i++) {
271  distance += (x[i]-y[i])*(x[i]-y[i])/(h[i]*h[i]);
272  }
273  distance = sqrt(distance);
274  return distance;

289 template<size_t dim>
290 double eval(const Geometry geo,
291  const Kernel kern,
292  const Wonton::Point<dim> x, const Wonton::Point<dim> y,
293  array<double,dim> h)
294 {
295  double result;
296  double norm = kernel(kern, 0.0);
297  switch (geo) {
298  case ELLIPTIC: {
299  double arg = elliptic<dim>(x,y,h);
300  result = kernel(kern, arg) / norm;
301  break;
302  }
303 
304  case TENSOR:{
305  result = 1.;
306  array<double,dim> arg = tensor<dim>(x,y,h);
307  for (size_t i=0; i<dim; i++) {
308  result *= kernel(kern, arg[i]) / norm;
309  }
310  break;
311  }
312  default:
313  throw std::runtime_error("invalid weight geometry");
314  }
315  return result;
316 }

326 template<size_t dim>
327 double faceted(const Wonton::Point<dim> x, const Wonton::Point<dim> y,
328  FacetData<dim>* facets, size_t nsides)
329 {
330  double result = 1.;
331  for (size_t i=0; i<nsides; i++) {
332  double arg = 0.;
333  for (size_t j=0; j<dim; j++) arg += facets[i].normal[j]*(y[j]-x[j]);
334  arg /= facets[i].smoothing;
335  result *= polyramp(arg);
336  }
337  return result;
338 }

341 template<size_t dim>
342 double eval(const Geometry geo,
343  const Kernel kern,
344  const Wonton::Point<dim> x, const Wonton::Point<dim> y,
345  vector<vector<double>> vh)
346 {
347  double result;
348  double norm = kernel(kern, 0.0);
349  switch (geo) {
350  case TENSOR:
351  case ELLIPTIC: {
352  array<double, dim> h;
353  for (size_t i=0; i<dim; i++) h[i] = vh[0][i];
354  result = eval<dim>(geo,kern,x,y,h);
355  break;
356  }
357 
358  case FACETED:{
359  size_t nsides = vh.size();
360  FacetData<dim> facets[nsides];
361  for (size_t i=0; i<nsides; i++) {
362  for (size_t j=0; j<dim; j++) facets[i].normal[j] = vh[i][j];
363  facets[i].smoothing = vh[i][dim];
364  }
365  result = faceted<dim>(x,y,facets,nsides);
366  break;
367  }
368 
369  default:
370  throw std::runtime_error("invalid weight geometry");
371  }
372  return result;
