237  void node_get_cell_adj_nodes(int const nodeid,
238  Entity_type const ptype,
239  std::vector<int> *adjnodes) const {
240  adjnodes->clear();
241 
242  // Find the cells attached to this node
243  std::vector<int> nodecells;
244  mesh_.node_get_cells(nodeid, &nodecells);
245 
246  // Loop over these cells, and find their nodes; these are the ones we seek
247  // but make sure we aren't duplicating them
248  for (auto const& c : nodecells) {
249  std::vector<int> cellnodes;
250  cell_get_nodes(c, &cellnodes);
251 
252  for (auto const& n : cellnodes) {
253  if (n == nodeid) continue;
254  if (std::find(adjnodes->begin(), adjnodes->end(), n) == adjnodes->end())
255  adjnodes->emplace_back(n);
256  }
257  }
258  } // node_get_cell_adj_nodes
259
