209  // Find the cells attached to this node

212 
213  // Loop over these cells, and find their nodes; these are the ones we seek
214  // but make sure we aren't duplicating them
215  for (auto const& c : nodecells) {
216  std::vector<int> cellnodes;
217  cell_get_nodes(c, &cellnodes);
218 
219  for (auto const& n : cellnodes) {
220  if (n == nodeid) continue;
221  if (std::find(adjnodes->begin(), adjnodes->end(), n) == adjnodes->end())
222  adjnodes->emplace_back(n);
