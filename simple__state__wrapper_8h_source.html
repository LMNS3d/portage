136  Entity_kind get_entity(std::string const name) const {
137  // Find where the name is in the variable name part of the map's keys
138  auto it = state_.begin();
139  while (it != state_.end()) {
140  if (it->first.first == name)
141  break;
142  else
143  ++it;
144  }
145  // Now get the second part of the key, if it was found
146  if (it != state_.end()) {
147  return it->first.second;
148  } else {
149  // We don't know this variable, so bail.
150  std::runtime_error("Requested variable not found.");
151  }
152  }
