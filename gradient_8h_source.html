229 template<int D, typename MeshType, typename StateType,
230 template<class, int, class, class> class InterfaceReconstructorType,
231 class Matpoly_Splitter, class Matpoly_Clipper>
232 Vector<D> Limited_Gradient <D, Entity_kind::CELL, MeshType, StateType,
233 InterfaceReconstructorType, Matpoly_Splitter, Matpoly_Clipper>::operator()(int cellid) {
234 
235  assert(this->vals_);
236 
237  double phi = 1.0;
238  Vector<D> grad;
239 
240  std::vector<int> nbrids{cellid}; // Include cell where grad is needed as first element
241 
242  if(cell_neighbors_.size()) {
243  nbrids.insert(std::end(nbrids),
244  std::begin(cell_neighbors_[cellid]),
245  std::end(cell_neighbors_[cellid]));
246  }
247  std::vector<Point<D>> ls_coords;
248  std::vector<double> ls_vals;
249 
250  // Loop over cell where grad is needed and its neighboring cells
251  for (auto nbrid_g : nbrids) {
252 
253 #ifdef HAVE_TANGRAM
254 
255  // Field values for each cells in each material are stored according to
256  // the material's cell list. So, get the local index of each neighbor cell
257  // in the material cell list to access the correct field value
258  int nbrid_l = (this->field_type_ == Field_type::MESH_FIELD) ?
259  nbrid_g : this->state_.cell_index_in_material(nbrid_g, this->matid_);
260 
261  // In the case of material-data, need to check that neighbors contain material of interest
262  // (i.e. have valid local id); in the case of mesh data, this is always true,
263  // since local cellid (nbrid_l) is equal to global cellid (nbrid_g)
264  if (nbrid_l >= 0) { // cell_index_in_material can return -1
265  std::vector<int> cellmats;
266  this->state_.cell_get_mats(nbrid_g, &cellmats);
267 
268  if (this->interface_reconstructor_ && cellmats.size() > 1){ // Multi-material cell
269  // Get cell's cellmatpoly
270  auto cellmatpoly =
271  this->interface_reconstructor_->cell_matpoly_data(nbrid_g);
272 
273  // Collect all the matpolys in this cell for the material of interest
274  std::vector<Tangram::MatPoly<D>> matpolys =
275  cellmatpoly.get_matpolys(this->matid_);
276 
277  // If there are multiple matpolys in this cell for the material of interest,
278  // aggregate moments to compute new centroid
279  for (int ipoly=0; ipoly<matpolys.size(); ipoly++) {
280  std::vector<double> moments = matpolys[ipoly].moments();
281  Point<D> centroid;
282  for (int k = 0; k < D; k++)
283  centroid[k] = moments[k+1]/moments[0];
284  ls_coords.push_back(centroid);
285  break; // TODO: Instead of cutting out after the first matpoly,
286  // Get matpoly moments directly using new interface in Tangram,
287  // aggregate, and use to compute overall material centroid
288  }
289 
290  // Populate least squares vectors with centroid for material
291  // of interest and field value in the current cell for that material
292  ls_vals.push_back(this->vals_[nbrid_l]);
293  } else if (cellmats.size() == 1) { // Single material cell
294 
295  // Ensure that the single material is the material of interest
296  if (cellmats[0] == this->matid_) {
297  // Get the cell-centered value for this material
298  Point<D> centroid;
299  this->mesh_.cell_centroid(nbrid_g, &centroid);
300  ls_coords.push_back(centroid);
301  ls_vals.push_back(this->vals_[nbrid_l]);
302  }
303  }
304  }
305 #endif
306  // If we get here, we must have mesh data which is cell-centered
307  // and not dependent on material, so just get the centroid and value
308  if (this->field_type_ == Field_type::MESH_FIELD){
309  Point<D> centroid;
310  this->mesh_.cell_centroid(nbrid_g, &centroid);
311  ls_coords.push_back(centroid);
312  ls_vals.push_back(this->vals_[nbrid_g]);
313  }
314  }
315 
316  grad = Wonton::ls_gradient(ls_coords, ls_vals);
317 
318  // Limit the gradient to enforce monotonicity preservation
319  if (this->limtype_ == BARTH_JESPERSEN &&
320  !this->mesh_.on_exterior_boundary(Entity_kind::CELL, cellid)) { // No limiting on boundary
321 
322  phi = 1.0;
323 
324  // Min and max vals of function (cell centered vals) among neighbors
325  // and the cell itself
327 
328  double minval = ls_vals[0];
329  double maxval = ls_vals[0];
330  double cellcenval = ls_vals[0];
331 
332  // Find min and max values among all neighbors (exlude the first element
333  // in nbrids because it corresponds to the cell itself, not a neighbor)
334  for (int i = 1; i < ls_vals.size(); ++i) {
335  minval = std::min(ls_vals[i], minval);
336  maxval = std::max(ls_vals[i], maxval);
337  }
338 
339  /* Per page 278 of [Kucharik, M. and Shaskov, M, "Conservative
340  Multi-material Remap for Staggered Multi-material Arbitrary
341  Lagrangian-Eulerian Methods," Journal of Computational Physics,
342  v 258, pp. 268-304, 2014], if a cell is a multimaterial cell and the
343  field is a material field, then the min value (for density, at
344  least) should be set to 0 and the max value to infinity (or a large
345  number), so that we don't end up limiting the gradient to 0 and drop
346  down to 1st order. But we don't know if a variable is density (don't
347  want to do silly things like string comparison) or pressure or
348  something else? What if we limit pressure to 0 and it should
349  actually be allowed to go -ve? In the end, along with the limiter
350  type, the application should be able to tell Portage the global
351  bounds that a variable has to satisfy. Then we can impose the global
352  limits at multi-material cells and boundary cells without limiting
353  the gradient to 0. */
354 
355  // Find the min and max of the reconstructed function in the cell
356  // Since the reconstruction is linear, this will occur at one of
357  // the nodes of the cell. So find the values of the reconstructed
358  // function at the nodes of the cell
359 
360  int dim = this->mesh_.space_dimension();
361  std::vector<Point<D>> cellcoords;
362  this->mesh_.cell_get_coordinates(cellid, &cellcoords);
363 
364  for (auto coord : cellcoords) {
365  Vector<D> vec = coord-ls_coords[0];
366  double diff = dot(grad, vec);
367  double extremeval = (diff > 0.0) ? maxval : minval;
368  double phi_new = (diff == 0.0) ? 1 : (extremeval-cellcenval)/diff;
369  phi = std::min(phi_new, phi);
370  }
371  }
372 
373  // Limited gradient is phi*grad
374  return phi*grad;
375  }

439  template<int D, typename MeshType, typename StateType>
440  Vector<D> Limited_Gradient <D, Entity_kind::NODE, MeshType, StateType>::operator() (int nodeid) {
441 
442  assert(this->vals_);
443 
444  double phi = 1.0;
445  Vector<D> grad;
446 
447  std::vector<int> const & nbrids = node_neighbors_[nodeid];
448  std::vector<Point<D>> nodecoords(nbrids.size()+1);
449  std::vector<double> nodevalues(nbrids.size()+1);
450  this->mesh_.node_get_coordinates(nodeid, &(nodecoords[0]));
451  nodevalues[0] = this->vals_[nodeid];
452 
453  int i = 1;
454  for (auto const & nbrnode : nbrids) {
455  this->mesh_.node_get_coordinates(nbrnode, &nodecoords[i]);
456  nodevalues[i] = this->vals_[nbrnode];
457  i++;
458  }
459 
460  grad = Wonton::ls_gradient(nodecoords, nodevalues);
461 
462  if (this->limtype_ == BARTH_JESPERSEN &&
463  !this->mesh_.on_exterior_boundary(Entity_kind::NODE, nodeid)) { // No limiting on boundary
464 
465  // Min and max vals of function (cell centered vals) among neighbors
466  double minval = this->vals_[nodeid];
467  double maxval = this->vals_[nodeid];
468 
469  for (auto const & val : nodevalues) {
470  minval = std::min(val, minval);
471  maxval = std::max(val, maxval);
472  }
473 
474  // Find the min and max of the reconstructed function in the cell
475  // Since the reconstruction is linear, this will occur at one of
476  // the nodes of the cell. So find the values of the reconstructed
477  // function at the nodes of the cell
478 
479  double nodeval = this->vals_[nodeid];
480 
481  std::vector<Point<D>> dualcellcoords;
482  this->mesh_.dual_cell_get_coordinates(nodeid, &dualcellcoords);
483 
484  for (auto const & coord : dualcellcoords) {
485  Vector<D> vec = coord-nodecoords[0];
486  double diff = dot(grad, vec);
487  double extremeval = (diff > 0.0) ? maxval : minval;
488  double phi_new = (diff == 0.0) ? 1 : (extremeval-nodeval)/diff;
489  phi = std::min(phi_new, phi);
490  }
491  }
492 
493  // Limited gradient is phi*grad
494 
495  return phi*grad;
496  }
