230 template<int D, typename MeshType, typename StateType,
231  template<class, int, class, class> class InterfaceReconstructorType,
232  class Matpoly_Splitter, class Matpoly_Clipper, class CoordSys>
233 Vector<D> Limited_Gradient <D, Entity_kind::CELL, MeshType, StateType,
234  InterfaceReconstructorType, Matpoly_Splitter, Matpoly_Clipper,
235  CoordSys>::operator()(int cellid) {
236 
237  assert(this->vals_);
238 
239  double phi = 1.0;
240  Vector<D> grad;
241 
242  std::vector<int> nbrids{cellid}; // Include cell where grad is needed as first element
243 
244  if(cell_neighbors_.size()) {
245  nbrids.insert(std::end(nbrids),
246  std::begin(cell_neighbors_[cellid]),
247  std::end(cell_neighbors_[cellid]));
248  }
249  std::vector<Point<D>> ls_coords;
250  std::vector<double> ls_vals;
251 
252  // Loop over cell where grad is needed and its neighboring cells
253  for (auto nbrid_g : nbrids) {
254 
255 #ifdef HAVE_TANGRAM
256 
257  // Field values for each cells in each material are stored according to
258  // the material's cell list. So, get the local index of each neighbor cell
259  // in the material cell list to access the correct field value
260  int nbrid_l = (this->field_type_ == Field_type::MESH_FIELD) ?
261  nbrid_g : this->state_.cell_index_in_material(nbrid_g, this->matid_);
262 
263  // In the case of material-data, need to check that neighbors contain material of interest
264  // (i.e. have valid local id); in the case of mesh data, this is always true,
265  // since local cellid (nbrid_l) is equal to global cellid (nbrid_g)
266  if (nbrid_l >= 0) { // cell_index_in_material can return -1
267  std::vector<int> cellmats;
268  this->state_.cell_get_mats(nbrid_g, &cellmats);
269 
270  if (this->interface_reconstructor_ && cellmats.size() > 1){ // Multi-material cell
271  // Get cell's cellmatpoly
272  auto cellmatpoly =
273  this->interface_reconstructor_->cell_matpoly_data(nbrid_g);
274 
275  // Collect all the matpolys in this cell for the material of interest
276  std::vector<Tangram::MatPoly<D>> matpolys =
277  cellmatpoly.get_matpolys(this->matid_);
278 
279  // If there are multiple matpolys in this cell for the material of interest,
280  // aggregate moments to compute new centroid
281  double mvol = 0.0;
282  Point<D> centroid;
283  for (int ipoly=0; ipoly<matpolys.size(); ipoly++) {
284  std::vector<double> moments = matpolys[ipoly].moments();
285  mvol += moments[0];
286  for (int k = 0; k < D; k++)
287  centroid[k] += moments[k+1];
288  }
289 
290  for (int k = 0; k < D; k++)
291  centroid[k] /= mvol;
292 
293  ls_coords.push_back(centroid);
294 
295  // Populate least squares vectors with centroid for material
296  // of interest and field value in the current cell for that material
297  ls_vals.push_back(this->vals_[nbrid_l]);
298  } else if (cellmats.size() == 1) { // Single material cell
299 
300  // Ensure that the single material is the material of interest
301  if (cellmats[0] == this->matid_) {
302  // Get the cell-centered value for this material
303  Point<D> centroid;
304  this->mesh_.cell_centroid(nbrid_g, &centroid);
305  ls_coords.push_back(centroid);
306  ls_vals.push_back(this->vals_[nbrid_l]);
307  }
308  }
309  }
310 #endif
311  // If we get here, we must have mesh data which is cell-centered
312  // and not dependent on material, so just get the centroid and value
313  if (this->field_type_ == Field_type::MESH_FIELD){
314  Point<D> centroid;
315  this->mesh_.cell_centroid(nbrid_g, &centroid);
316  ls_coords.push_back(centroid);
317  ls_vals.push_back(this->vals_[nbrid_g]);
318  }
319  }
320 
321  grad = Wonton::ls_gradient<D,CoordSys>(ls_coords, ls_vals);
322 
323  // Limit the gradient to enforce monotonicity preservation
324  if (this->limtype_ == BARTH_JESPERSEN &&
325  !this->mesh_.on_exterior_boundary(Entity_kind::CELL, cellid)) { // No limiting on boundary
326 
327  phi = 1.0;
328 
329  // Min and max vals of function (cell centered vals) among neighbors
330  // and the cell itself
332 
333  double minval = ls_vals[0];
334  double maxval = ls_vals[0];
335  double cellcenval = ls_vals[0];
336 
337  // Find min and max values among all neighbors (exlude the first element
338  // in nbrids because it corresponds to the cell itself, not a neighbor)
339  for (int i = 1; i < ls_vals.size(); ++i) {
340  minval = std::min(ls_vals[i], minval);
341  maxval = std::max(ls_vals[i], maxval);
342  }
343 
344  /* Per page 278 of [Kucharik, M. and Shaskov, M, "Conservative
345  Multi-material Remap for Staggered Multi-material Arbitrary
346  Lagrangian-Eulerian Methods," Journal of Computational Physics,
347  v 258, pp. 268-304, 2014], if a cell is a multimaterial cell and the
348  field is a material field, then the min value (for density, at
349  least) should be set to 0 and the max value to infinity (or a large
350  number), so that we don't end up limiting the gradient to 0 and drop
351  down to 1st order. But we don't know if a variable is density (don't
352  want to do silly things like string comparison) or pressure or
353  something else? What if we limit pressure to 0 and it should
354  actually be allowed to go -ve? In the end, along with the limiter
355  type, the application should be able to tell Portage the global
356  bounds that a variable has to satisfy. Then we can impose the global
357  limits at multi-material cells and boundary cells without limiting
358  the gradient to 0. */
359 
360  // Find the min and max of the reconstructed function in the cell
361  // Since the reconstruction is linear, this will occur at one of
362  // the nodes of the cell. So find the values of the reconstructed
363  // function at the nodes of the cell
364 
365  int dim = this->mesh_.space_dimension();
366  std::vector<Point<D>> cellcoords;
367  this->mesh_.cell_get_coordinates(cellid, &cellcoords);
368 
369  for (auto coord : cellcoords) {
370  Vector<D> vec = coord-ls_coords[0];
371  double diff = dot(grad, vec);
372  double extremeval = (diff > 0.0) ? maxval : minval;
373  double phi_new = (diff == 0.0) ? 1 : (extremeval-cellcenval)/diff;
374  phi = std::min(phi_new, phi);
375  }
376  }
377 
378  // Limited gradient is phi*grad
379  return phi*grad;
380  }

448 template<int D, typename MeshType, typename StateType,
449  template<class, int, class, class> class InterfaceReconstructorType,
450  class Matpoly_Splitter, class Matpoly_Clipper, class CoordSys>
451 Vector<D> Limited_Gradient <D, Entity_kind::NODE, MeshType, StateType,
452  InterfaceReconstructorType, Matpoly_Splitter, Matpoly_Clipper,
453  CoordSys>::operator() (int nodeid) {
454 
455  assert(this->vals_);
456 
457  double phi = 1.0;
458  Vector<D> grad;
459 
460  std::vector<int> const & nbrids = node_neighbors_[nodeid];
461  std::vector<Point<D>> nodecoords(nbrids.size()+1);
462  std::vector<double> nodevalues(nbrids.size()+1);
463  this->mesh_.node_get_coordinates(nodeid, &(nodecoords[0]));
464  nodevalues[0] = this->vals_[nodeid];
465 
466  int i = 1;
467  for (auto const & nbrnode : nbrids) {
468  this->mesh_.node_get_coordinates(nbrnode, &nodecoords[i]);
469  nodevalues[i] = this->vals_[nbrnode];
470  i++;
471  }
472 
473  grad = Wonton::ls_gradient<D,CoordSys>(nodecoords, nodevalues);
474 
475  if (this->limtype_ == BARTH_JESPERSEN &&
476  !this->mesh_.on_exterior_boundary(Entity_kind::NODE, nodeid)) { // No limiting on boundary
477 
478  // Min and max vals of function (cell centered vals) among neighbors
479  double minval = this->vals_[nodeid];
480  double maxval = this->vals_[nodeid];
481 
482  for (auto const & val : nodevalues) {
483  minval = std::min(val, minval);
484  maxval = std::max(val, maxval);
485  }
486 
487  // Find the min and max of the reconstructed function in the cell
488  // Since the reconstruction is linear, this will occur at one of
489  // the nodes of the cell. So find the values of the reconstructed
490  // function at the nodes of the cell
491 
492  double nodeval = this->vals_[nodeid];
493 
494  std::vector<Point<D>> dualcellcoords;
495  this->mesh_.dual_cell_get_coordinates(nodeid, &dualcellcoords);
496 
497  for (auto const & coord : dualcellcoords) {
498  Vector<D> vec = coord-nodecoords[0];
499  double diff = dot(grad, vec);
500  double extremeval = (diff > 0.0) ? maxval : minval;
501  double phi_new = (diff == 0.0) ? 1 : (extremeval-nodeval)/diff;
502  phi = std::min(phi_new, phi);
503  }
504  }
505 
506  // Limited gradient is phi*grad
507 
508  return phi*grad;
509  }
