157 template<int D, typename MeshType, typename StateType>
158 Vector<D>
159 Limited_Gradient<D, CELL, MeshType, StateType>::operator() (int const cellid) {
160 
161  assert(D == mesh_.space_dimension());
162  assert(D == 2 || D == 3);
163  double phi = 1.0;
164  Vector<D> grad;
165 
166  std::vector<int> const & nbrids = cell_neighbors_[cellid];
167 
168  std::vector<Point<D>> cellcenters(nbrids.size()+1);
169  std::vector<double> cellvalues(nbrids.size()+1);
170 
171  mesh_.cell_centroid(cellid, &(cellcenters[0]));
172 
173  cellvalues[0] = vals_[cellid];
174 
175  int i = 1;
176  for (auto nbrcell : nbrids) {
177  mesh_.cell_centroid(nbrcell, &(cellcenters[i]));
178 
179  cellvalues[i] = vals_[nbrcell];
180  i++;
181  }
182 
183  grad = ls_gradient(cellcenters, cellvalues);
184 
185  // Limit the gradient to enforce monotonicity preservation
186 
187  if (limtype_ == BARTH_JESPERSEN &&
188  !mesh_.on_exterior_boundary(CELL, cellid)) { // No limiting on boundary
189 
190  phi = 1.0;
191 
192  // Min and max vals of function (cell centered vals) among neighbors
194 
195  double minval = vals_[cellid];
196  double maxval = vals_[cellid];
197 
198  int nnbr = nbrids.size();
199  for (int i = 0; i < nnbr; ++i) {
200  minval = std::min(cellvalues[i], minval);
201  maxval = std::max(cellvalues[i], maxval);
202  }
203 
204  // Find the min and max of the reconstructed function in the cell
205  // Since the reconstruction is linear, this will occur at one of
206  // the nodes of the cell. So find the values of the reconstructed
207  // function at the nodes of the cell
208 
209  int dim = mesh_.space_dimension();
210 
211  double cellcenval = vals_[cellid];
212  std::vector<Point<D>> cellcoords;
213  mesh_.cell_get_coordinates(cellid, &cellcoords);
214 
215  for (auto coord : cellcoords) {
216  Vector<D> vec = coord-cellcenters[0];
217  double diff = dot(grad, vec);
218  double extremeval = (diff > 0.0) ? maxval : minval;
219  double phi_new = (diff == 0.0) ? 1 : (extremeval-cellcenval)/diff;
220  phi = std::min(phi_new, phi);
221  }
222  }
223 
224  // Limited gradient is phi*grad
225 
226  return phi*grad;
227 }
