72 template<long D>
73 Vector<D> ls_gradient(std::vector<Point<D>> const & coords,
74  std::vector<double> const & vals) {
75 
76  Point<D> coord0 = coords[0];
77 
78  double val0 = vals[0];
79 
80  // There are nvals but the first is the reference point where we
81  // are trying to compute the gradient; so the matrix sizes etc
82  // will only be nvals-1
83 
84  int nvals = vals.size();
85 
86  // Each row of A contains the components of the vector from
87  // coord0 to the candidate point being used in the Least Squares
88  // approximation (X_i-X_0).
89 
90  Matrix A(nvals-1, D);
91  for (int i = 0; i < nvals-1; ++i) {
92  for (int j = 0; j < D; ++j)
93  A[i][j] = coords[i+1][j]-coord0[j];
94  }
95 
96 
97  // A is a matrix of size nvals-1 by D (where D is the space
98  // dimension). So transpose(A)*A is D by D
99 
100  Matrix AT = A.transpose();
101 
102  Matrix ATA = AT*A;
103 
104  // Each entry/row of F contains the difference between the
105  // function value at the candidate point and the function value
106  // at the point where we are computing (f-f_0)
107 
108  std::vector<double> F(nvals-1);
109  for (int i = 0; i < nvals-1; ++i)
110  F[i] = vals[i+1]-val0;
111 
112  // F is a vector of nvals. So transpose(A)*F is vector of D
113  // (where D is the space dimension)
114 
115  Vector<D> ATF = Vector<D>(AT*F);
116 
117  // Inverse of ATA
118 
119  Matrix ATAinv = ATA.inverse();
120 
121  // Gradient of length D
122 
123  return ATAinv*ATF;
124 }

261 template<typename MeshType, typename StateType, long D>
262 Vector<D>
263 Limited_Gradient<MeshType, StateType, CELL, D>::operator() (int const cellid) {
264 
265  assert(D == mesh_.space_dimension());
266  assert(D == 2 || D == 3);
267  double phi = 1.0;
268  Vector<D> grad;
269 
270  std::vector<int> const & nbrids = cell_neighbors_[cellid];
271 
272  std::vector<Point<D>> cellcenters(nbrids.size()+1);
273  std::vector<double> cellvalues(nbrids.size()+1);
274 
275  mesh_.cell_centroid(cellid, &(cellcenters[0]));
276 
277  cellvalues[0] = vals_[cellid];
278 
279  int i = 1;
280  for (auto nbrcell : nbrids) {
281  mesh_.cell_centroid(nbrcell, &(cellcenters[i]));
282 
283  cellvalues[i] = vals_[nbrcell];
284  i++;
285  }
286 
287  grad = ls_gradient(cellcenters, cellvalues);
288 
289  // Limit the gradient to enforce monotonicity preservation
290 
291  if (limtype_ == BARTH_JESPERSEN) {
292 
293  phi = 1.0;
294 
295  // Min and max vals of function (cell centered vals) among neighbors
297 
298  double minval = vals_[cellid];
299  double maxval = vals_[cellid];
300 
301  int nnbr = nbrids.size();
302  for (int i = 0; i < nnbr; ++i) {
303  minval = std::min(cellvalues[i], minval);
304  maxval = std::max(cellvalues[i], maxval);
305  }
306 
307  // Find the min and max of the reconstructed function in the cell
308  // Since the reconstruction is linear, this will occur at one of
309  // the nodes of the cell. So find the values of the reconstructed
310  // function at the nodes of the cell
311 
312  int dim = mesh_.space_dimension();
313 
314  double cellcenval = vals_[cellid];
315  std::vector<Point<D>> cellcoords;
316  mesh_.cell_get_coordinates(cellid, &cellcoords);
317 
318  for (auto coord : cellcoords) {
319  Vector<D> vec = coord-cellcenters[0];
320  double diff = dot(grad, vec);
321  double extremeval = (diff > 0.0) ? maxval : minval;
322  double phi_new = (diff == 0.0) ? 1 : (extremeval-cellcenval)/diff;
323  phi = std::min(phi_new, phi);
324  }
325  }
326 
327  // Limited gradient is phi*grad
328 
329  return phi*grad;
330 }
