169 template<int D, typename MeshType, typename StateType>
170  Vector<D*(D+3)/2>
171 Limited_Quadfit<D, CELL, MeshType, StateType>::operator() (int const cellid) {
172 
173  assert(D == mesh_.space_dimension());
174  assert(D == 2 || D == 3);
175  double phi = 1.0;
176  Vector<D*(D+3)/2> qfit;
177  Vector<D*(D+3)/2> dvec;
178 
179  std::vector<int> const & nbrids = cell_neighbors_[cellid];
180 
181  std::vector<Point<D>> cellcenters(nbrids.size()+1);
182  std::vector<double> cellvalues(nbrids.size()+1);
183 
184  // get centroid and value for cellid at center of point cloud
185  mesh_.cell_centroid(cellid, &(cellcenters[0]));
186 
187  cellvalues[0] = vals_[cellid];
188 
189  int i = 1;
190  for (auto nbrcell : nbrids) {
191  mesh_.cell_centroid(nbrcell, &(cellcenters[i]));
192  cellvalues[i] = vals_[nbrcell];
193  i++;
194  }
195 
196  bool boundary_cell = mesh_.on_exterior_boundary(CELL, cellid);
197  qfit = ls_quadfit(cellcenters, cellvalues, boundary_cell);
198  // Limit the gradient to enforce monotonicity preservation
199 
200  if (limtype_ == BARTH_JESPERSEN && !boundary_cell) { // No limiting on boundary
201 
202  // Min and max vals of function (cell centered vals) among neighbors
204 
205  double minval = vals_[cellid];
206  double maxval = vals_[cellid];
207 
208  int nnbr = nbrids.size();
209  for (int i = 0; i < nnbr; ++i) {
210  minval = std::min(cellvalues[i], minval);
211  maxval = std::max(cellvalues[i], maxval);
212  }
213 
214  // Find the min and max of the reconstructed function in the cell
215  // Since the reconstruction is linear, this will occur at one of
216  // the nodes of the cell. So find the values of the reconstructed
217  // function at the nodes of the cell
218 
219  int dim = mesh_.space_dimension();
220 
221  double cellcenval = vals_[cellid];
222  std::vector<Point<D>> cellcoords;
223  mesh_.cell_get_coordinates(cellid, &cellcoords);
224 
225  for (auto coord : cellcoords) {
226  Vector<D> vec = coord-cellcenters[0];
227  //Vector<D*(D+3)/2> dvec;
228  for (int j = 0; j < D; ++j) {
229  dvec[j] = vec[j];
230  // Add the quadratic terms
231  for (int k = 0; k < j; ++k) {
232  dvec[j+k+D-1] = dvec[k]*dvec[j];
233  }
234  }
235  double diff = dot(qfit, dvec);
236  double extremeval = (diff > 0.0) ? maxval : minval;
237  double phi_new = (diff == 0.0) ? 1 : (extremeval-cellcenval)/diff;
238  phi = std::min(phi_new, phi);
239  }
240 
241  }
242 
243  // Limited quadfit is phi*fit
244  return phi*qfit;
245 }
