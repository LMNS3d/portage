258 template<int D,
259  typename SourceMeshType, typename TargetMeshType, typename StateType>
260 double Interpolate_3rdOrder<D, Entity_kind::CELL, SourceMeshType, TargetMeshType,
261  StateType>::operator()
262  (int const targetCellID, std::vector<Weights_t> const & sources_and_weights)
263  const {
264 
265  int nsrccells = sources_and_weights.size();
266  if (!nsrccells) {
267 #ifdef DEBUG
268  std::cerr << "WARNING: No source cells contribute to target cell." <<
269  std::endl;
270 #endif
271  return 0.0;
272  }
273 
274  double totalval = 0.0;
275 
276  // contribution of the source cell is its field value weighted by
277  // its "weight" (in this case, its 0th moment/area/volume)
278 
280 
281  for (int j = 0; j < nsrccells; ++j) {
282  int srccell = sources_and_weights[j].entityID;
283  // int N = D*(D+3)/2;
284  std::vector<double> xsect_weights = sources_and_weights[j].weights;
285  double xsect_volume = xsect_weights[0];
286 
287  double eps = 1e-30;
288  if (xsect_volume <= eps) continue; // no intersection
289 
290  Point<D> srccell_centroid;
291  source_mesh_.cell_centroid(srccell, &srccell_centroid);
292 
293  Point<D> xsect_centroid;
294  for (int i = 0; i < D; ++i)
295  xsect_centroid[i] = xsect_weights[1+i]/xsect_volume; // (1st moment)/vol
296 
297  Vector<D*(D+3)/2> quadfit = quadfits_[srccell];
298  Vector<D> vec = xsect_centroid - srccell_centroid;
299  Vector<D*(D+3)/2> dvec;
300  for (int j = 0; j < D; ++j) {
301  int j1 = D;
302  dvec[j] = vec[j];
303  // Add the quadratic terms
304  for (int k = 0; k <= j; ++k) {
305  dvec[j1] = dvec[k]*dvec[j];
306  j1 += 1;
307  //dvec[D+j+k] = dvec[k]*dvec[j];
308  }
309  }
310  double val = source_vals_[srccell] + dot(quadfit,dvec);
311  val *= xsect_volume;
312  totalval += val;
313  }
314 
315  // Normalize the value by sum of all the 0th weights (which is the
316  // same as the total volume of the source cell)
317 
318  totalval /= target_mesh_.cell_volume(targetCellID);
319 
320  return totalval;
321 }

445  if (!nsrcnodes) {
446 #ifdef DEBUG

448  std::endl;
449 #endif
450  return 0.0;
451  }
452 
453  double totalval = 0.0;
454 
455  // contribution of the source cell is its field value weighted by
456  // its "weight" (in this case, its 0th moment/area/volume)
457 
459 
460  for (int j = 0; j < nsrcnodes; ++j) {
461  int srcnode = sources_and_weights[j].entityID;
462  std::vector<double> xsect_weights = sources_and_weights[j].weights;
463  double xsect_volume = xsect_weights[0];
464 
465  double eps = 1e-30;
466  if (xsect_volume <= eps) continue; // no intersection
467 
468  // note: here we are getting the node coord, not the centroid of
469  // the dual cell
470  Point<D> srcnode_coord;
471  source_mesh_.node_get_coordinates(srcnode, &srcnode_coord);
472 
473  Point<D> xsect_centroid;
474  for (int i = 0; i < D; ++i)
475  // (1st moment)/(vol)
476  xsect_centroid[i] = xsect_weights[1+i]/xsect_volume;
477 
478  Vector<D*(D+3)/2> quadfit = quadfits_[srcnode];
479  Vector<D> vec = xsect_centroid - srcnode_coord;
480  Vector<D*(D+3)/2> dvec;
481  for (int j = 0; j < D; ++j) {
482  int j1 = D;
483  dvec[j] = vec[j];
484  // Add the quadratic terms
485  for (int k = 0; k <= j; ++k) {
486  dvec[j1] = dvec[k]*dvec[j];
487  j1 += 1;
488  // dvec[D+j+k] = dvec[k]*dvec[j];
489  }
490  }
491  double val = source_vals_[srcnode] + dot(quadfit,dvec);
492  val *= xsect_volume;
493  totalval += val;
494  }
495 
496  // Normalize the value by volume of the target dual cell
