261 template<int D,
262  typename SourceMeshType, typename TargetMeshType, typename StateType>
263 double Interpolate_3rdOrder<D, Entity_kind::CELL, SourceMeshType, TargetMeshType,
264  StateType>::operator()
265  (int const targetCellID, std::vector<Weights_t> const & sources_and_weights)
266  const {
267 
268  int nsrccells = sources_and_weights.size();
269  if (!nsrccells) {
270 #ifdef DEBUG
271  std::cerr << "WARNING: No source cells contribute to target cell." <<
272  std::endl;
273 #endif
274  return 0.0;
275  }
276 
277  double totalval = 0.0;
278 
279  // contribution of the source cell is its field value weighted by
280  // its "weight" (in this case, its 0th moment/area/volume)
281 
283 
284  double vol = target_mesh_.cell_volume(targetCellID);
285  for (int j = 0; j < nsrccells; ++j) {
286  int srccell = sources_and_weights[j].entityID;
287  // int N = D*(D+3)/2;
288  std::vector<double> xsect_weights = sources_and_weights[j].weights;
289  double xsect_volume = xsect_weights[0];
290 
291  if (xsect_volume/vol <= num_tols_.min_relative_volume)
292  continue; // no intersection
293 
294  Point<D> srccell_centroid;
295  source_mesh_.cell_centroid(srccell, &srccell_centroid);
296 
297  Point<D> xsect_centroid;
298  for (int i = 0; i < D; ++i)
299  xsect_centroid[i] = xsect_weights[1+i]/xsect_volume; // (1st moment)/vol
300 
301  Vector<D*(D+3)/2> quadfit = quadfits_[srccell];
302  Vector<D> vec = xsect_centroid - srccell_centroid;
303  Vector<D*(D+3)/2> dvec;
304  for (int j = 0; j < D; ++j) {
305  int j1 = D;
306  dvec[j] = vec[j];
307  // Add the quadratic terms
308  for (int k = 0; k <= j; ++k) {
309  dvec[j1] = dvec[k]*dvec[j];
310  j1 += 1;
311  //dvec[D+j+k] = dvec[k]*dvec[j];
312  }
313  }
314  double val = source_vals_[srccell] + dot(quadfit,dvec);
315  val *= xsect_volume;
316  totalval += val;
317  }
318 
319  // Normalize the value by sum of all the 0th weights (which is the
320  // same as the total volume of the source cell)
321 
322  totalval /= target_mesh_.cell_volume(targetCellID);
323 
324  return totalval;
325 }

445  StateType> :: operator()
446  (int const targetNodeID, std::vector<Weights_t> const & sources_and_weights)
447  const {
448 
449  int nsrcnodes = sources_and_weights.size();
450  if (!nsrcnodes) {
451 #ifdef DEBUG
452  std::cerr << "WARNING: No source nodes contribute to target node." <<
453  std::endl;
454 #endif
455  return 0.0;
456  }
457 
458  double totalval = 0.0;
459 
460  // contribution of the source cell is its field value weighted by
461  // its "weight" (in this case, its 0th moment/area/volume)
462 
464 
465  double vol = target_mesh_.dual_cell_volume(targetNodeID);
466  for (int j = 0; j < nsrcnodes; ++j) {
467  int srcnode = sources_and_weights[j].entityID;
468  std::vector<double> xsect_weights = sources_and_weights[j].weights;
469  double xsect_volume = xsect_weights[0];
470 
471  if (xsect_volume/vol <= num_tols_.min_relative_volume)
472  continue; // no intersection
473 
474  // note: here we are getting the node coord, not the centroid of
475  // the dual cell
476  Point<D> srcnode_coord;
477  source_mesh_.node_get_coordinates(srcnode, &srcnode_coord);
478 
479  Point<D> xsect_centroid;
480  for (int i = 0; i < D; ++i)
481  // (1st moment)/(vol)
482  xsect_centroid[i] = xsect_weights[1+i]/xsect_volume;
483 
484  Vector<D*(D+3)/2> quadfit = quadfits_[srcnode];
485  Vector<D> vec = xsect_centroid - srcnode_coord;
486  Vector<D*(D+3)/2> dvec;
487  for (int j = 0; j < D; ++j) {
488  int j1 = D;
489  dvec[j] = vec[j];
490  // Add the quadratic terms
491  for (int k = 0; k <= j; ++k) {
492  dvec[j1] = dvec[k]*dvec[j];
493  j1 += 1;
494  // dvec[D+j+k] = dvec[k]*dvec[j];
495  }
496  }
497  double val = source_vals_[srcnode] + dot(quadfit,dvec);
498  val *= xsect_volume;
499  totalval += val;
500  }
501 
502  // Normalize the value by volume of the target dual cell
503 
504  totalval /= target_mesh_.dual_cell_volume(targetNodeID);
505
