249 template <long D> inline bool
250 approxEq(const Point<D>& p1, const Point<D>& p2, double tol = 1.0e-8) {
251  // This uses the L_infty norm to avoid nearby subtractions
252  for (int i = 0; i < D; i++)
253  if (p1[i] < (p2[i] - tol) || p1[i] > (p2[i] + tol))
254  return false;
255  return true;
256 }
257 
258 template <long D> bool
259 operator<(const Point<D>& p1, const Point<D>& p2) {
260  if (approxEq(p1, p2))
261  return false;
262  else
263  for (int i = 0; i < D; ++i) {
264  if (p1[i] < p2[i])
265  return true;
266  else if (p2[i] < p1[i])
267  return false;
268  }
269  return false;
270 }
