81 Point<D> toPortagePoint(const JaliGeometry::Point& jp) {
82  Point<D> pp;
83  assert(jp.dim() == D);
84  for (int d = 0; d < D; ++d)
85  pp[d] = jp[d];
86  return pp;

217  void node_get_cell_adj_nodes(int const nodeid,
218  Entity_type const ptype,
219  std::vector<int> *adjnodes) const {
220  adjnodes->clear();
221 
222  Jali::Entity_ID_List nodecells;
223  jali_mesh_.node_get_cells(nodeid, (Jali::Entity_type) ptype, &nodecells);
224 
225  for (auto const& c : nodecells) {
226  Jali::Entity_ID_List cellnodes;
227  jali_mesh_.cell_get_nodes(c, &cellnodes);
228 
229  for (auto const& n : cellnodes) {
230  if (n == nodeid) continue;
231  if (std::find(adjnodes->begin(), adjnodes->end(), n) == adjnodes->end())
232  adjnodes->emplace_back(n);
233  }
234  }
235  }
