184  void node_get_cell_adj_nodes(int const nodeid,
185  Entity_type const ptype,
186  std::vector<int> *adjnodes) const {
187  adjnodes->clear();
188 
189  Jali::Entity_ID_List nodecells;
190  jali_mesh_.node_get_cells(nodeid, (Jali::Entity_type) ptype, &nodecells);
191 
192  for (auto const& c : nodecells) {
193  Jali::Entity_ID_List cellnodes;
194  jali_mesh_.cell_get_nodes(c, &cellnodes);
195 
196  for (auto const& n : cellnodes) {
197  if (n == nodeid) continue;
198  if (std::find(adjnodes->begin(), adjnodes->end(), n) == adjnodes->end())
199  adjnodes->emplace_back(n);
200  }
201  }
202  }
