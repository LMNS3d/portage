166  void make_index_maps() {
167 
168  // Global to local maps for cells and nodes
169  std::map<int, int> globalCellMap;
170  std::vector<int> cellUniqueRep;
171  cellUniqueRep.reserve(cellGlobalIds_.size());
172  for (unsigned int i=0; i<cellGlobalIds_.size(); ++i) {
173  auto itr = globalCellMap.find(cellGlobalIds_[i]);
174  if (itr == globalCellMap.end()) {
175  globalCellMap[cellGlobalIds_[i]] = i;
176  cellUniqueRep[i] = i;
177  }
178  else {
179  cellUniqueRep[i] = itr->second;
180  }
181  }
182 
183  std::map<int, int> globalNodeMap;
184  std::vector<int> nodeUniqueRep;
185  nodeUniqueRep.reserve(nodeGlobalIds_.size());
186  for (unsigned int i=0; i<nodeGlobalIds_.size(); ++i) {
187  auto itr = globalNodeMap.find(nodeGlobalIds_[i]);
188  if (itr == globalNodeMap.end()) {
189  globalNodeMap[nodeGlobalIds_[i]] = i;
190  nodeUniqueRep[i] = i;
191  }
192  else {
193  nodeUniqueRep[i] = itr->second;
194  }
195  }
196 
197  // Compute node and face offsets
198  if (dim_ == 2)
199  {
200  compute_offsets(cellNodeCounts_, &cellNodeOffsets_);
201  }
202  if (dim_ == 3)
203  {
204  compute_offsets(faceNodeCounts_, &faceNodeOffsets_);
205  compute_offsets(cellFaceCounts_, &cellFaceOffsets_);
206  }
207 
208  // Remove duplicate nodes
209  if (dim_ == 2)
210  {
211  for (unsigned int i=0; i<cellToNodeList_.size(); ++i)
212  cellToNodeList_[i] = nodeUniqueRep[cellToNodeList_[i]];
213  }
214  if (dim_ == 3)
215  {
216  for (unsigned int i=0; i<faceToNodeList_.size(); ++i)
217  faceToNodeList_[i] = nodeUniqueRep[faceToNodeList_[i]];
218  }
219 
220  // Compute cell-to-node adjacency lists (3D only)
221  if (dim_ == 3)
222  {
223  cellNodeCounts_.clear();
224  cellToNodeList_.clear();
225  cellNodeCounts_.reserve(cellFaceCounts_.size());
226  cellToNodeList_.reserve(cellFaceCounts_.size() * 4);
227  for (unsigned int c=0; c<cellFaceCounts_.size(); ++c) {
228  std::vector<int> cellfaces, dummy;
229  cell_get_faces_and_dirs(c, &cellfaces, &dummy);
230  std::set<int> cellnodes;
231 
232  for (auto const& f : cellfaces) {
233  std::vector<int> facenodes;
234  face_get_nodes(f, &facenodes);
235  cellnodes.insert(facenodes.begin(), facenodes.end());
236  }
237 
238  cellNodeCounts_.emplace_back(cellnodes.size());
239  cellToNodeList_.insert(cellToNodeList_.end(),
240  cellnodes.begin(), cellnodes.end());
241  }
242  compute_offsets(cellNodeCounts_, &cellNodeOffsets_);
243  }
244 
245  // Compute node-to-cell adjacency lists
246  int numNodes = nodeCoords_.size() / dim_;
247  std::vector<std::set<int>> nodeToCellTmp(numNodes);
248  for (unsigned int c=0; c<cellNodeCounts_.size(); ++c) {
249  int offset = cellNodeOffsets_[c];
250  int count = cellNodeCounts_[c];
251  for (unsigned int i=0; i<count; ++i) {
252  int n = cellToNodeList_[offset+i];
253  nodeToCellTmp[n].insert(cellUniqueRep[c]);
254  }
255  }
256  nodeCellCounts_.clear();
257  nodeToCellList_.clear();
258  nodeCellCounts_.reserve(numNodes);
259  nodeToCellList_.reserve(cellToNodeList_.size());
260  for (unsigned int n=0; n<numNodes; ++n) {
261  const std::set<int>& nodes = nodeToCellTmp[n];
262  nodeCellCounts_.emplace_back(nodes.size());
263  nodeToCellList_.insert(nodeToCellList_.end(), nodes.begin(), nodes.end());
264  }
265  compute_offsets(nodeCellCounts_, &nodeCellOffsets_);
266 
267  // Compute cell-to-face and face-to-node (2D only)
268  // This isn't quite accurate: some faces at rank boundaries
269  // may end up being "owned" by two ranks. But for what we are
270  // doing, I don't think this will make a difference.
271  if (dim_ == 2) {
272  cellToFaceList_.clear();
273  cellToFaceDirs_.clear();
274  faceToNodeList_.clear();
275  cellToFaceList_.reserve(cellNodeCounts_.size());
276  cellToFaceDirs_.reserve(cellNodeCounts_.size());
277  faceToNodeList_.reserve(cellToNodeList_.size()); // slight underestimate
278  std::map<std::pair<int, int>, int> nodeToFaceTmp;
279  int facecount = 0;
280  for (unsigned int c=0; c<cellNodeCounts_.size(); ++c) {
281  int offset = cellNodeOffsets_[c];
282  int count = cellNodeCounts_[c];
283  for (unsigned int i=0; i<count; ++i) {
284  int n0 = cellToNodeList_[offset+i];
285  int n1 = cellToNodeList_[offset+((i+1)%count)];
286  // put nodes in canonical order
287  int p0 = std::min(n0, n1);
288  int p1 = std::max(n0, n1);
289  auto npair = std::make_pair(p0, p1);
290  auto it = nodeToFaceTmp.find(npair);
291  int face;
292  if (it == nodeToFaceTmp.end()) {
293  // new face
294  face = facecount;
295  nodeToFaceTmp.insert(std::make_pair(npair, face));
296  faceToNodeList_.emplace_back(p0);
297  faceToNodeList_.emplace_back(p1);
298  ++facecount;
299  }
300  else {
301  // existing face
302  face = it->second;
303  }
304  cellToFaceList_.emplace_back(face);
305  cellToFaceDirs_.push_back(n0 == p0);
306  } // for i
307 
308  if (c == numOwnedCells_ - 1) numOwnedFaces_ = facecount;
309 
310  } // for c
311  } // if dim == 2
312 
313  // Delete adjacency lists for inactive cells
314  // (Note that this leaves unused spaces in the list; at some later
315  // time we could compress the list, but skip that for now)
316  for (unsigned int c=numOwnedCells_; c<cellNodeCounts_.size(); ++c) {
317  if (cellUniqueRep[c] != c) {
318  cellNodeCounts_[c] = 0;
319  }
320  }
321 
322  } // make_index_maps

594 template<size_t D>
595 void cell_radius(Wonton::Flat_Mesh_Wrapper<double> &wrapper,
596  int const cellid, double *radius) {
597  Point<D> centroid;
598  wrapper.cell_centroid<D>(cellid, &centroid);
599  std::vector<int> nodes;
600  wrapper.cell_get_nodes(cellid, &nodes);
601  Point<D> arm, node;
602  *radius = 0.;
603  for (int i=0; i<nodes.size(); i++) {
604  wrapper.node_get_coordinates(nodes[i], &node);
605  for (int j=0; j<D; j++) arm[j] = node[j]-centroid[j];
606  double distance = 0.0;
607  for (int j=0; j<D; j++) distance += arm[j]*arm[j];
608  distance = sqrt(distance);
609  if (distance > *radius) *radius = distance;
610  }
611 }
