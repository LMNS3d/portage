165  void make_index_maps() {
166 
167  // Global to local maps for cells and nodes
168  std::map<int, int> globalCellMap;
169  std::vector<int> cellUniqueRep;
170  cellUniqueRep.reserve(cellGlobalIds_.size());
171  for (unsigned int i=0; i<cellGlobalIds_.size(); ++i) {
172  auto itr = globalCellMap.find(cellGlobalIds_[i]);
173  if (itr == globalCellMap.end()) {
174  globalCellMap[cellGlobalIds_[i]] = i;
175  cellUniqueRep[i] = i;
176  }
177  else {
178  cellUniqueRep[i] = itr->second;
179  }
180  }
181 
182  std::map<int, int> globalNodeMap;
183  std::vector<int> nodeUniqueRep;
184  nodeUniqueRep.reserve(nodeGlobalIds_.size());
185  for (unsigned int i=0; i<nodeGlobalIds_.size(); ++i) {
186  auto itr = globalNodeMap.find(nodeGlobalIds_[i]);
187  if (itr == globalNodeMap.end()) {
188  globalNodeMap[nodeGlobalIds_[i]] = i;
189  nodeUniqueRep[i] = i;
190  }
191  else {
192  nodeUniqueRep[i] = itr->second;
193  }
194  }
195 
196  // Compute node and face offsets
197  if (dim_ == 2)
198  {
199  compute_offsets(cellNodeCounts_, &cellNodeOffsets_);
200  }
201  if (dim_ == 3)
202  {
203  compute_offsets(faceNodeCounts_, &faceNodeOffsets_);
204  compute_offsets(cellFaceCounts_, &cellFaceOffsets_);
205  }
206 
207  // Remove duplicate nodes
208  if (dim_ == 2)
209  {
210  for (unsigned int i=0; i<cellToNodeList_.size(); ++i)
211  cellToNodeList_[i] = nodeUniqueRep[cellToNodeList_[i]];
212  }
213  if (dim_ == 3)
214  {
215  for (unsigned int i=0; i<faceToNodeList_.size(); ++i)
216  faceToNodeList_[i] = nodeUniqueRep[faceToNodeList_[i]];
217  }
218 
219  // Compute cell-to-node adjacency lists (3D only)
220  if (dim_ == 3)
221  {
222  cellNodeCounts_.clear();
223  cellToNodeList_.clear();
224  cellNodeCounts_.reserve(cellFaceCounts_.size());
225  cellToNodeList_.reserve(cellFaceCounts_.size() * 4);
226  for (unsigned int c=0; c<cellFaceCounts_.size(); ++c) {
227  std::vector<int> cellfaces, dummy;
228  cell_get_faces_and_dirs(c, &cellfaces, &dummy);
229  std::set<int> cellnodes;
230 
231  for (auto const& f : cellfaces) {
232  std::vector<int> facenodes;
233  face_get_nodes(f, &facenodes);
234  cellnodes.insert(facenodes.begin(), facenodes.end());
235  }
236 
237  cellNodeCounts_.emplace_back(cellnodes.size());
238  cellToNodeList_.insert(cellToNodeList_.end(),
239  cellnodes.begin(), cellnodes.end());
240  }
241  compute_offsets(cellNodeCounts_, &cellNodeOffsets_);
242  }
243 
244  // Compute node-to-cell adjacency lists
245  int numNodes = nodeCoords_.size() / dim_;
246  std::vector<std::set<int>> nodeToCellTmp(numNodes);
247  for (unsigned int c=0; c<cellNodeCounts_.size(); ++c) {
248  int offset = cellNodeOffsets_[c];
249  int count = cellNodeCounts_[c];
250  for (unsigned int i=0; i<count; ++i) {
251  int n = cellToNodeList_[offset+i];
252  nodeToCellTmp[n].insert(cellUniqueRep[c]);
253  }
254  }
255  nodeCellCounts_.clear();
256  nodeToCellList_.clear();
257  nodeCellCounts_.reserve(numNodes);
258  nodeToCellList_.reserve(cellToNodeList_.size());
259  for (unsigned int n=0; n<numNodes; ++n) {
260  const std::set<int>& nodes = nodeToCellTmp[n];
261  nodeCellCounts_.emplace_back(nodes.size());
262  nodeToCellList_.insert(nodeToCellList_.end(), nodes.begin(), nodes.end());
263  }
264  compute_offsets(nodeCellCounts_, &nodeCellOffsets_);
265 
266  // Compute cell-to-face and face-to-node (2D only)
267  // This isn't quite accurate: some faces at rank boundaries
268  // may end up being "owned" by two ranks. But for what we are
269  // doing, I don't think this will make a difference.
270  if (dim_ == 2) {
271  cellToFaceList_.clear();
272  cellToFaceDirs_.clear();
273  faceToNodeList_.clear();
274  cellToFaceList_.reserve(cellNodeCounts_.size());
275  cellToFaceDirs_.reserve(cellNodeCounts_.size());
276  faceToNodeList_.reserve(cellToNodeList_.size()); // slight underestimate
277  std::map<std::pair<int, int>, int> nodeToFaceTmp;
278  int facecount = 0;
279  for (unsigned int c=0; c<cellNodeCounts_.size(); ++c) {
280  int offset = cellNodeOffsets_[c];
281  int count = cellNodeCounts_[c];
282  for (unsigned int i=0; i<count; ++i) {
283  int n0 = cellToNodeList_[offset+i];
284  int n1 = cellToNodeList_[offset+((i+1)%count)];
285  // put nodes in canonical order
286  int p0 = std::min(n0, n1);
287  int p1 = std::max(n0, n1);
288  auto npair = std::make_pair(p0, p1);
289  auto it = nodeToFaceTmp.find(npair);
290  int face;
291  if (it == nodeToFaceTmp.end()) {
292  // new face
293  face = facecount;
294  nodeToFaceTmp.insert(std::make_pair(npair, face));
295  faceToNodeList_.emplace_back(p0);
296  faceToNodeList_.emplace_back(p1);
297  ++facecount;
298  }
299  else {
300  // existing face
301  face = it->second;
302  }
303  cellToFaceList_.emplace_back(face);
304  cellToFaceDirs_.push_back(n0 == p0);
305  } // for i
306 
307  if (c == numOwnedCells_ - 1) numOwnedFaces_ = facecount;
308 
309  } // for c
310  } // if dim == 2
311 
312  // Delete adjacency lists for inactive cells
313  // (Note that this leaves unused spaces in the list; at some later
314  // time we could compress the list, but skip that for now)
315  for (unsigned int c=numOwnedCells_; c<cellNodeCounts_.size(); ++c) {
316  if (cellUniqueRep[c] != c) {
317  cellNodeCounts_[c] = 0;
318  }
319  }
320 
321  } // make_index_maps

545 template<size_t D>
546 void cell_radius(Portage::Flat_Mesh_Wrapper<double> &wrapper,
547  int const cellid, double *radius) {
548  Point<D> centroid;
549  wrapper.cell_centroid<D>(cellid, &centroid);
550  std::vector<int> nodes;
551  wrapper.cell_get_nodes(cellid, &nodes);
552  Point<D> arm, node;
553  *radius = 0.;
554  for (int i=0; i<nodes.size(); i++) {
555  wrapper.node_get_coordinates(nodes[i], &node);
556  for (int j=0; j<D; j++) arm[j] = node[j]-centroid[j];
557  double distance = 0.0;
558  for (int j=0; j<D; j++) distance += arm[j]*arm[j];
559  distance = sqrt(distance);
560  if (distance > *radius) *radius = distance;
561  }
562 }
