82  template<class Mesh_Wrapper>
83  void initialize(Mesh_Wrapper& input)
84  {
85  dim_ = input.space_dimension();
86  int numCells = input.num_owned_cells() + input.num_ghost_cells();
87  numOwnedCells_ = input.num_owned_cells();
88  int numNodes = input.num_owned_nodes() + input.num_ghost_nodes();
89  numOwnedNodes_ = input.num_owned_nodes();
90  int numFaces = -1;
91  if (dim_ == 3)
92  {
93  numFaces = input.num_owned_faces() + input.num_ghost_faces();
94  numOwnedFaces_ = input.num_owned_faces();
95  }
96  nodeCoords_.clear();
97  cellNodeCounts_.clear();
98  cellToNodeList_.clear();
99  cellFaceCounts_.clear();
100  cellToFaceList_.clear();
101  faceNodeCounts_.clear();
102  faceToNodeList_.clear();
103  cellGlobalIds_.clear();
104  nodeCoords_.reserve(numNodes*dim_);
105  cellNodeCounts_.reserve(numCells);
106  // reserve (dim_+1) nodes per cell (lower bound)
107  cellToNodeList_.reserve(numCells*(dim_+1));
108  if (dim_ == 3)
109  {
110  cellFaceCounts_.reserve(numCells);
111  // reserve (dim_+1) faces per cell (lower bound)
112  cellToFaceList_.reserve(numCells*(dim_+1));
113  faceNodeCounts_.reserve(numFaces);
114  // reserve dim_ nodes per face (lower bound)
115  faceToNodeList_.reserve(numFaces*dim_);
116  }
117  cellGlobalIds_.reserve(numCells);
118 
119  for (unsigned int c=0; c<numCells; c++)
120  {
121  cellGlobalIds_.push_back(input.get_global_id(c, Entity_kind::CELL));
122 
123  std::vector<int> cellNodes;
124  input.cell_get_nodes(c, &cellNodes);
125  int cellNumNodes = cellNodes.size();
126  cellNodeCounts_.push_back(cellNumNodes);
127  cellToNodeList_.insert(cellToNodeList_.end(),
128  cellNodes.begin(), cellNodes.end());
129 
130 
131  std::vector<int> cellNeighbors;
132  input.cell_get_node_adj_cells(c, ALL, &(cellNeighbors));
133  cellNeighborCounts_.push_back(cellNeighbors.size());
134  for (unsigned int j=0; j<cellNeighbors.size(); j++)
135  neighbors_.push_back(
136  input.get_global_id(cellNeighbors[j], Entity_kind::CELL));
137  }
138 
139  if (dim_ == 3)
140  {
141  for (unsigned int c=0; c<numCells; ++c)
142  {
143  std::vector<int> cellFaces, cfDirs;
144  input.cell_get_faces_and_dirs(c, &cellFaces, &cfDirs);
145  int cellNumFaces = cellFaces.size();
146  cellFaceCounts_.push_back(cellNumFaces);
147  cellToFaceList_.insert(cellToFaceList_.end(),
148  cellFaces.begin(), cellFaces.end());
149  for (unsigned int j=0; j<cellNumFaces; ++j)
150  cellToFaceDirs_.push_back(cfDirs[j] >= 0);
151  }
152 
153  for (unsigned int f=0; f<numFaces; ++f)
154  {
155  std::vector<int> faceNodes;
156  input.face_get_nodes(f, &faceNodes);
157  int faceNumNodes = faceNodes.size();
158  faceNodeCounts_.push_back(faceNumNodes);
159  faceToNodeList_.insert(faceToNodeList_.end(),
160  faceNodes.begin(), faceNodes.end());
161  }
162  } // if dim_ == 3
163 
164  // ugly hack, since dim_ is not known at compile time
165  if (dim_ == 3) {
166  for (unsigned int n=0; n<numNodes; ++n) {
167  Portage::Point<3> nodeCoord;
168  input.node_get_coordinates(n, &nodeCoord);
169  for (unsigned int j=0; j<3; ++j)
170  nodeCoords_.push_back(nodeCoord[j]);
171  }
172  }
173  else if (dim_ == 2) {
174  for (unsigned int n=0; n<numNodes; ++n) {
175  Portage::Point<2> nodeCoord;
176  input.node_get_coordinates(n, &nodeCoord);
177  for (unsigned int j=0; j<2; ++j)
178  nodeCoords_.push_back(nodeCoord[j]);
179  }
180  }
181 
182  finish_init();
183  }

204  void make_index_maps() {
205 
206  // Global to local map
207  globalCellMap_.clear();
208  for (unsigned int i=0; i<cellGlobalIds_.size(); i++)
209  if (globalCellMap_.find(cellGlobalIds_[i]) == globalCellMap_.end())
210  globalCellMap_[cellGlobalIds_[i]] = i;
211 
212  // Node, face and neighbor offsets
213  compute_offsets(cellNodeCounts_, &cellNodeOffsets_);
214  if (dim_ == 3)
215  {
216  compute_offsets(faceNodeCounts_, &faceNodeOffsets_);
217  compute_offsets(cellFaceCounts_, &cellFaceOffsets_);
218  }
219  compute_offsets(cellNeighborCounts_, &cellNeighborOffsets_);
220  }

223  int compute_offsets(const std::vector<int>& counts,
224  std::vector<int>* offsets) {
225  offsets->resize(counts.size());
226  (*offsets)[0] = 0;
227  std::partial_sum(counts.begin(),
228  counts.end()-1,
229  offsets->begin()+1);
230  }

289  void cell_get_faces_and_dirs(int const cellid, std::vector<int> *cfaces,
290  std::vector<int> *cfdirs) const {
291  if (dim_ == 2)
292  {
293  int count = cellNodeCounts_[cellid];
294  int offset = cellNodeOffsets_[cellid];
295  cfaces->resize(count);
296  std::iota(cfaces->begin(), cfaces->end(), offset);
297  cfdirs->assign(count, 1);
298  }
299  else
300  {
301  int count = cellFaceCounts_[cellid];
302  int offset = cellFaceOffsets_[cellid];
303  cfaces->assign(&cellToFaceList_[offset],
304  &cellToFaceList_[offset+count]);
305  cfdirs->clear();
306  for (unsigned int j=0; j<count; ++j)
307  cfdirs->push_back(cellToFaceDirs_[offset + j] ? 1 : -1);
308  }
309  }

312  void cell_get_nodes(int const cellid, std::vector<int> *nodes) const {
313  int offset = cellNodeOffsets_[cellid];
314  int count = cellNodeCounts_[cellid];
315  nodes->assign(&cellToNodeList_[offset],
316  &cellToNodeList_[offset+count]);

320  void face_get_nodes(int const faceid, std::vector<int> *fnodes) const {
321  if (dim_ == 2)
322  {
323  auto itr = std::upper_bound(cellNodeOffsets_.begin(),
324  cellNodeOffsets_.end(), faceid);
325  int cellid = itr - cellNodeOffsets_.begin() - 1;
326  int offset = cellNodeOffsets_[cellid];
327  int count = cellNodeCounts_[cellid];
328  int fid = faceid - offset;
329  fnodes->resize(2);
330  (*fnodes)[0] = cellToNodeList_[faceid];
331  (*fnodes)[1] = cellToNodeList_[offset + ((fid + 1) % count)];
332  }
333  else
334  {
335  int offset = faceNodeOffsets_[faceid];
336  int count = faceNodeCounts_[faceid];
337  fnodes->assign(&faceToNodeList_[offset],
338  &faceToNodeList_[offset+count]);
339  }
340  }
