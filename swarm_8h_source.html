140  unsigned long seed;
141  auto clock = std::chrono::high_resolution_clock::now();
142  auto clockdur = clock.time_since_epoch();
143  auto clockmicro = std::chrono::duration_cast<std::chrono::microseconds>(clockdur);

149 // Factories for making swarms in 1/2/3 dimensions with random or uniform
150 // distribution of particles. Reason we are having to return a
151 // shared_ptr to the Swarm is because of how its used in the
152 // DriverTest
153 
154 std::shared_ptr<Swarm<1>> SwarmFactory(double xmin, double xmax,
155  unsigned int nparticles,
156  unsigned int distribution,
157  unsigned int rand_seed=0) {
158 
159  shared_ptr<typename Swarm<1>::PointVec> pts_sp = std::make_shared<typename Swarm<1>::PointVec>(nparticles);
160  typename Swarm<1>::PointVec &pts(*pts_sp);
161  drand48_data rand_state;
162  if (distribution == 0 or distribution == 2) { // random distribution of particles
163  long int seed;
164  if (rand_seed == 0) {
165  seed = get_seed();
166  } else {
167  seed = rand_seed;
168  }
169  srand48_r(seed,&rand_state);
170  }
171  if (distribution == 0) { // random distribution of particles
172  for (size_t i=0; i<nparticles; i++) {
173  // for some reason intel does not like these two lines combined when using thrust, i.e. pts[i][0]
174  Point<1> pt=pts[i];
175  double rnum; drand48_r(&rand_state, &rnum);
176  pt[0] = xmin + (xmax-xmin)*rnum;
177  pts[i] = pt;
178  }
179  } else { // regular distribution
180  double h = (xmax-xmin)/(nparticles-1);
181  for (size_t i=0; i < nparticles; i++) {
182  Point<1> pt=pts[i];
183  pt[0] = xmin + i*h;
184  pts[i] = pt;
185  }
186 
187  if (distribution == 2) { // perturbed regular
188  for (size_t i=0; i < nparticles; i++) {
189  Point<1> pt=pts[i];
190  double rnum; drand48_r(&rand_state, &rnum);
191  pt[0] += 0.25*h*(2*rnum-1.0);
192  pt[0] = fmax(xmin,fmin(xmax,pt[0]));
193  pts[i] = pt;
194  }
195  }
196  }
197 
198  std::shared_ptr<Swarm<1>> swarm = make_shared<Swarm<1>>(pts_sp);
199  return swarm;
200 }

209  auto pts_sp = std::make_shared<typename Swarm<2>::PointVec>(nparticles);
210  typename Swarm<2>::PointVec &pts(*pts_sp);
211  drand48_data rand_state;
212  if (distribution == 0 or distribution == 2) { // random distribution of particles
213  long int seed;
214  if (rand_seed == 0) {
215  seed = get_seed();
216  std::cout << "seed is " << seed << std::endl;
217  } else {
218  seed = rand_seed;
219  }
220  srand48_r(seed,&rand_state);
221  }
222  if (distribution == 0) {
223  for (size_t i = 0; i < nparticles; i++) {
224  Point<2> pt=pts[i];
225  double rnum;
226  drand48_r(&rand_state, &rnum);
227  pt[0] = xmin + (xmax-xmin)*rnum;
228  drand48_r(&rand_state, &rnum);
229  pt[1] = ymin + (ymax-ymin)*rnum;
230  pts[i] = pt;
231  }
232  } else {
233  int npdim = sqrt(nparticles);
234  double hx = (xmax-xmin)/(npdim-1);
235  double hy = (ymax-ymin)/(npdim-1);
236  int n = 0;
237  for (size_t i = 0; i < npdim; i++) {
238  for (size_t j = 0; j < npdim; j++) {
239  Point<2> pt=pts[n];
240  pt[0] = xmin + i*hx;
241  pt[1] = ymin + j*hy;
242  pts[n] = pt;
243  n++;
244  }
245  }
246  if (distribution == 2) {
247  for (size_t i = 0; i < nparticles; i++) {
248  Point<2> pt=pts[i];
249  double rnum;
250  drand48_r(&rand_state, &rnum);
251  pt[0] += 0.25*hx*(2*rnum-1.0);
252  drand48_r(&rand_state, &rnum);
253  pt[1] += 0.25*hy*(2*rnum-1.0);
254  pt[0] = fmax(xmin,fmin(xmax,pt[0]));
255  pt[1] = fmax(ymin,fmin(ymax,pt[1]));
256  pts[i] = pt;
257  }
258  }
259  }
260 
261  std::shared_ptr<Swarm<2>> swarm = make_shared<Swarm<2>>(pts_sp);
262  return swarm;

348 template<size_t dim, class MeshWrapper>
349 std::shared_ptr<Swarm<dim>> SwarmFactory(MeshWrapper &wrapper, Portage::Entity_kind entity)
350 {
351  size_t npoints_owned;
352  typename Swarm<dim>::PointVecPtr points;
353  if (entity != Entity_kind::NODE and entity != Entity_kind::CELL) {
354  throw(std::runtime_error("only nodes and cells allowed"));
355  }
356 
357  if (entity == Entity_kind::NODE) {
358  npoints_owned = wrapper.num_owned_nodes();
359  points = make_shared<typename Swarm<dim>::PointVec>(npoints_owned);
360  Point<dim> node;
361  for (size_t i=0; i<npoints_owned; i++) {
362  wrapper.node_get_coordinates(i, &node);
363  (*points)[i] = node;
364  }
365  } else if (entity == Entity_kind::CELL) {
366  npoints_owned = wrapper.num_owned_cells();
367  points = make_shared<typename Swarm<dim>::PointVec>(npoints_owned);
368  Point<dim> centroid;
369  for (size_t i=0; i<npoints_owned; i++) {
370  wrapper.cell_centroid(i, &centroid);
371  (*points)[i] = centroid;
372  }
373  }
374 
375  std::shared_ptr<Swarm<dim>> result = make_shared<Swarm<dim>>(points);
376  return result;
377 }
