61  Swarm(Portage::Flat_Mesh_Wrapper<double> &wrapper, Portage::Entity_kind entity)
62  : points_(NULL), npoints_(0)
63  {
64  if (entity != NODE and entity != CELL) {
65  throw(std::runtime_error("only nodes and cells allowed"));
66  }
67 
68  if (entity == NODE) {
69  npoints_ = wrapper.num_owned_nodes();
70  points_ = make_shared<vector<Point<dim>>>(npoints_);
71  Point<dim> node;
72  for (size_t i=0; i<npoints_; i++) {
73  wrapper.node_get_coordinates(i, &node);
74  (*points_)[i] = node;
75  }
76  } else if (entity == CELL) {
77  npoints_ = wrapper.num_owned_cells();
78  points_ = make_shared<vector<Point<dim>>>(npoints_);
79  Point<dim> centroid;
80  for (size_t i=0; i<npoints_; i++) {
81  wrapper.cell_centroid<dim>(i, &centroid);
82  (*points_)[i] = centroid;
83  }
84  }

159 // Factory for making swarms in 1 dimensions with random or uniform
160 // distribution of particles. Reason we are having to return a
161 // shared_ptr to the Swarm is because of how its used in the
162 // DriverTest
163 
164 std::shared_ptr<Swarm<1>> SwarmFactory(double xmin, double xmax,
165  unsigned int nparticles,
166  unsigned int distribution) {
167 
168  auto pts = std::make_shared<typename Swarm<1>::PointVec>(nparticles);
169  if (distribution == 0) { // random distribution of particles
170  srand(time(NULL));
171  unsigned int rand_state;
172  for (size_t i = 0; i < nparticles; i++)
173  (*pts)[i][0] = xmin +
174  (xmax-xmin)*(static_cast<double>(rand_r(&rand_state))/RAND_MAX);
175  } else {
176  double h = (xmax-xmin)/(nparticles-1);
177  for (size_t i = 0; i < nparticles; i++)
178  (*pts)[i][0] = xmin + i*h;
179 
180  if (distribution == 2) {
181  srand(time(NULL));
182  unsigned int rand_state;
183  double h = (xmax-xmin)/(nparticles-1);
184  for (size_t i = 0; i < nparticles; i++)
185  (*pts)[i][0] +=
186  0.25*h*((2*static_cast<double>(rand_r(&rand_state))/RAND_MAX)-1.0);
187  }
188  }
189 
190  std::shared_ptr<Swarm<1>> swarm = make_shared<Swarm<1>>(pts);
191  return swarm;
192 }

197 std::shared_ptr<Swarm<2>> SwarmFactory(double xmin, double ymin,
198  double xmax, double ymax,
199  unsigned int nparticles,
200  unsigned int distribution) {
201 
202  auto pts = std::make_shared<typename Swarm<2>::PointVec>(nparticles);
203  if (distribution == 0) { // random distribution of particles
204  srand(time(NULL));
205  unsigned int rand_state;
206  for (size_t i = 0; i < nparticles; i++) {
207  (*pts)[i][0] = xmin +
208  (xmax-xmin)*(static_cast<double>(rand_r(&rand_state))/RAND_MAX);
209  (*pts)[i][1] = ymin +
210  (ymax-ymin)*(static_cast<double>(rand_r(&rand_state))/RAND_MAX);
211  }
212  } else {
213  int npdim = sqrt(nparticles);
214  if (npdim*npdim != nparticles) {
215  std::cerr << "Requested number of particles not a perfect square\n";
216  std::cerr << "Generating only " << npdim << "particles in each dimension\n";
217  }
218  double hx = (xmax-xmin)/(npdim-1);
219  double hy = (ymax-ymin)/(npdim-1);
220  int n = 0;
221  for (size_t i = 0; i < npdim; i++) {
222  for (size_t j = 0; j < npdim; j++) {
223  (*pts)[n][0] = xmin + i*hx;
224  (*pts)[n][1] = ymin + j*hy;
225  n++;
226  }
227  }
228  if (distribution == 2) {
229  srand(time(NULL));
230  unsigned int rand_state;
231  for (size_t i = 0; i < nparticles; i++) {
232  (*pts)[i][0] +=
233  0.25*hx*((2*static_cast<double>(rand_r(&rand_state))/RAND_MAX)-1.0);
234  (*pts)[i][1] +=
235  0.25*hy*((2*static_cast<double>(rand_r(&rand_state))/RAND_MAX)-1.0);
236  }
237  }
238  }
239 
240  std::shared_ptr<Swarm<2>> swarm = make_shared<Swarm<2>>(pts);
241  return swarm;
