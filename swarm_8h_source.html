136 // Factories for making swarms in 1/2/3 dimensions with random or uniform
137 // distribution of particles. Reason we are having to return a
138 // shared_ptr to the Swarm is because of how its used in the

141 std::shared_ptr<Swarm<1>> SwarmFactory(double xmin, double xmax,
142  unsigned int nparticles,
143  unsigned int distribution) {
144 
145  shared_ptr<typename Swarm<1>::PointVec> pts_sp = std::make_shared<typename Swarm<1>::PointVec>(nparticles);
146  typename Swarm<1>::PointVec &pts(*pts_sp);
147  if (distribution == 0) { // random distribution of particles
148  srand(time(NULL));
149  unsigned int rand_state;
150  for (size_t i=0; i<nparticles; i++) {
151  // for some reason intel does not like these two lines combined when using thrust, i.e. pts[i][0]
152  Point<1> pt=pts[i];
153  pt[0] = xmin + (xmax-xmin)*(static_cast<double>(rand_r(&rand_state))/RAND_MAX);
154  pts[i] = pt;
155  }
156  } else { // regular distribution
157  double h = (xmax-xmin)/(nparticles-1);
158  for (size_t i=0; i < nparticles; i++) {
159  Point<1> pt=pts[i];
160  pt[0] = xmin + i*h;
161  pts[i] = pt;
162  }
163 
164  if (distribution == 2) { // perturbed regular
165  srand(time(NULL));
166  unsigned int rand_state;
167  for (size_t i=0; i < nparticles; i++) {
168  Point<1> pt=pts[i];
169  pt[0] += 0.25*h*((2*static_cast<double>(rand_r(&rand_state))/RAND_MAX)-1.0);
170  pt[0] = fmax(xmin,fmin(xmax,pt[0]));
171  pts[i] = pt;
172  }
173  }
174  }
175 
176  std::shared_ptr<Swarm<1>> swarm = make_shared<Swarm<1>>(pts_sp);
177  return swarm;
178 }

181 std::shared_ptr<Swarm<2>> SwarmFactory(double xmin, double ymin,
182  double xmax, double ymax,
183  unsigned int nparticles,
184  unsigned int distribution) {
185 
186  auto pts_sp = std::make_shared<typename Swarm<2>::PointVec>(nparticles);
187  typename Swarm<2>::PointVec &pts(*pts_sp);
188  if (distribution == 0) { // random distribution of particles
189  srand(time(NULL));
190  unsigned int rand_state;
191  for (size_t i = 0; i < nparticles; i++) {
192  Point<2> pt=pts[i];
193  pt[0] = xmin + (xmax-xmin)*(static_cast<double>(rand_r(&rand_state))/RAND_MAX);
194  pt[1] = ymin + (ymax-ymin)*(static_cast<double>(rand_r(&rand_state))/RAND_MAX);
195  pts[i] = pt;
196  }
197  } else {
198  int npdim = sqrt(nparticles);
199  if (npdim*npdim != nparticles) {
200  std::cerr << "Requested number of particles not a perfect square\n";
201  std::cerr << "Generating only " << npdim << "particles in each dimension\n";
202  }
203  double hx = (xmax-xmin)/(npdim-1);
204  double hy = (ymax-ymin)/(npdim-1);
205  int n = 0;
206  for (size_t i = 0; i < npdim; i++) {
207  for (size_t j = 0; j < npdim; j++) {
208  Point<2> pt=pts[n];
209  pt[0] = xmin + i*hx;
210  pt[1] = ymin + j*hy;
211  pts[n] = pt;
212  n++;
213  }
214  }
215  if (distribution == 2) {
216  srand(time(NULL));
217  unsigned int rand_state;
218  for (size_t i = 0; i < nparticles; i++) {
219  Point<2> pt=pts[i];
220  pt[0] += 0.25*hx*((2*static_cast<double>(rand_r(&rand_state))/RAND_MAX)-1.0);
221  pt[1] += 0.25*hy*((2*static_cast<double>(rand_r(&rand_state))/RAND_MAX)-1.0);
222  pt[0] = fmax(xmin,fmin(xmax,pt[0]));
223  pt[1] = fmax(ymin,fmin(ymax,pt[1]));
224  pts[i] = pt;
225  }
226  }
227  }
228 
229  std::shared_ptr<Swarm<2>> swarm = make_shared<Swarm<2>>(pts_sp);
230  return swarm;
231 }

301 template<size_t dim, class MeshWrapper>
302 std::shared_ptr<Swarm<dim>> SwarmFactory(MeshWrapper &wrapper, Portage::Entity_kind entity)
303 {
304  size_t npoints_owned;
305  typename Swarm<dim>::PointVecPtr points;
306  if (entity != Entity_kind::NODE and entity != Entity_kind::CELL) {
307  throw(std::runtime_error("only nodes and cells allowed"));
308  }
309 
310  if (entity == Entity_kind::NODE) {
311  npoints_owned = wrapper.num_owned_nodes();
312  points = make_shared<typename Swarm<dim>::PointVec>(npoints_owned);
313  Point<dim> node;
314  for (size_t i=0; i<npoints_owned; i++) {
315  wrapper.node_get_coordinates(i, &node);
316  (*points)[i] = node;
317  }
318  } else if (entity == Entity_kind::CELL) {
319  npoints_owned = wrapper.num_owned_cells();
320  points = make_shared<typename Swarm<dim>::PointVec>(npoints_owned);
321  Point<dim> centroid;
322  for (size_t i=0; i<npoints_owned; i++) {
323  wrapper.cell_centroid(i, &centroid);
324  (*points)[i] = centroid;
325  }
326  }
327 
328  std::shared_ptr<Swarm<dim>> result = make_shared<Swarm<dim>>(points);
329  return result;
330 }
