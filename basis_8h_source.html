129 template<Type type, size_t dim>
130 array<double, Traits<type, dim>::function_size>
131 shift(Wonton::Point<dim> x, Wonton::Point<dim> y){
132  auto ijet = inverse_jet<type,dim>(x);
133  auto by = function<type,dim>(y);
134  array<double, Traits<type, dim>::function_size> r{0.0};
135 
136  size_t n = Traits<type, dim>::function_size;
137  for (size_t i=0; i<n; i++) r[i] = 0.;
138  for (size_t i=0; i<n; i++) for (size_t j=0; j<n; j++) {
139  r[i] += ijet[i][j]*by[j];
140  }
141 
142  return r;
143 }
