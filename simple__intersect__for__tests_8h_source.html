29 bool intersect_boxes(JaliGeometry::Point min1, JaliGeometry::Point max1,
30  JaliGeometry::Point min2, JaliGeometry::Point max2,
31  std::vector<double> *xsect_moments) {
32 
33  assert(min1.dim() == max1.dim());
34  assert(min1.dim() == min2.dim());
35  assert(min1.dim() == max2.dim());
36 
37  int dim = min1.dim();
38  JaliGeometry::Point intmin(dim), intmax(dim);
39 
40  for (int d = 0; d < dim; ++d) {
41  // check for non-intersection in this dimension
42 
43  if (min1[d] > max2[d]) return false;
44  if (min2[d] > max1[d]) return false;
45 
46  // sort the min max vals in this dimension
47  double val[4];
48  val[0] = min1[d]; val[1] = max1[d]; val[2] = min2[d]; val[3] = max2[d];
49  for (int i = 0; i < 3; i++)
50  for (int j = i+1; j < 4; j++)
51  if (val[i] > val[j]) {
52  double tmp = val[i];
53  val[i] = val[j];
54  val[j] = tmp;
55  }
56 
57  // pick the middle two as the min max coordinates of intersection
58  // box in this dimension
59 
60  intmin[d] = val[1]; intmax[d] = val[2];
61  }
62 
63  // Calculate the volume
64 
65  double vol = 1.0;
66  for (int d = 0; d < dim; d++)
67  vol *= intmax[d]-intmin[d];
68 
69  // Sanity check
70 
71  assert(vol >= 0.0);
72 
73  if (vol == 0.0) return false;
74 
75  // Calculate the centroid
76 
77  JaliGeometry::Point centroid = (intmin+intmax)/2.0;
78 
79  // moments
80 
81  xsect_moments->clear();
82  xsect_moments->push_back(vol);
83  for (int d = 0; d < dim; d++)
84  xsect_moments->push_back(centroid[d]*vol);
85 
86  return true;
87 }
