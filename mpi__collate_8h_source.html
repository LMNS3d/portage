18  // Collates local vectors lvec (of various lengths) into a global vector gvec
19  // on rank 0. The `gvec` vector is only defined on rank 0, and it is resized
20  // there to the proper size. On other ranks it is resized to size 1 and left
21  // undefined.
22  template <typename T>
23  void collate_type(MPI_Comm comm, const int rank, const int numpe,
24  const MPI_Datatype mpi_type,
25  std::vector<T> &lvec, std::vector<T> &gvec) {
26  std::vector<int> lvec_sizes(numpe);
27  int lvec_size = lvec.size();
28  MPI_Gather(&lvec_size, 1, MPI_INT, &lvec_sizes[0], 1, MPI_INT, 0, comm);
29  std::vector<int> displs;
30  if (rank == 0) {
31  int gvec_size = std::accumulate(lvec_sizes.begin(), lvec_sizes.end(),
32  0);
33  gvec.resize(gvec_size);
34  displs.resize(lvec_sizes.size());
35  int idx = 0;
36  for (int i=0; i < lvec_sizes.size(); i++) {
37  displs[i] = idx;
38  idx += lvec_sizes[i];
39  }
40  } else {
41  // We resize to size 1, so that the expressions &gvec[0], &displs[0] below
42  // are well defined on all ranks.
43  gvec.resize(1);
44  displs.resize(1);
45  }
46  MPI_Gatherv(&lvec[0], lvec.size(), mpi_type, &gvec[0], &lvec_sizes[0],
47  &displs[0], mpi_type, 0, comm);
48  }
