55  // Collates local vectors lvec (of various lengths) into a global vector gvec
56  // on rank 0. The `gvec` vector is only defined on rank 0, and it is resized
57  // there to the proper size. On other ranks it is resized to size 1 and left
58  // undefined.
59  template <typename T>
60  void collate_type(MPI_Comm comm, const int rank, const int numpe,
61  const MPI_Datatype mpi_type,
62  std::vector<T> &lvec, std::vector<T> &gvec) {
63  std::vector<int> lvec_sizes(numpe);
64  int lvec_size = lvec.size();
65  MPI_Gather(&lvec_size, 1, MPI_INT, &lvec_sizes[0], 1, MPI_INT, 0, comm);
66  std::vector<int> displs;
67  if (rank == 0) {
68  int gvec_size = std::accumulate(lvec_sizes.begin(), lvec_sizes.end(),
69  0);
70  gvec.resize(gvec_size);
71  displs.resize(lvec_sizes.size());
72  int idx = 0;
73  for (int i=0; i < lvec_sizes.size(); i++) {
74  displs[i] = idx;
75  idx += lvec_sizes[i];
76  }
77  } else {
78  // We resize to size 1, so that the expressions &gvec[0], &displs[0] below
79  // are well defined on all ranks.
80  gvec.resize(1);
81  displs.resize(1);
82  }
83  MPI_Gatherv(&lvec[0], lvec.size(), mpi_type, &gvec[0], &lvec_sizes[0],
84  &displs[0], mpi_type, 0, comm);
85  }

100  idx.resize(x.size());

106  template <typename T>

108  std::vector<T> y(x.size());
