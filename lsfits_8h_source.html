44  std::vector<double> const & vals) {
45 
46  Point<D> coord0 = coords[0];
47 
48  double val0 = vals[0];
49 
50  // There are nvals but the first is the reference point where we
51  // are trying to compute the gradient; so the matrix sizes etc
52  // will only be nvals-1
53 
54  int nvals = vals.size();
55 
56  // Each row of A contains the components of the vector from
57  // coord0 to the candidate point being used in the Least Squares
58  // approximation (X_i-X_0).
59 
60  Matrix A(nvals-1, D);
61  for (int i = 0; i < nvals-1; ++i) {
62  for (int j = 0; j < D; ++j)
63  A[i][j] = coords[i+1][j]-coord0[j];
64  }
65 
66 
67  // A is a matrix of size nvals-1 by D (where D is the space
68  // dimension). So transpose(A)*A is D by D
69 
70  Matrix AT = A.transpose();
71 
72  Matrix ATA = AT*A;
73 
74  // Each entry/row of F contains the difference between the
75  // function value at the candidate point and the function value
76  // at the point where we are computing (f-f_0)
77 
78  std::vector<double> F(nvals-1);
79  for (int i = 0; i < nvals-1; ++i)
80  F[i] = vals[i+1]-val0;
81 
82  // F is a vector of nvals. So transpose(A)*F is vector of D
83  // (where D is the space dimension)
84 
85  Vector<D> ATF = Vector<D>(AT*F);
86 
87  // Inverse of ATA
88 
89  Matrix ATAinv = ATA.inverse();
90 
91  // Gradient of length D
92 
93  return ATAinv*ATF;

139  // coord0 to the candidate point being used in the Least Squares
140  // approximation (X_i-X_0), up to quadratic order. Index j0

148  for (int i = 0; i < nvals-1; i++) {

150  for (int j0 = 0; j0 < D; ++j0) {

152  // Add columns with the remaining quadratic delta terms
153  // for each i, starting at D+j, reusing linear delta terms
154  for (int k=0; k <= j0; k++) {
