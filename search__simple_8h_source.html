65 void getBoundingBox(
66  const std::vector<Portage::Point<2>> &cell_coord,
67  double* xlow, double* xhigh,
68  double* ylow, double* yhigh)
69 {
70  const double big = 1.e99;
71  double xl = big; double xh = -big;
72  double yl = big; double yh = -big;
73 
74  for (const auto& p : cell_coord) {
75  const double x = p[0];
76  const double y = p[1];
77  xl = std::min(xl, x); xh = std::max(xh, x);
78  yl = std::min(yl, y); yh = std::max(yh, y);
79  }
80 
81  *xlow = xl; *xhigh = xh;
82  *ylow = yl; *yhigh = yh;
83 
84 } // getBoundingBox

173 template<typename SourceMeshType, typename TargetMeshType>
174 void SearchSimple<SourceMeshType, TargetMeshType>::
175 operator() (const int cellId, std::vector<int> *candidates)
176 const {
177  // find bounding box for target cell
178  std::vector<Point<2>> cell_coord;
179  targetMesh_.cell_get_coordinates(cellId, &cell_coord);
180  double txlow, txhigh, tylow, tyhigh;
181  search_simple::getBoundingBox(cell_coord, &txlow, &txhigh, &tylow, &tyhigh);
182 
183  // now see which sourceMesh cells have bounding boxes overlapping
184  // with target cell
185  // do a naive linear search
186  const int numCells = sourceMesh_.num_owned_cells() +
187  sourceMesh_.num_ghost_cells();
188  for (int c = 0; c < numCells; ++c) {
189  if (std::max(txlow, xlow_[c]) < std::min(txhigh, xhigh_[c]) &&
190  std::max(tylow, ylow_[c]) < std::min(tyhigh, yhigh_[c])) {
191  candidates->push_back(c);
192  }
193  }
