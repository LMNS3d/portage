133  vector<Weights_t>
134  operator() (size_t const particleB, vector<unsigned int> const& source_particles) {
135  vector<Weights_t> result;
136  result.reserve(source_particles.size());
137 
138  switch (estimate_) {
139  case KernelDensity: {
140  for (auto const& particleA : source_particles) {
141  double weight_val = weight(particleA, particleB);
142  vector<double> pair_result(1, weight_val);
143  result.emplace_back(particleA, pair_result);
144  }
145  break;
146  }
147  case LocalRegression: {
148  size_t nbasis = Basis::function_size<dim>(basis_);
149  Point<dim> x = target_.get_particle_coordinates(particleB);
150 
151  // Calculate weights and moment matrix (P*W*transpose(P))
152  if (source_particles.size() < nbasis) {
153  throw std::runtime_error("too few source particles for local regression");
154  }
155  vector<double> weight_val(source_particles.size());
156  Matrix moment(nbasis,nbasis,0.);
157  size_t iA = 0;
158  for (auto const& particleA : source_particles) {
159  weight_val[iA] = weight(particleA, particleB); // save weights for later
160  Point<dim> y = source_.get_particle_coordinates(particleA);
161  auto basis = Basis::shift<dim>(basis_,x,y);
162  for (size_t i=0; i<nbasis; i++) {
163  for (size_t j=0; j<nbasis; j++) {
164  moment[i][j] += basis[i]*basis[j]*weight_val[iA];
165  }
166  }
167  iA++;
168  }
169 
170  // Calculate inverse(P*W*transpose(P))*P*W
171  iA = 0;
172  for (auto const& particleA : source_particles) {
173  vector<double> pair_result(nbasis);
174  Point<dim> y = source_.get_particle_coordinates(particleA);
175  vector<double> basis = Basis::shift<dim>(basis_,x,y);
176 
177  // recast as a Portage::Matrix
178  Matrix basis_matrix(nbasis,1);
179  for (size_t i=0; i<nbasis; i++) basis_matrix[i][0] = basis[i];
180 
181  // solve the linear system
182 #ifdef HAVE_LAPACKE
183  Matrix pair_result_matrix = moment.solve(basis_matrix, "lapack-posv");
184 #else
185  Matrix pair_result_matrix = moment.solve(basis_matrix);
186 #endif
187 
188  for (size_t i=0; i<nbasis; i++) pair_result[i] = pair_result_matrix[i][0]*weight_val[iA];
189  result.emplace_back(particleA, pair_result);
190  iA++;
191  }
192  break;
193  }
194  default: assert(false);
195  }
196  return result;
197  }
