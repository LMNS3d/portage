150  std::vector<Weights_t>
151  operator() (size_t const particleB, std::vector<unsigned int> const& source_particles) {
152  std::vector<Weights_t> result;
153  result.reserve(source_particles.size());
154 
155  switch (estimate_) {
156  case KernelDensity: {
157  for (auto const& particleA : source_particles) {
158  double weight_val = weight(particleA, particleB);
159  std::vector<double> pair_result(1, weight_val);
160  result.emplace_back(particleA, pair_result);
161  }
162  break;
163  }
164  case OperatorRegression:
165  case LocalRegression: {
166  size_t nbasis = Basis::function_size<dim>(basis_);
167  Point<dim> x = target_.get_particle_coordinates(particleB);
168 
169  // If too few particles, set estimate to zero for this target
170  bool zilchit = false;
171  if (source_particles.size() < nbasis) {
172  zilchit = true;
173  }
174 
175  // Calculate weights and moment matrix (P*W*transpose(P))
176  std::vector<double> weight_val(source_particles.size());
177  Matrix moment(nbasis,nbasis,0.);
178  size_t iA = 0;
179  if (not zilchit) {
180  for (auto const& particleA : source_particles) {
181  weight_val[iA] = weight(particleA, particleB); // save weights for later
182  Point<dim> y = source_.get_particle_coordinates(particleA);
183  auto basis = Basis::shift<dim>(basis_,x,y);
184  for (size_t i=0; i<nbasis; i++) {
185  for (size_t j=0; j<nbasis; j++) {
186  moment[i][j] += basis[i]*basis[j]*weight_val[iA];
187  }
188  }
189  iA++;
190  }
191  }
192 
193  // Calculate inverse(P*W*transpose(P))*P*W
194  iA = 0;
195  for (auto const& particleA : source_particles) {
196  std::vector<double> pair_result(nbasis);
197  Point<dim> y = source_.get_particle_coordinates(particleA);
198  std::vector<double> basis = Basis::shift<dim>(basis_,x,y);
199 
200  // recast as a Portage::Matrix
201  Matrix basis_matrix(nbasis,1);
202  for (size_t i=0; i<nbasis; i++) basis_matrix[i][0] = basis[i];
203 
204  // solve the linear system
205  if (not zilchit) {
206 #ifdef HAVE_LAPACKE
207  Matrix pair_result_matrix = moment.solve(basis_matrix, "lapack-sytr");
208 #else
209  Matrix pair_result_matrix = moment.solve(basis_matrix);
210 #endif
211  for (size_t i=0; i<nbasis; i++) pair_result[i] = pair_result_matrix[i][0]*weight_val[iA];
212  } else if (zilchit) {
213  for (size_t i=0; i<nbasis; i++) pair_result[i] = 0.;
214  }
215 
216  // If an operator is being applied, adjust final weights.
217  if (estimate_ == OperatorRegression) {
218  auto ijet = Basis::inverse_jet<dim>(basis_, x);
219  std::vector<std::vector<double>> basisop;
220  Operator::apply<dim>(operator_spec_, basis_, operator_domain_[particleB],
221  operator_data_[particleB], basisop);
222  size_t opsize = Operator::size_info(operator_spec_, basis_,
223  operator_domain_[particleB])[0];
224  std::vector<double> operator_result(opsize, 0.);
225  for (int j=0; j<opsize; j++) {
226  for (int k=0; k<nbasis; k++) {
227  for (int m=0; m<nbasis; m++) {
228  operator_result[j] += pair_result[k]*ijet[k][m]*basisop[m][j];
229  }
230  }
231  }
232  for (int j=0; j<nbasis; j++) pair_result[j] = operator_result[j];
233  }
234  result.emplace_back(particleA, pair_result);
235  iA++;
236  }
237  break;
238  }
239  default: // invalid estimate
240  assert(false);
241  }
242  return result;
243  }
