150  Matrix inverse() const {
151  if (Rows_ != Columns_) {
152  std::cerr << "Matrix is rectangular" << std::endl;
153  throw std::exception();
154  }
155 
156  Matrix Ainv(Rows_, Rows_, 0.0);
157 
158  // Create a temporary matrix with twice as many columns
159  Matrix D(Rows_, 2*Rows_);
160 
161  // Initialize the reduction matrix
162  int Rows2 = 2*Rows_;
163  for (int i = 0; i < Rows_; i++) {
164  for (int j = 0; j < Rows_; j++) {
165  D[i][j] = A_[i*Columns_+j];
166  D[i][Rows_+j] = 0.0;
167  }
168  D[i][Rows_+i] = 1.0;
169  }
170 
171  // Do the reduction
172  for (int i = 0; i < Rows_; i++) {
173  double alpha = D[i][i];
174  if (alpha == 0.0) {
175  std::cerr << "invert: Singular Matrix" << std::endl;
176  return Ainv;
177  }
178 
179  for (int j = 0; j < Rows2; j++)
180  D[i][j] = D[i][j]/alpha;
181 
182  for (int k = 0; k < Rows_; k++) {
183  if ((k - i) == 0)
184  continue;
185 
186  double beta = D[k][i];
187  for (int j = 0; j < Rows2; j++)
188  D[k][j] = D[k][j] - beta*D[i][j];
189  }
190  }
191 
192  // Copy result into output matrix
193  for (int i = 0; i < Rows_; i++)
194  for (int j = 0; j < Rows_; j++)
195  Ainv[i][j] = D[i][j + Rows_];
196 
197  return Ainv;
198  }
