117  Matrix inverse() const {
118  if (Rows_ != Columns_) {
119  std::cerr << "Matrix is rectangular" << std::endl;
120  throw std::exception();
121  }
122 
123  Matrix Ainv(Rows_, Rows_, 0.0);
124 
125  // Create a temporary matrix with twice as many columns
126  Matrix D(Rows_, 2*Rows_);
127 
128  // Initialize the reduction matrix
129  int Rows2 = 2*Rows_;
130  for (int i = 0; i < Rows_; i++) {
131  for (int j = 0; j < Rows_; j++) {
132  D[i][j] = A_[i*Columns_+j];
133  D[i][Rows_+j] = 0.0;
134  }
135  D[i][Rows_+i] = 1.0;
136  }
137 
138  // Do the reduction
139  for (int i = 0; i < Rows_; i++) {
140  double alpha = D[i][i];
141  if (alpha == 0.0) {
142  std::cerr << "invert: Singular Matrix" << std::endl;
143  return Ainv;
144  }
145 
146  for (int j = 0; j < Rows2; j++)
147  D[i][j] = D[i][j]/alpha;
148 
149  for (int k = 0; k < Rows_; k++) {
150  if ((k - i) == 0)
151  continue;
152 
153  double beta = D[k][i];
154  for (int j = 0; j < Rows2; j++)
155  D[k][j] = D[k][j] - beta*D[i][j];
156  }
157  }
158 
159  // Copy result into output matrix
160  for (int i = 0; i < Rows_; i++)
161  for (int j = 0; j < Rows_; j++)
162  Ainv[i][j] = D[i][j + Rows_];
163 
164  return Ainv;
165  }
