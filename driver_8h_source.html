261 
262  // Collect all cell based variables and remap them
263 
264  std::vector<std::string> source_cellvar_names;
265  std::vector<std::string> target_cellvar_names;
266  for (int i = 0; i < nvars; ++i) {

268  if (onwhat == CELL) {
269  source_cellvar_names.emplace_back(source_remap_var_names_[i]);

276  if (distributed)

278  else
279 #endif
280  remap<CELL>(source_cellvar_names, target_cellvar_names);
281  }
282 
283 
284  // Collect all node based variables and remap them
285 
286  std::vector<std::string> source_nodevar_names;
287  std::vector<std::string> target_nodevar_names;

289  for (int i = 0; i < nvars; ++i) {
