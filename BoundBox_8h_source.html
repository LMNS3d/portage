90  // Functions to add new elements to the box
92  void add(const Point<D>& p) {
93  if(empty()) {
94  min = p; max = p;
95  }
96  else {
97  for(long i = 0; i < D; i++) {
98  if(p[i] < min[i]) min[i] = p[i];
99  else if(p[i] > max[i]) max[i] = p[i];
100  }
101  }
102  }

105  void add(const IsotheticBBox<D>& box) {
106  add(box.getMin());
107  add(box.getMax());
108  }

111  void bulge(double epsilon) {

144  long longAxis() const {
145  // Find a longest axis
146  const Vector<D> diff = max-min;
147  double maxVal = diff[0];
148  long maxIdx = 0;

197  template <long D> bool approxEq(const IsotheticBBox<D>& box1,
198  const IsotheticBBox<D>& box2,
199  const double& tol)
200  {
201  const Point<D>& min1 = box1.getMin();
202  const Point<D>& max1 = box1.getMax();
203  const Point<D>& min2 = box2.getMin();

209  template <long D> bool interval(const IsotheticBBox<D>& box,
210  const Point<D>& orig,
211  const Vector<D>& magdir,
212  double& a, double& b)
213  {
214  a = 0;
215  b = 1;
216 
217  for(long i = 0; i < D; i++) {
218  if(magdir[i] == 0) {
219  if(orig[i] < box.getMin(i) || orig[i] > box.getMax(i)) return false;
220  } else {
221  register const double dMin = (box.getMin(i)-orig[i])/magdir[i];
222  register const double dMax = (box.getMax(i)-orig[i])/magdir[i];
223  if(dMax < dMin) {
224  if(b < dMax || a > dMin) return false;
225  if(a < dMax) a = dMax;
226  if(b > dMin) b = dMin;
227  } else {
228  if(b < dMin || a > dMax) return false;
229  if(a < dMin) a = dMin;
230  if(b > dMax) b = dMax;
231  }
232  }
233  }
234  return true;
235  }
