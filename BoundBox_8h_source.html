55  // Functions to add new elements to the box
57  void add(const Point<D>& p) {
58  if(empty()) {
59  min = p; max = p;
60  }
61  else {
62  for(long i = 0; i < D; i++) {
63  if(p[i] < min[i]) min[i] = p[i];
64  else if(p[i] > max[i]) max[i] = p[i];
65  }
66  }
67  }

162  template <long D> bool approxEq(const IsotheticBBox<D>& box1,
163  const IsotheticBBox<D>& box2,
164  const double& tol)
165  {
166  const Point<D>& min1 = box1.getMin();
167  const Point<D>& max1 = box1.getMax();
168  const Point<D>& min2 = box2.getMin();
169  const Point<D>& max2 = box2.getMax();

171  return (approxEq(min1,min2,tol) && approxEq(max1,max2,tol));

174  template <long D> bool interval(const IsotheticBBox<D>& box,
175  const Point<D>& orig,
176  const Vector<D>& magdir,
177  double& a, double& b)
178  {
179  a = 0;
180  b = 1;
181 
182  for(long i = 0; i < D; i++) {
183  if(magdir[i] == 0) {
184  if(orig[i] < box.getMin(i) || orig[i] > box.getMax(i)) return false;
185  } else {
186  register const double dMin = (box.getMin(i)-orig[i])/magdir[i];
187  register const double dMax = (box.getMax(i)-orig[i])/magdir[i];
188  if(dMax < dMin) {
189  if(b < dMax || a > dMin) return false;
190  if(a < dMax) a = dMax;
191  if(b > dMin) b = dMin;
192  } else {
193  if(b < dMin || a > dMax) return false;
194  if(a < dMin) a = dMin;
195  if(b > dMax) b = dMax;
196  }
197  }
198  }
199  return true;
200  }
