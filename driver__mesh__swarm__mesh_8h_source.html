197 
198 #ifdef PORTAGE_ENABLE_MPI

201  if (mpiexecutor && mpiexecutor->mpicomm != MPI_COMM_NULL) {
202  mycomm = mpiexecutor->mpicomm;
203  MPI_Comm_rank(mycomm, &comm_rank);
204  MPI_Comm_size(mycomm, &nprocs);
205  if (nprocs > 1) {
206  distributed = true;
207  std::cerr << "Cannot run Mesh-Swarm-Mesh driver in distributed mode yet\n";
208  return;

215  int numTargetCells = target_mesh_.num_owned_cells();
216  std::cout << "Number of target cells in target mesh on rank "
217  << comm_rank << ": "
218  << numTargetCells << std::endl;

220  int nvars = source_remap_var_names_.size();
221 
222  float tot_seconds = 0.0, tot_seconds_srch = 0.0,

226  gettimeofday(&begin_timeval, 0);

228  // CELL VARIABLE SECTION ---------------------------------------------------
229 
230  // get cell variable names
231  std::vector<std::string> source_cellvar_names;
232  std::vector<std::string> target_cellvar_names;
233  for (int i = 0; i < nvars; ++i) {
234  Entity_kind onwhat =
235  source_state_.get_entity(source_remap_var_names_[i]);
236 
237  if (onwhat == Entity_kind::CELL) {
238  source_cellvar_names.emplace_back(source_remap_var_names_[i]);
239  target_cellvar_names.emplace_back(target_remap_var_names_[i]);
240  }
241  }
242 
243  // Collect all cell based variables and remap them
244  if (source_cellvar_names.size() > 0)
245  {
246  // convert mesh wrappers to swarms
247  std::shared_ptr<Meshfree::Swarm<Dim>> source_swarm_ptr =
248  Meshfree::SwarmFactory<Dim>(source_mesh_, Entity_kind::CELL);
249  std::shared_ptr<Meshfree::Swarm<Dim>> target_swarm_ptr =
250  Meshfree::SwarmFactory<Dim>(target_mesh_, Entity_kind::CELL);
251  Meshfree::Swarm<Dim> &source_swarm(*source_swarm_ptr);
252  Meshfree::Swarm<Dim> &target_swarm(*target_swarm_ptr);
253 
254  // convert state wrappers to swarm variety
255  std::shared_ptr<Meshfree::SwarmState<Dim>> source_swarm_state_ptr =
256  Meshfree::SwarmStateFactory<Dim>(source_state_, Entity_kind::CELL);
257  std::shared_ptr<Meshfree::SwarmState<Dim>> target_swarm_state_ptr =
258  Meshfree::SwarmStateFactory<Dim>(target_state_, Entity_kind::CELL);
259  Meshfree::SwarmState<Dim> &source_swarm_state(*source_swarm_state_ptr);
260  Meshfree::SwarmState<Dim> &target_swarm_state(*target_swarm_state_ptr);
261 
262  // create spherically symmetric smoothing lengths for now
263  int ncells;
264  if (center_ == Meshfree::Scatter) ncells = source_mesh_.num_owned_cells();
265  else if (center_ == Meshfree::Gather) ncells = target_mesh_.num_owned_cells();
266  vector<std::vector<std::vector<double>>> smoothing_lengths
267  (ncells, std::vector<std::vector<double>>(1, std::vector<double>(Dim)));
268  for (int i=0; i<ncells; i++) {
269  double radius;
270  if (center_ == Meshfree::Scatter)
271  Wonton::cell_radius<Dim>(source_mesh_, i, &radius);
272  else if (center_ == Meshfree::Gather)
273  Wonton::cell_radius<Dim>(target_mesh_, i, &radius);
274  std::vector<std::vector<double>> h=smoothing_lengths[i];
275  h[0] = std::vector<double>(Dim, radius*smoothing_factor_);
276  smoothing_lengths[i]=h;
277  }
278

280  Meshfree::SwarmDriver<

290  target_swarm, target_swarm_state,
291  smoothing_lengths,
292  kernel_,
293  geometry_,
294  center_);
295

297  target_cellvar_names,
298  estimate_,
299  basis_,
300  operator_spec_,
301  operator_domains_,
302  operator_data_);
303 
304  // do the remap
305  swarm_driver.run(executor, true);
306 
307  // transfer data back to target mesh
308  for (auto name=target_cellvar_names.begin();
309  name!=target_cellvar_names.end(); name++)
310  {
311  typename Meshfree::SwarmState<Dim>::DblVecPtr sfield;
312  target_swarm_state.get_field(*name, sfield);
313  double *mfield;
314  target_state_.mesh_get_data(Entity_kind::CELL, *name, &mfield);
315  for (int i=0; i<target_swarm_state.get_size(); i++) {
316  mfield[i] = (*sfield)[i];
317  }
318  }
319  }

349  Meshfree::SwarmStateFactory<Dim>(target_state_, Entity_kind::NODE);

351  Meshfree::SwarmState<Dim> &target_swarm_state(*target_swarm_state_ptr);
352

354  int nnodes;

356  else if (center_ == Meshfree::Gather) nnodes = target_mesh_.num_owned_nodes();

363  std::vector<std::vector<double>> h=smoothing_lengths[i];
