212  void run(bool distributed) {
213 
214 #ifndef ENABLE_MPI
215  if (distributed) {
216  std::cout << "Request is for a parallel run but Portage is compiled for serial runs only\n";
217  return;
218  }
219 #endif
220 
221  int comm_rank = 0;
222 
223 #ifdef ENABLE_MPI
224  MPI_Comm_rank(MPI_COMM_WORLD, &comm_rank);
225 #endif
226 
227  if (comm_rank == 0) std::printf("in MSM_Driver::run()...\n");
228 
229  int numTargetCells = target_mesh_.num_owned_cells();
230  std::cout << "Number of target cells in target mesh on rank "
231  << comm_rank << ": "
232  << numTargetCells << std::endl;
233 
234  int nvars = source_remap_var_names_.size();
235 
236  // convert incoming and outgoing mesh wrappers to flat variety
237  Flat_Mesh_Wrapper<> source_mesh_flat;
238  source_mesh_flat.initialize(source_mesh_);
239 
240  Flat_Mesh_Wrapper<double> target_mesh_flat;
241  target_mesh_flat.initialize(target_mesh_);
242 
243  float tot_seconds = 0.0, tot_seconds_srch = 0.0,
244  tot_seconds_xsect = 0.0, tot_seconds_interp = 0.0;
245  struct timeval begin_timeval, end_timeval, diff_timeval;
246 
247  gettimeofday(&begin_timeval, 0);
248 
249  // CELL VARIABLE SECTION ---------------------------------------------------
250 
251  // get cell variable names
252  std::vector<std::string> source_cellvar_names;
253  std::vector<std::string> target_cellvar_names;
254  for (int i = 0; i < nvars; ++i) {
255  Entity_kind onwhat =
256  source_state_.get_entity(source_remap_var_names_[i]);
257 
258  if (onwhat == CELL) {
259  source_cellvar_names.emplace_back(source_remap_var_names_[i]);
260  target_cellvar_names.emplace_back(target_remap_var_names_[i]);
261  }
262  }
263 
264  // Collect all cell based variables and remap them
265  if (source_cellvar_names.size() > 0)
266  {
267  // convert flat mesh wrappers to swarm variety
268  Meshfree::Swarm<Dim> source_swarm(source_mesh_flat, CELL);
269  Meshfree::Swarm<Dim> target_swarm(target_mesh_flat, CELL);
270 
271  // convert state wrappers to flat variety
272  Flat_State_Wrapper<double> source_state_flat;
273  source_state_flat.initialize(source_state_, source_cellvar_names);
274 
275  Flat_State_Wrapper<double> target_state_flat;
276  target_state_flat.initialize(target_state_, target_cellvar_names);
277 
278  // convert flat wrappers to swarm variety
279  Meshfree::SwarmState<Dim> source_swarm_state(source_mesh_flat, CELL,
280  source_state_flat);
281  Meshfree::SwarmState<Dim> target_swarm_state(target_mesh_flat, CELL,
282  target_state_flat);
283 
284  // create spherically symmetric smoothing lengths for now
285  using std::vector;
286  int ncells = source_mesh_flat.num_owned_cells();
287  vector<vector<vector<double>>> smoothing_lengths
288  (ncells, vector<vector<double>>(1, vector<double>(Dim)));
289  for (int i=0; i<ncells; i++) {
290  double radius;
291  cell_radius<Dim>(source_mesh_flat, i, &radius);
292  smoothing_lengths[i][0] = vector<double>(Dim, radius*smoothing_factor_);
293  }
294 
295  // create swarm remap driver
296  Meshfree::SwarmDriver<
297  Search,
298  Accumulate,
299  Estimate,
300  Dim,
301  Meshfree::Swarm<Dim>,
302  Meshfree::SwarmState<Dim>,
303  Meshfree::Swarm<Dim>,
304  Meshfree::SwarmState<Dim>
305  > swarm_driver(source_swarm, source_swarm_state,
306  target_swarm, target_swarm_state,
307  smoothing_lengths,
308  kernel_,
309  geometry_,
310  Meshfree::Scatter);
311 
312  swarm_driver.set_remap_var_names(source_cellvar_names,
313  target_cellvar_names,
314  estimate_,
315  basis_);
316 
317  // do the remap
318  swarm_driver.run(false, false);
319 
320  // transfer data back to target mesh
321  for (auto name=target_cellvar_names.begin();
322  name!=target_cellvar_names.end(); name++)
323  {
324  typename Meshfree::SwarmState<Dim>::DblVecPtr sfield;
325  target_swarm_state.get_field(*name, sfield);
326  double *mfield;
327  target_state_.get_data(CELL, *name, &mfield);
328  for (int i=0; i<target_swarm_state.get_size(); i++) {
329  mfield[i] = (*sfield)[i];
330  }
331  }
332  }
333 
334  // NODE VARIABLE SECTION ---------------------------------------------------
335 
336  // get node variable names
337  std::vector<std::string> source_nodevar_names;
338  std::vector<std::string> target_nodevar_names;
339  for (int i = 0; i < nvars; ++i) {
340  Entity_kind onwhat =
341  source_state_.get_entity(source_remap_var_names_[i]);
342 
343  if (onwhat == NODE) {
344  source_nodevar_names.emplace_back(source_remap_var_names_[i]);
345  target_nodevar_names.emplace_back(target_remap_var_names_[i]);
346  }
347  }
348 
349  if (source_nodevar_names.size() > 0) {
350  // convert flat mesh wrappers to swarm variety
351  Meshfree::Swarm<Dim> source_swarm(source_mesh_flat, NODE);
352  Meshfree::Swarm<Dim> target_swarm(target_mesh_flat, NODE);
353 
354  // convert state wrappers to flat variety
355  Flat_State_Wrapper<double> source_state_flat;
356  source_state_flat.initialize(source_state_, source_nodevar_names);
357 
358  Flat_State_Wrapper<double> target_state_flat;
359  target_state_flat.initialize(target_state_, target_nodevar_names);
360 
361  // convert flat state wrappers to swarm variety
362  Meshfree::SwarmState<Dim> source_swarm_state(source_mesh_flat, NODE,
363  source_state_flat);
364  Meshfree::SwarmState<Dim> target_swarm_state(target_mesh_flat, NODE,
365  target_state_flat);
366 
367  // create smoothing lengths
368  using std::vector;
369  int nnodes = source_mesh_flat.num_owned_nodes();
370  vector<vector<vector<double>>> smoothing_lengths
371  (nnodes, vector<vector<double>>(1, vector<double>(Dim)));
372  for (int i=0; i<nnodes; i++) {
373  double radius;
374  node_radius<Dim>(source_mesh_flat, i, &radius);
375  smoothing_lengths[i][0] = vector<double>(Dim, radius*smoothing_factor_);
376  }
377 
378  // create swarm remap driver
379  Meshfree::SwarmDriver<
380  Search,
381  Accumulate,
382  Estimate,
383  Dim,
384  Meshfree::Swarm<Dim>,
385  Meshfree::SwarmState<Dim>,
386  Meshfree::Swarm<Dim>,
387  Meshfree::SwarmState<Dim>
388  > swarm_driver(source_swarm, source_swarm_state,
389  target_swarm, target_swarm_state,
390  smoothing_lengths,
391  kernel_,
392  geometry_,
393  Meshfree::Scatter);
394 
395  swarm_driver.set_remap_var_names(source_nodevar_names,
396  target_nodevar_names,
397  estimate_,
398  basis_);
399 
400  // do the remap
401  swarm_driver.run(false, false);
402 
403  // transfer data back to target mesh
404  for (auto name=target_nodevar_names.begin();
405  name!=target_nodevar_names.end(); name++)
406  {
407  typename Meshfree::SwarmState<Dim>::DblVecPtr sfield;
408  target_swarm_state.get_field(*name, sfield);
409  double *mfield;
410  target_state_.get_data(NODE, *name, &mfield);
411  for (int i=0; i<target_swarm_state.get_size(); i++) {
412  mfield[i] = (*sfield)[i];
413  }
414  }
415  }
416 
417  gettimeofday(&end_timeval, 0);
418  timersub(&end_timeval, &begin_timeval, &diff_timeval);
419  tot_seconds_interp = diff_timeval.tv_sec + 1.0E-6*diff_timeval.tv_usec;
420 
421  std::cout << "Mesh-Swarm-Mesh Time for Rank " << comm_rank << " (s): " <<
422  tot_seconds << std::endl;
423  }
