93 /****************************************************************************/
94 /* File :MedianSelect.c */
95 /* */
96 /* Creation Date :April 20, 1996 */
97 /* */
98 /* Purpose :MedianSelect takes an long `k', a double array 'arr', */
99 /* and an long array `prm' of length `n' */
100 /* and reorders `prm' so that `arr[prm[k]]' is the k'th */
101 /* largest element in the set `{arr[prm[i]], i=1,..,n}', */
102 /* and we have in addition the partitioning properties: */
103 /* i<k ==> arr[prm[i]] <= arr[prm[k]] and */
104 /* i>k ==> arr[prm[i]] >= arr[prm[k]]. */
105 /* */
106 /* Note that setting `k=(1+n)/2', this subroutine */
107 /* computes the MEDIAN VALUE of `n' values in `arr', */
108 /* and it does this in Order(N) time. This is thus a more */
109 /* efficient way of computing the median than a heap sort */
110 /* which is Order(NlogN). */
111 /* */
112 /****************************************************************************/
113 
114 template<long D> void MedianSelect (long k,
115  long n,
116  std::vector<Point<D> >& arr,
117  long *prm,
118  long icut)
119 {
120  long i, j, r, l, mid, ia;
121 
122  l = 0;
123  r = n-1;
124 
125  while ( r-l > 1 ) {
126 
127  mid = (l+r)/2;
128  SWAP(prm[mid], prm[l+1], long);
129 
130  if (arr[prm[l+1]][icut] > arr[prm[r]][icut]) {
131  SWAP(prm[l+1], prm[r], long);
132  }
133 
134  if (arr[prm[l]][icut] > arr[prm[r]][icut]) {
135  SWAP(prm[l], prm[r], long);
136  }
137 
138  if (arr[prm[l+1]][icut] > arr[prm[l]][icut]) {
139  SWAP(prm[l+1], prm[l], long);
140  }
141 
142  i = l + 1;
143  j = r;
144  ia = prm[l];
145 
146  i++;
147  while ( arr[prm[i]][icut] < arr[ia][icut] ) i++;
148 
149  j--;
150  while ( arr[prm[j]][icut] > arr[ia][icut] ) j--;
151 
152  while ( j >= i ) {
153  SWAP(prm[i], prm[j], long);
154 
155  i++;
156  while ( arr[prm[i]][icut] < arr[ia][icut] ) i++;
157 
158  j--;
159  while ( arr[prm[j]][icut] > arr[ia][icut] ) j--;
160  }
161 
162  prm[l] = prm[j];
163  prm[j] = ia;
164 
165  if (j >= k) r = j - 1;
166  if (j <= k) l = i;
167  }
168 
169  if ( (r-l == 1) && (arr[prm[r]][icut] < arr[prm[l]][icut]) ) {
170  SWAP(prm[l], prm[r], long);
171  }
172 
173 }

179 /* File :kdtree.c */
180 /* */
181 /* Creation Date :April 20, 1996 */
182 /* */
183 /* MODIFIED: 15 March, 2001 */
184 /* */
185 /* Purpose :KDTREE takes the set of Safety Boxes and */
186 /* produces a k-D tree that is stored in the array LINKP. */
187 /* Leaf nodes in LINKP each coincide with exactly one */
188 /* Safety Box. For each node in the k-D tree, */
189 /* there is a corresponding Safety Box which is just */
190 /* big enough to contain all the Safety Boxes ``under'' */
191 /* the node. */
192 /* */
193 /****************************************************************************/
194 
195 template <long D>
196 KDTree<D> *KDTreeCreate(const std::vector<IsotheticBBox<D> >& sboxp)
197 {
198  KDTree<D> *kdtree;
199  std::vector<Point<D> > bbc;
200  long *ipoly = NULL;
201  long i;
202  long node, nextp, itop;
203  long imn, imx, icut, imd;
204  long icrstack[100], imin[100], imax[100];
205  long ict[100];
206  Vector<D> dim;
207 
208 
209  if( sboxp.empty()) std::abort();
210  else {
211 
212  kdtree = new KDTree<D>;
213 
214  /* Obtain allocations for arrays */
215  kdtree->num_entities = sboxp.size();
216  ipoly = new long[sboxp.size()];
217  kdtree->sbox = new IsotheticBBox<D>[2*sboxp.size()];
218  kdtree->linkp = new long[2*sboxp.size()];
219  bbc.resize(sboxp.size());
220 
221  for (i = 0; i < sboxp.size(); i++) {
222 
223  /* Compute the centers of the bounding boxes */
224  bbc[i] = sboxp[i].center();
225 
226  /* Compute the root node of the k-D tree */
227  kdtree->sbox[0].add(sboxp[i]);
228  }
229 
230  /* If there is only one safety box, the root node is a leaf.
231  (Our convention is to set the link corresponding to a leaf
232  equal to the negative of the unique item contained in
233  that leaf.) If the root is a leaf, our work is done. */
234 
235  if ( sboxp.size() == 1 ) kdtree->linkp[0] = 0;
236  else
237  {
238  /* dimx, dimy, dimz are equal to the x, y, and z dimensions
239  of the bounding box corresponding to the current node
240  under consideration. */
241 
242  dim = kdtree->sbox[0].getMax() - kdtree->sbox[0].getMin();
243 
244 
245  /* ipoly will contain a permutation of the integers
246  {0,...,sboxp.size()-1}. This permutation will be altered as we
247  create our balanced binary tree. nextp contains the
248  address of the next available node to be filled. */
249 
250  nextp = 1;
251  for (i = 0; i < sboxp.size(); i++) ipoly[i] = i;
252 
253  /* Use a stack to create the tree. Put the root node
254  ``0'' on the top of the stack (icrstack). The array
255  subset of ipoly (i.e., subset of boxes associated
256  with this node) is recorded using the arrays
257  imin and imax. */
258 
259  itop = 0;
260  icrstack[itop] = 0;
261  imin[itop] = 0;
262  imax[itop] = sboxp.size()-1;
263 
264  /* Finally, we record in ict the appropriate ``cutting
265  direction'' for bisecting the set of safety boxes
266  corresponding to this node. This direction is either
267  the x, y, or z directions, depending on which dimension
268  of the bounding box is largest. */
269 
270  MaxComponent(dim,ict[itop]);
271 
272  /* Pop nodes off stack, create children nodes and put them
273  on stack. Continue until k-D tree has been created. */
274 
275  while ( itop >= 0) {
276 
277  /* Pop top node off stack. */
278 
279  node = icrstack[itop];
280 
281  /* Make this node point to next available node location (nextp).
282  This link represents the location of the FIRST CHILD
283  of the node. The adjacent location (nextp+1)
284  is implicitly taken to be the location of the SECOND
285  child of the node. */
286 
287  kdtree->linkp[node] = nextp;
288  imn = imin[itop];
289  imx = imax[itop];
290  icut = ict[itop];
291  itop--;
292 
293  /* Partition safety box subset associated with this node.
294  Using the appropriate cutting direction, use SELECT to
295  reorder (ipoly) so that the safety box with median bounding
296  box center coordinate is ipoly(imd), while the
297  boxes {ipoly[i], i<imd} have SMALLER (or equal)
298  bounding box coordinates, and the boxes with
299  {ipoly[i], i>imd} have GREATER (or equal) bounding box
300  coordinates. */
301 
302  imd = (imn+imx)/2;
303 
304  MedianSelect(imd-imn+1,imx-imn+1,bbc,&(ipoly[imn]),icut);
305 
306  /* If the first child's subset of safety boxes is a singleton,
307  the child is a leaf. Set the child's link to point to the
308  negative of the box number. Set the child's bounding
309  box to be equal to the safety box box. */
310 
311  if (imn == imd) {
312  kdtree->linkp[nextp]= -ipoly[imn];
313  kdtree->sbox[nextp] = sboxp[ipoly[imn]];
314  nextp = nextp+1;
315  }
316  else {
317 
318  /* In this case, the subset of safety boxes corres to the
319  first child is more than one, and the child is
320  not a leaf. Compute the bounding box of this child to
321  be the smallest box containing all the associated safety'
322  boxes. */
323 
324  kdtree->sbox[nextp] = sboxp[ipoly[imn]];
325 
326  for (i=imn+1; i<=imd; i++)
327  kdtree->sbox[nextp].add(sboxp[ipoly[i]]);
328 
329  /* Put the first child onto the stack, noting the
330  associated triangle subset in imin and imax, and
331  putting the appropriate cutting direction in ict. */
332 
333  dim = kdtree->sbox[nextp].getMax()
334  - kdtree->sbox[nextp].getMin();
335 
336  itop++;
337  icrstack[itop] = nextp;
338  imin[itop] = imn;
339  imax[itop] = imd;
340 
341  MaxComponent(dim,ict[itop]);
342  nextp++;
343  }
344 
345  /* If the second child's subset of safety boxes is a singleton,
346  the child is a leaf. Set the child's link to point to the
347  negative of the sbox number. Set the child's bounding
348  box to be equal to that of the safety box. */
349 
350  if ((imd+1) == imx) {
351  kdtree->linkp[nextp] = -ipoly[imx];
352  kdtree->sbox[nextp] = sboxp[ipoly[imx]];
353  nextp = nextp+1;
354  }
355  else {
356 
357  /* In this case, the subset of boxes corresponding to the
358  second child is more than one safety box, and the child is
359  not a leaf. Compute the bounding box of this child to
360  be the smallest box containing all the associated safety
361  boxes. */
362 
363  kdtree->sbox[nextp] = sboxp[ipoly[imd+1]];
364 
365  for (i=imd+2; i<=imx; i++)
366  kdtree->sbox[nextp].add(sboxp[ipoly[i]]);
367 
368  /* Put the second child onto the stack, noting the
369  associated triangle subset in imin and imax, and
370  putting the appropriate cutting direction in ict. */
371 
372  dim = kdtree->sbox[nextp].getMax()
373  - kdtree->sbox[nextp].getMin();
374 
375  itop++;
376  icrstack[itop] = nextp;
377  imin[itop] = imd+1;
378  imax[itop] = imx;
379 
380  MaxComponent(dim,ict[itop]);
381  nextp++;
382  }
383 
384  } /* End of the while loop */
385 
386  delete [] ipoly;
387  }
388  }
389 
390  return kdtree;
391 }

395 // Return a list of BBox id's containing the query point
396 template<long D>
397 void LocatePoint(const Point<D>& qp,
398  const KDTree<D>* kdtree,
399  std::vector<long>& pfound)
400 {
401  pfound.clear();
402  long itop, node, ind, j;
403  long istack[100];
404 
405  /* If root node is a leaf, return leaf. */
406 
407  if (kdtree->linkp[0] <= 0) {
408  if (kdtree->sbox[0].intersect(qp))
409  pfound.push_back(-kdtree->linkp[0]);
410  }
411  else {
412  itop = 0;
413  istack[itop] = 0;
414 
415  /* Traverse (relevant part of) k-D tree using stack. */
416 
417  while (itop >= 0) {
418 
419  /* pop node off of stack. */
420  node = istack[itop]; itop--;
421 
422  ind = kdtree->linkp[node];
423 
424  /* check if either child of NODE is a leaf or should be
425  put on stack. */
426  for (j=0; j<=1; j++) {
427  if (kdtree->sbox[ind+j].intersect(qp)) {
428  /* If child is a leaf, add to the list. */
429  if (kdtree->linkp[ind+j] <= 0) {
430  pfound.push_back(-kdtree->linkp[ind+j]);
431  }
432  else {
433  itop++;
434  istack[itop] = ind+j;
435  }
436  }
437  }
438  }
439  }
440 }
