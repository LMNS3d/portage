240 
241  int comm_rank = 0;
242 #ifndef ENABLE_MPI
243  if (distributed) {
244  std::cout << "Request is for a parallel run but Portage is compiled for serial runs only\n";
245  return;
246  }
247 #endif
248 
249  // EVEN OTHERWISE WE DON'T HAVE THE ABILITY TO DO DISTRIBUTED SWARMS
250  if (distributed) {
251  std::cout << "CANNOT DO DISTRIBUTED REMAPPING WITH SWARMS\n";
252  return;
253  }
254 
255  int numSourcePts = source_swarm_.num_owned_particles();
256  int numTargetPts = target_swarm_.num_owned_particles();
257  std::cout << "Number of target cells in target swarm on rank "
258  << comm_rank << ": "
259  << numTargetPts << std::endl;
260 
261  int nvars = source_remap_var_names_.size();
262 
263  // Collect all variables and remap them
264 
265  std::vector<std::string> source_var_names;
266  std::vector<std::string> target_var_names;
267  for (int i = 0; i < nvars; ++i) {
268  source_var_names.emplace_back(source_remap_var_names_[i]);
269  target_var_names.emplace_back(target_remap_var_names_[i]);
270  }
271 
272  if (source_var_names.size() > 0)
273  {
274 
275  float tot_seconds = 0.0, tot_seconds_srch = 0.0,
276  tot_seconds_xsect = 0.0, tot_seconds_interp = 0.0;
277  struct timeval begin_timeval, end_timeval, diff_timeval;
278 
279  // SEARCH
280  Portage::vector<std::vector<unsigned int>> candidates(numTargetPts);
281 
282  // Get an instance of the desired search algorithm type which is expected
283  // to be a functor with an operator() of the right form
284 
285  gettimeofday(&begin_timeval, 0);
286 
287  // search boxes around source points are of zero dimension but
288  // those around target points are determined by particle
289  // smoothing lengths
290 
291  // code below does not with with facted weightssourceExtents =
292  std::shared_ptr<std::vector<Point<Dim>>> sourceExtents;
293  std::shared_ptr<std::vector<Point<Dim>>> targetExtents;
294  if (weight_center_ == Portage::Meshfree::Gather) {
295  targetExtents = std::make_shared<std::vector<Point<Dim>>>(numTargetPts);
296  for (int i = 0; i < numTargetPts; i++) {
297  if (geom_types_[i] == Weight::FACETED) {
298  throw std::runtime_error("FACETED geometry is not available here");
299  }
300  (*targetExtents)[i] = Point<Dim>(smoothing_lengths_[i][0]);
301  }
302  }
303  if (weight_center_ == Portage::Meshfree::Scatter) {
304  sourceExtents = std::make_shared<std::vector<Point<Dim>>>(numSourcePts);
305  for (int i = 0; i < numSourcePts; i++) {
306  if (geom_types_[i] == Weight::FACETED) {
307  throw std::runtime_error("FACETED geometry is not available here");
308  }
309  (*sourceExtents)[i] = Point<Dim>(smoothing_lengths_[i][0]);
310  }
311  }
312 
313  const Search<Dim, SourceSwarm, TargetSwarm>
314  searchfunctor(source_swarm_, target_swarm_,
315  sourceExtents, targetExtents,
316  weight_center_);
317 
318  Portage::transform(target_swarm_.begin(PARTICLE, PARALLEL_OWNED),
319  target_swarm_.end(PARTICLE, PARALLEL_OWNED),
320  candidates.begin(), searchfunctor);
321 
322  gettimeofday(&end_timeval, 0);
323  timersub(&end_timeval, &begin_timeval, &diff_timeval);
324  tot_seconds_srch = diff_timeval.tv_sec + 1.0E-6*diff_timeval.tv_usec;
325 
326 
327 
328  // ACCUMULATE (build moment matrix, calculate shape functions)
329  // EQUIVALENT TO INTERSECT IN MESH-MESH REMAP
330 
331  gettimeofday(&begin_timeval, 0);
332 
333  // Get an instance of the desired accumulate algorithm type which is
334  // expected to be a functor with an operator() of the right form
335 
336  const Accumulate<Dim, SourceSwarm, TargetSwarm>
337  accumulateFunctor(source_swarm_, target_swarm_,
338  estimator_type_, weight_center_,
339  kernel_types_, geom_types_, smoothing_lengths_,
340  basis_type_);
341 
342  Portage::vector<std::vector<Weights_t>> source_pts_and_mults(numTargetPts);
343 
344  // For each particle in the target swarm get the shape functions
345  // (multipliers for source particle values)
346 
347  Portage::transform(target_swarm_.begin(PARTICLE, PARALLEL_OWNED),
348  target_swarm_.end(PARTICLE, PARALLEL_OWNED),
349  candidates.begin(),

354  timersub(&end_timeval, &begin_timeval, &diff_timeval);
355  tot_seconds_xsect = diff_timeval.tv_sec + 1.0E-6*diff_timeval.tv_usec;
356 
357 
358  // ESTIMATE (one variable at a time)
359 
360  gettimeofday(&begin_timeval, 0);
361 
362  nvars = source_var_names.size();
363  if (comm_rank == 0)
364  std::cout << "number of variables to remap is " << nvars <<
365  std::endl;
366 
367  // Get an instance of the desired interpolate algorithm type
368  Estimate<Dim, SourceState> estimateFunctor(source_state_);
369 
370  for (int i = 0; i < nvars; ++i) {
371  //amh: ?? add back accuracy output statement??
372  if (comm_rank == 0) std::cout << "Remapping cell variable " <<
373  source_var_names[i] <<
374  " to variable " << target_var_names[i] <<
375  std::endl;
376 
377  estimateFunctor.set_variable(source_var_names[i]);
378 
379  // This populates targetField with the values returned by the
380  // remapper operator
381 
382  // ***************** NOTE NOTE NOTE NOTE ********************
383  // THE CURRENT SWARM_STATE DOES NOT HAVE AN OPERATOR TO RETURN
384  // A RAW POINTER TO THE DATA. INSTEAD IT RETURNS THIS REQUIRES
385  // A SPECIFIC TYPE OF THE SWARM_STATE CLASS. THIS IS A BAD
386  // IDEA BECAUSE IT FORCES ALL OTHER SWARM_STATE CLASSES TO
387  // HAVE THIS SAME DEFINITION. IT ALSO LEADS TO THE UGLY USAGE
388  // WE SEE BELOW BECAUSE WE NEED A PORTAGE POINTER TO BE ABLE
389  // TO USE THRUST
390 
391  typename SwarmState<Dim>::DblVecPtr target_field_shared_ptr;
392 
393  target_state_.get_field(target_var_names[i], target_field_shared_ptr);
394  Portage::pointer<double> target_field(&((*target_field_shared_ptr)[0]));
395 
396  Portage::transform(target_swarm_.begin(PARTICLE, PARALLEL_OWNED),
397  target_swarm_.end(PARTICLE, PARALLEL_OWNED),
398  source_pts_and_mults.begin(),
399  target_field, estimateFunctor);
400  }
401 
402  gettimeofday(&end_timeval, 0);
403  timersub(&end_timeval, &begin_timeval, &diff_timeval);
404  tot_seconds_interp = diff_timeval.tv_sec + 1.0E-6*diff_timeval.tv_usec;
405 
406  tot_seconds = tot_seconds_srch + tot_seconds_xsect + tot_seconds_interp;
407 
408  if (report_time) {
409  std::cout << "Swarm Transform Time Rank " << comm_rank << " (s): " <<
410  tot_seconds << std::endl;
411  std::cout << " Swarm Search Time Rank " << comm_rank << " (s): " <<
412  tot_seconds_srch << std::endl;
413  std::cout << " Swarm Accumulate Time Rank " << comm_rank << " (s): " <<
414  tot_seconds_xsect << std::endl;
415  std::cout << " Swarm Estimate Time Rank " << comm_rank << " (s): " <<
416  tot_seconds_interp << std::endl;
417  }
418  }
