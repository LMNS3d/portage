177  std::vector<Weights_t> operator() (const int tgt_cell,
178  const std::vector<int> src_cells) const {
179 
180  std::vector<std::array<Point<3>, 4>> target_tet_coords;
181 
182  // We should avoid any decomposition for cells of a rectangular
183  // mesh but for now we will decompose the target all the time
184 
185  targetMeshWrapper.decompose_cell_into_tets(tgt_cell, &target_tet_coords,
186  rectangular_mesh_);
187 
188  // CAN MAKE THIS INTO A THRUST::TRANSFORM CALL
189  int nsrc = src_cells.size();
190  std::vector<Weights_t> sources_and_weights(nsrc);
191  int ninserted = 0;
192  for (int i = 0; i < nsrc; i++) {
193  int s = src_cells[i];
194 
195  Weights_t & this_wt = sources_and_weights[ninserted];
196  this_wt.entityID = s;
197 
198 #ifdef HAVE_TANGRAM
199  int nmats = sourceStateWrapper.cell_get_num_mats(s);
200  std::vector<int> cellmats;
201  sourceStateWrapper.cell_get_mats(s, &cellmats);
202 
203  if (!nmats || (matid_ == -1) || (nmats == 1 && cellmats[0] == matid_)) {
204  // ---------- Intersection with pure cell ---------------
205  // nmats == 0 -- no materials ==> single material
206  // matid_ == -1 -- intersect with mesh not a particular material
207  // nmats == 1 && cellmats[0] == matid -- intersection with pure cell
208  // containing matid
209 
210  facetedpoly_t srcpoly;
211  sourceMeshWrapper.cell_get_facetization(s, &srcpoly.facetpoints,
212  &srcpoly.points);
213 
214  this_wt.weights = intersect_polys_r3d(srcpoly, target_tet_coords);
215 
216  } else if (std::find(cellmats.begin(), cellmats.end(), matid_) !=
217  cellmats.end()) {
218  // mixed cell containing this material - intersect with
219  // polygon approximation of this material in the cell
220  // (obtained from interface reconstruction)
221 
222  Tangram::CellMatPoly<3> const& cellmatpoly =
223  interface_reconstructor->cell_matpoly_data(s);
224  std::vector<Tangram::MatPoly<3>> matpolys =
225  cellmatpoly.get_matpolys(matid_);
226 
227  this_wt.weights.resize(4,0.0);
228  for (int j = 0; j < matpolys.size(); j++) {
229  facetedpoly_t srcpoly = get_faceted_matpoly(matpolys[j]);
230 
231  std::vector<double> momvec = intersect_polys_r3d(srcpoly,
232  target_tet_coords);
233  for (int k = 0; k < 4; k++)
234  this_wt.weights[k] += momvec[k];
235  }
236 
237  }
238 #else
239  facetedpoly_t srcpoly;
240  sourceMeshWrapper.cell_get_facetization(s, &srcpoly.facetpoints,
241  &srcpoly.points);
242 
243  this_wt.weights = intersect_polys_r3d(srcpoly, target_tet_coords);
244 #endif
245  // Increment if vol of intersection > 0; otherwise, allow overwrite
246  if (this_wt.weights.size() && this_wt.weights[0] > 0.0)
247  ninserted++;
248  }
249 
250  sources_and_weights.resize(ninserted);
251  return sources_and_weights;
252  }
