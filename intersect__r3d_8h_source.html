41 std::vector<double>
42 intersect_3Dpolys(facetedpoly_t srcpoly,
43  std::vector<std::array<Point<3>, 4>> target_tet_coords) {
44 
45  // Bounding box of the target cell - will be used to compute
46  // epsilon for bounding box check. We could use the source cell
47  // coordinates to find the bounds as well but its probably an
48  // unnecessary computation (EVEN if the target cell is much
49  // smaller than the source cell)
50 
51  std::vector<double> source_cell_bounds = {1e99, -1e99, 1e99, -1e99,
52  1e99, -1e99};
53 
54  // Initialize the source polyhedron description in a form R3D wants
55  // Simultaneously compute the bounding box
56  r3d_poly src_r3dpoly;
57  int num_verts = srcpoly.points.size();
58  r3d_rvec3 *verts = new r3d_rvec3[num_verts];
59  for (int i = 0; i < num_verts; i++) {
60  for (int j = 0; j < 3; j++) {
61  verts[i].xyz[j] = srcpoly.points[i][j];
62 
63  if (source_cell_bounds[2*j] > verts[i].xyz[j])
64  source_cell_bounds[2*j] = verts[i].xyz[j];
65  if (source_cell_bounds[2*j+1] < verts[i].xyz[j])
66  source_cell_bounds[2*j+1] = verts[i].xyz[j];
67  }
68  }
69 
70  double MAXLEN = -1e99;
71  for (int j = 0; j < 3; j++) {
72  double len = source_cell_bounds[2*j+1]-source_cell_bounds[2*j];
73  if (MAXLEN < len) MAXLEN = len;
74  }
75  double bbeps = 1.0e-12*MAXLEN; // used only for bounding box check
76  // // not for intersections
77 
78 
79  int num_faces = srcpoly.facetpoints.size();
80  r3d_int *face_num_verts = new r3d_int[num_faces];
81  for (int i = 0; i < num_faces; i++)
82  face_num_verts[i] = srcpoly.facetpoints[i].size();
83 
84  r3d_int **face_vert_ids = new r3d_int *[num_faces];
85  for (int i = 0; i < num_faces; i++)
86  face_vert_ids[i] = new r3d_int[face_num_verts[i]];
87 
88  for (int i = 0; i < num_faces; i++)
89  std::copy(srcpoly.facetpoints[i].begin(), srcpoly.facetpoints[i].end(),
90  face_vert_ids[i]);
91 
92 #ifdef DEBUG
93  // Lets check the volume of the source polygon - If its convex or
94  // mildly non-convex, we should get a positive volume
95 
96  // First calculate a center point
97  Point<3> cen(0.0, 0.0, 0.0);
98  for (int i = 0; i < num_verts; i++)
99  cen += srcpoly.points[i];
100  cen /= num_verts;
101 
102  // Assume triangular facets only
103  double polyvol = 0.0;
104  for (int i = 0; i < num_faces; i++) {
105  // p0, p1, p2 traversed in order form a triangle whose normal
106  // points out of the source polyhedron
107  Point<3> &p0 = srcpoly.points[srcpoly.facetpoints[i][0]];
108  Point<3> &p1 = srcpoly.points[srcpoly.facetpoints[i][1]];
109  Point<3> &p2 = srcpoly.points[srcpoly.facetpoints[i][2]];
110 
111  Vector<3> v0 = p1-p0;
112  Vector<3> v1 = p2-p0;
113  Vector<3> outnormal = cross(v0, v1);
114  Vector<3> v2 = cen-p0;
115  double tetvol = -dot(v2, outnormal)/6.0;
116  if (tetvol < 0.0)
117  std::cerr << "Wrong orientation of facets or non-convex polyhedron" <<
118  std::endl;
119  polyvol += tetvol;
120  }
121  if (polyvol < 0.0)
122  throw std::runtime_error("Source polyhedron has negative volume");
123 #endif
124 
125  r3d_init_poly(&src_r3dpoly, verts, num_verts, face_vert_ids, face_num_verts,
126  num_faces);
127 
128  // Finished building source poly; now intersect with tets of target cell
129 
130  std::vector<double> moments(4, 0);
131  for (auto const & target_cell_tet : target_tet_coords) {
132  std::vector<r3d_plane> faces(4);
133 
134  std::vector<double> target_tet_bounds = {1e99, -1e99, 1e99,
135  -1e99, 1e99, -1e99};
136  r3d_rvec3 verts2[4];
137  for (int i = 0; i < 4; i++)
138  for (int j = 0; j < 3; j++) {
139  verts2[i].xyz[j] = target_cell_tet[i][j];
140 
141  if (target_tet_bounds[2*j] > verts2[i].xyz[j])
142  target_tet_bounds[2*j] = verts2[i].xyz[j];
143  if (target_tet_bounds[2*j+1] < verts2[i].xyz[j])
144  target_tet_bounds[2*j+1] = verts2[i].xyz[j];
145  }
146 
147  // Check if the target and source bounding boxes overlap - bbeps
148  // is used to subject touching tets to the full intersection
149  // (just in case)
150 
151  bool check_bb = true;
152  if (check_bb) {
153  bool disjoint = false;
154  for (int j = 0; j < 3 && !disjoint; ++j)
155  disjoint = (target_tet_bounds[2*j] > source_cell_bounds[2*j+1]+bbeps ||
156  target_tet_bounds[2*j+1] < source_cell_bounds[2*j]-bbeps);
157  if (disjoint) continue;
158  }
159 
160 #ifdef DEBUG
161  if (r3d_orient(verts2) < 0)
162  throw std::runtime_error("target_wedge has negative volume");
163 #endif
164 
165  r3d_tet_faces_from_verts(&faces[0], verts2);
166 
167  // clip the source poly against the faces of the target tet - but
168  // make a copy of src_r3dpoly first because it will get modified
169  // in the process of clipping
170  r3d_poly src_r3dpoly_copy = src_r3dpoly;
171  r3d_clip(&src_r3dpoly_copy, &faces[0], 4);
172 
173  // find the moments (up to quadratic order) of the clipped poly
174  const int POLY_ORDER = 1;
175  r3d_real om[R3D_NUM_MOMENTS(POLY_ORDER)];
176  r3d_reduce(&src_r3dpoly_copy, om, POLY_ORDER);
177 
178  // Check that the returned volume is positive (if the volume is
179  // zero, i.e. abs(om[0]) < eps, then it can sometimes be
180  // slightly negative, like om[0] == -1.24811e-16. For this
181  // reason we use the condition om[0] < -eps.
182 
183  const double eps = 1e-14; // @todo multiply by domain or element size
184  if (om[0] < -eps) throw std::runtime_error("Negative volume");
185 
186  // Accumulate moments:
187  for (int i = 0; i < 4; i++)
188  moments[i] += om[i];
189  }
190 
191  delete [] verts;
192  delete [] face_num_verts;
193  for (int i = 0; i < num_faces; i++)
194  delete [] face_vert_ids[i];
195  delete [] face_vert_ids;
196 
197  return moments;
198 } // intersect_3Dpolys

269  std::vector<Weights_t> operator() (const int tgt_cell,
270  const std::vector<int> src_cells) const {
271 
272  std::vector<std::array<Point<3>, 4>> target_tet_coords;
273 
274  // We should avoid any decomposition for cells of a rectangular
275  // mesh but for now we will decompose the target all the time
276 
277  targetMeshWrapper.decompose_cell_into_tets(tgt_cell, &target_tet_coords,
278  rectangular_mesh_);

280 
281  // CAN MAKE THIS INTO A THRUST::TRANSFORM CALL

283  std::vector<Weights_t> sources_and_weights(nsrc);

285  for (int i = 0; i < nsrc; i++) {
286  int s = src_cells[i];
287  facetedpoly_t srcpoly;

289  &srcpoly.points);

291  Weights_t & this_wt = sources_and_weights[ninserted];
292  this_wt.entityID = s;

294 
295  // Increment if vol of intersection > 0; otherwise, allow overwrite

297  ninserted++;
298  }

301  return sources_and_weights;
302  }
