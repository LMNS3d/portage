40  std::vector<std::vector<double>> operator() (const int cellA,
41  const int cellB) const {
42  std::vector<std::array<Portage::Point<3>, 4>> source_coords, target_coords;
43  sourceMeshWrapper.decompose_cell_into_tets(cellA, &source_coords,
44  planar_hex_);
45  targetMeshWrapper.decompose_cell_into_tets(cellB, &target_coords,
46  planar_hex_);
47 
48  // Bounding box of the target cell - will be used to compute
49  // epsilon for bounding box check. We could use the source cell
50  // coordinates to find the bounds as well but its probably an
51  // unnecessary computation (EVEN if the target cell is much
52  // smaller than the source cell)
53 
54  std::array<double, 6> target_cell_bounds = {1e99, -1e99, 1e99, -1e99,
55  1e99, -1e99};
56 
57  std::vector<double> moments(4, 0);
58 
59  std::vector<std::array<r3d_plane, 4>> target_tetfaces_all;
60  target_tetfaces_all.reserve(target_coords.size());
61  std::vector<std::array<double, 6>> target_tetbounds_all;
62  target_tetbounds_all.reserve(target_coords.size());
63 
64  for (const auto &target_cell_tet : target_coords) {
65  std::array<r3d_plane, 4> faces;
66  std::array<double, 6> bounds = {1e99, -1e99, 1e99, -1e99, 1e99, -1e99};
67  r3d_rvec3 verts2[4];
68  for (int i=0; i<4; i++)
69  for (int j=0; j<3; j++) {
70  verts2[i].xyz[j] = target_cell_tet[i][j];
71  if (bounds[2*j] > verts2[i].xyz[j])
72  bounds[2*j] = verts2[i].xyz[j];
73  if (bounds[2*j+1] < verts2[i].xyz[j])
74  bounds[2*j+1] = verts2[i].xyz[j];
75  }
76  target_tetbounds_all.emplace_back(bounds);
77 
78  for (int j = 0; j < 3; j++) {
79  if (target_cell_bounds[2*j] > bounds[2*j])
80  target_cell_bounds[2*j] = bounds[2*j];
81  if (target_cell_bounds[2*j+1] < bounds[2*j+1])
82  target_cell_bounds[2*j+1] = bounds[2*j+1];
83  }
84 
85  // Only do this check in Debug mode. This test is important,
86  // otherwise R3D returns invalid results.
87 #ifdef DEBUG
88  if (r3d_orient(verts2) < 0)
89  throw std::runtime_error("target_wedge has negative volume");
90 #endif
91 
92  r3d_tet_faces_from_verts(&faces[0], verts2);
93  target_tetfaces_all.emplace_back(faces);
94  }
95 
96  double MAXLEN = -1e99;
97  for (int j = 0; j < 3; j++) {
98  double len = target_cell_bounds[2*j+1]-target_cell_bounds[2*j];
99  if (MAXLEN < len) MAXLEN = len;
100  }
101 
102  double bbeps = 1.0e-12*MAXLEN; // used only for bounding box check
103  // // not for intersections
104 
105 
106  // Now intersect each tet from the source cell against each tet of
107  // the target cell (IF their bounding boxes overlap)
108 
109  for (const auto &source_cell_tet : source_coords) {
110  r3d_rvec3 verts1[4];
111  std::array<double, 6> source_tetbounds = {1e99, -1e99, 1e99, -1e99, 1e99, -1e99};
112  for (int i=0; i<4; i++)
113  for (int j=0; j<3; j++) {
114  verts1[i].xyz[j] = source_cell_tet[i][j];
115  if (source_tetbounds[2*j] > verts1[i].xyz[j])
116  source_tetbounds[2*j] = verts1[i].xyz[j];
117  if (source_tetbounds[2*j+1] < verts1[i].xyz[j])
118  source_tetbounds[2*j+1] = verts1[i].xyz[j];
119  }
120 
121  // Only do this check in Debug mode. This test is important,
122  // otherwise R3D returns invalid results.
123 #ifdef DEBUG
124  if (r3d_orient(verts1) < 0)
125  throw std::runtime_error("source_wedge has negative volume");
126 #endif
127 
128  int t = 0;
129  for (auto &faces : target_tetfaces_all) {
130 
131  const std::array<double, 6>& target_tetbounds = target_tetbounds_all[t++];
132 
133  // Check if the target and source bounding boxes overlap - bbeps
134  // is used to subject touching tets to the full intersection
135  // (just in case)
136 
137  bool check_bb = true;
138  if (check_bb) {
139  bool disjoint = false;
140  for (int j = 0; j < 3 && !disjoint; ++j)
141  disjoint = (target_tetbounds[2*j] > source_tetbounds[2*j+1]+bbeps ||
142  target_tetbounds[2*j+1] < source_tetbounds[2*j]-bbeps);
143  if (disjoint) continue;
144  }
145 
146  r3d_poly poly;
147  r3d_init_tet(&poly, verts1);
148  // Only do this check in Debug mode:
149 #ifdef DEBUG
150  if (r3d_is_good(&poly) == 0)
151  throw std::runtime_error("source_wedge: invalid poly");
152 #endif
153 
154  // clip the first tet against the faces of the second
155  r3d_clip(&poly, &faces[0], 4);
156 
157  // find the moments (up to quadratic order) of the clipped poly
158  const int POLY_ORDER=1;
159  // Only do this check in Debug mode:
160 #ifdef DEBUG
161  if (R3D_NUM_MOMENTS(POLY_ORDER) != 4)
162  throw std::runtime_error("Invalid number of moments");
163 #endif
164  r3d_real om[R3D_NUM_MOMENTS(POLY_ORDER)];
165  r3d_reduce(&poly, om, POLY_ORDER);
166 
167  // Check that the returned volume is positive (if the volume is zero,
168  // i.e. abs(om[0]) < eps, then it can sometimes be slightly negative,
169  // like om[0] == -1.24811e-16. For this reason we use the condition
170  // om[0] < -eps.
171  const double eps=1e-14; // @todo Should multiply by domain or element size
172  if (om[0] < -eps) throw std::runtime_error("Negative volume");
173 
174  // Accumulate moments:
175  for (int i=0; i<4; i++) {
176  moments[i] += om[i];
177  }
178  }
179  }
180 
181  std::vector<std::vector<double>> moments_all;
182  moments_all.push_back(moments);
183  return moments_all;
184  }
