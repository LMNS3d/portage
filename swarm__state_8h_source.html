146 template<size_t dim>
147 SwarmState<dim>::SwarmState(Portage::Flat_Mesh_Wrapper<double> &mesh,
148  Portage::Entity_kind entity,
149  Portage::Flat_State_Wrapper<double> &state)
150  : num_particles_(0)
151 {
152  if (dim != mesh.space_dimension()) {
153  throw std::runtime_error(string("dimension mismatch"));
154  }
155 
156  if (entity==CELL) {
157  num_particles_ = mesh.num_owned_cells();
158  } else if (entity==NODE) {
159  num_particles_ = mesh.num_owned_nodes();
160  }
161 
162  assert(state.get_entity_size(entity) == num_particles_);
163 
164  std::vector<std::string> dnames;
165  state.get_names(entity, dnames);
166 
167  for (auto iter=dnames.begin(); iter!=dnames.end(); iter++) {
168  double *datap;
169  state.get_data(entity, *iter, &datap);
170  DblVecPtr data = make_shared<vector<double>>(num_particles_);
171  for (size_t i=0; i<num_particles_; i++) (*data)[i] = datap[i];
172  add_field(*iter, data);
173  }
174 }

176 template<size_t dim>
177 void SwarmState<dim>::add_field(const string name, IntVecPtr value) {
178  // check size
179  if (value->size() != num_particles_) {
180  throw std::runtime_error(
181  string("incorrect size when adding attempting to add int field ")+name);
182  }
183 
184  // check duplicate
185  auto checkdup = int_field_map_.find(name);
186  if (checkdup != int_field_map_.end()) {
187  assert(checkdup->first == name);
188  throw std::runtime_error(string("tried to add int field ")+name+
189  "when it already existed");
190  }
191 
192  // add it
193  int_field_map_.insert(pair<string, IntVecPtr>(name, value));
194 }
