145 template<size_t dim>
146 void SwarmState<dim>::add_field(const string name, IntVecPtr value) {
147  // check size
148  if (value->size() != npoints_owned_) {
149  throw std::runtime_error(
150  string("incorrect size when adding attempting to add int field ")+name);
151  }
152 
153  // check duplicate
154  auto checkdup = int_field_map_.find(name);
155  if (checkdup != int_field_map_.end()) {
156  assert(checkdup->first == name);
157  throw std::runtime_error(string("tried to add int field ")+name+
158  "when it already existed");
159  }
160 
161  // add it
162  int_field_map_.insert(pair<string, IntVecPtr>(name, value));
163 }

247 template<size_t dim, class StateWrapper>
248 shared_ptr<SwarmState<dim>> SwarmStateFactory(
249  const StateWrapper &state,
250  const Portage::Entity_kind entity)
251 {
252  // create return value
253  size_t ndata=0;
254 
255  for (const std::string name : state.names()) {
256  // Simple_State does not store separte lists of names by entity,
257  // so we have to filter.
258  if (state.get_entity(name) == entity) {
259  ndata = state.get_data_size(entity, name);
260  break;
261  }
262  }
263  shared_ptr<SwarmState<dim>> result=make_shared<SwarmState<dim>>(ndata);
264 
265  // copy data
266  for (const std::string name : state.names()) {
267  if (state.get_entity(name) != entity) continue;
268 
269  // make sure all fields have same size
270  assert(state.get_data_size(entity, name) == ndata);
271 
272  const double *datap;
273  state.mesh_get_data(entity, name, &datap);
274  typename SwarmState<dim>::DblVecPtr data = make_shared<vector<double>>(ndata);
275 
276  for (size_t i=0; i<ndata; i++) (*data)[i] = datap[i];
277  result->add_field(name, data);
278  }
279 
280  return result;
281 }
