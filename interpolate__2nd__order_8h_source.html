242  void set_interpolation_variable(std::string const & interp_var_name,
243  Limiter_type limiter_type = NOLIMITER) {
244 
245  interp_var_name_ = interp_var_name;
246  limiter_type_ = limiter_type;
247 
248  // Extract the field data from the statemanager and the source cells for
249  // which the gradient has to be computed.
250  int nentities;
251  std::vector<int> cellids;
252  field_type_ = source_state_.field_type(Entity_kind::CELL, interp_var_name);
253  if (field_type_ == Field_type::MESH_FIELD)
254  {
255  source_state_.mesh_get_data(Entity_kind::CELL, interp_var_name, &source_vals_);
256  nentities = source_mesh_.num_entities(Entity_kind::CELL);
257  }
258  else
259  {
260  source_state_.mat_get_celldata(interp_var_name, matid_, &source_vals_);
261  source_state_.mat_get_cells(matid_, &cellids);
262  nentities = cellids.size();
263  }
264  // Compute the limited gradients for the field
265 #ifdef HAVE_TANGRAM
266  Limited_Gradient<D, Entity_kind::CELL, SourceMeshType, StateType, InterfaceReconstructorType,
267  Matpoly_Splitter, Matpoly_Clipper>
268  limgrad(source_mesh_, source_state_, interp_var_name_, limiter_type_,
269  interface_reconstructor_);
270  if (field_type_ == Field_type::MULTIMATERIAL_FIELD)
271  limgrad.set_material(matid_);
272 #else
273  Limited_Gradient<D, Entity_kind::CELL, SourceMeshType, StateType>
274  limgrad(source_mesh_, source_state_, interp_var_name_, limiter_type_);
275 #endif
276 
277  gradients_.resize(nentities);
278 
279  // call transform functor to take the values of the variable on
280  // the cells and compute a "limited" gradient of the field on the
281  // cells (for transform definition, see portage.h)
282 
283  // Even though we defined Portage::transform (to be
284  // thrust::transform or boost::transform) in portage.h, the
285  // compiler is not able to disambiguate this call and is getting
286  // confused. So we will explicitly state that this is Portage::transform
287  if (field_type_ == Field_type::MESH_FIELD){
288  Portage::transform(source_mesh_.begin(Entity_kind::CELL), source_mesh_.end(Entity_kind::CELL),
289  gradients_.begin(), limgrad);
290  }
291  else{
292  Portage::transform(cellids.begin(), cellids.end(),
293  gradients_.begin(), limgrad);
294  }

331  double operator() (int const targetCellID,
332  std::vector<Weights_t> const & sources_and_weights) const
333  {
334  int nsrccells = sources_and_weights.size();
335  if (!nsrccells) return 0.0;
336 
337  double totalval = 0.0;
338  double wtsum0 = 0.0;
339 
340  // contribution of the source cell is its field value weighted by
341  // its "weight" (in this case, its 0th moment/area/volume)
342 
344  double vol = target_mesh_.cell_volume(targetCellID);
345  int nsummed = 0;
346 
347  // Loop over source cells
348  for (int j = 0; j < nsrccells; ++j) {
349 
350  // Get source cell and the intersection weights
351  int srccell = sources_and_weights[j].entityID;
352  std::vector<double> xsect_weights = sources_and_weights[j].weights;
353  double xsect_volume = xsect_weights[0];
354 
355  double eps = 1e-12;
356  if (xsect_volume/vol <= eps) continue; // no intersection
357 
358  // Obtain source cell centroid
359  Point<D> src_centroid;
360  if (field_type_ == Field_type::MESH_FIELD){
361  source_mesh_.cell_centroid(srccell, &src_centroid);
362  }
363  else if (field_type_ == Field_type::MULTIMATERIAL_FIELD){
364 #ifdef HAVE_TANGRAM
365  int nmats = source_state_.cell_get_num_mats(srccell);
366  std::vector<int> cellmats;
367  source_state_.cell_get_mats(srccell, &cellmats);
368 
369  if (!nmats || (nmats == 1 && cellmats[0] == matid_))
370  { // pure cell
371  source_mesh_.cell_centroid(srccell, &src_centroid);
372  }
373  else
374  { // multi-material cell
375  assert(interface_reconstructor_ != nullptr); // cannot be nullptr
376 
377  if (std::find(cellmats.begin(), cellmats.end(), matid_) !=
378  cellmats.end())
379  { // mixed cell containing this material
380 
381  // Obtain matpoly's for this material
382  Tangram::CellMatPoly<D> const& cellmatpoly =
383  interface_reconstructor_->cell_matpoly_data(srccell);
384  std::vector<Tangram::MatPoly<D>> matpolys =
385  cellmatpoly.get_matpolys(matid_);
386 
387  int cnt = 0;
388  for (int k = 0; k < D; k++) src_centroid[k]=0;
389 
390  // Compute centroid of all matpoly's
391  for (int j = 0; j < matpolys.size(); j++)
392  {
393  /* This code snippet should be turned on when the PR
394  with changes in Matpoly is merged to Tangram.*/
395  std::vector<double> moments = matpolys[j].moments();
396  cnt += 1;
397  for (int k = 0; k < D; k++)
398  src_centroid[k]=moments[k+1]/moments[0];
399  }
400  src_centroid = src_centroid/cnt;
401  }
402  }
403 #endif
404  }
405 
406  // Compute intersection centroid
407  Point<D> xsect_centroid;
408  for (int i = 0; i < D; ++i)
409  xsect_centroid[i] = xsect_weights[1+i]/xsect_volume; // (1st moment)/vol
410 
411  // Get correct source cell index
412  int srcindex;
413  if (field_type_ == Field_type::MESH_FIELD)
414  srcindex = srccell;
415  else if (field_type_ == Field_type::MULTIMATERIAL_FIELD)
416  srcindex = source_state_.cell_index_in_material(srccell, matid_);
417 
418  Vector<D> gradient = gradients_[srcindex];
419  Vector<D> vec = xsect_centroid - src_centroid;
420  double val = source_vals_[srcindex] + dot(gradient, vec);
421  val *= xsect_volume;
422  totalval += val;
423  wtsum0 += xsect_volume;
424  nsummed++;
425  }
426 
427  // Normalize the value by the volume of the intersection of the target cells
428  // with the source mesh. This will do the right thing for single-material
429  // and multi-material remap (conservative and constant preserving) if there
430  // is NO mismatch between source and target mesh boundaries. IF THERE IS A
431  // MISMATCH, THIS WILL PRESERVE CONSTANT VALUES BUT NOT BE CONSERVATIVE. THEN
432  // WE HAVE TO DO A SEMI-LOCAL OR GLOBAL REPAIR.
433 
434  if (nsummed)
435  totalval /= wtsum0;
436  else
437  totalval = 0.0;
438 
439  return totalval;
440 
441  } // operator()
