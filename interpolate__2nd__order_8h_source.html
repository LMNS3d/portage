254 template<typename SourceMeshType, typename TargetMeshType, typename StateType,
255  long D>
256 double Interpolate_2ndOrder<SourceMeshType, TargetMeshType,
257  StateType, CELL, D>::operator()
258  (int const targetCellID, std::vector<Weights_t> const & sources_and_weights)
259  const {
260 
261  int nsrccells = sources_and_weights.size();
262  if (!nsrccells) {
263 #ifdef DEBUG
264  std::cerr << "WARNING: No source cells contribute to target cell." <<
265  std::endl;
266 #endif
267  return 0.0;
268  }
269 
270  double totalval = 0.0;
271  double wtsum0 = 0.0;
272 
273  // contribution of the source cell is its field value weighted by
274  // its "weight" (in this case, its 0th moment/area/volume)
275 
277 
278  for (int j = 0; j < nsrccells; ++j) {
279  int srccell = sources_and_weights[j].entityID;
280  std::vector<double> xsect_weights = sources_and_weights[j].weights;
281  double xsect_volume = xsect_weights[0];
282 
283  double eps = 1e-30;
284  if (xsect_volume <= eps) continue; // no intersection
285 
286  Point<D> srccell_centroid;
287  source_mesh_.cell_centroid(srccell, &srccell_centroid);
288 
289  Point<D> xsect_centroid;
290  for (int i = 0; i < D; ++i)
291  xsect_centroid[i] = xsect_weights[1+i]/xsect_volume; // (1st moment)/vol
292 
293  Vector<D> gradient = gradients_[srccell];
294  Vector<D> vec = xsect_centroid - srccell_centroid;
295  double val = source_vals_[srccell] + dot(gradient, vec);
296  val *= xsect_volume;
297  totalval += val;
298  wtsum0 += xsect_volume;
299  }
300 
301  // Normalize the value by sum of all the 0th weights (which is the
302  // same as the total volume of the source cell)
303 
304  double vol = target_mesh_.cell_volume(targetCellID);
305  totalval /= vol;
306 
307 #ifdef DEBUG
308  static bool first = true;
309  if (first && fabs((vol-wtsum0)/vol) > 1.0e-10) {
310  std::cerr << "WARNING: Meshes may be mismatched in the neighborhood of cell " <<
311  targetCellID << " in the target mesh (and maybe other places too)\n";
312  first = false;
313  }
314 #endif
315 
316  return totalval;
317 }
