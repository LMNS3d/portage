262  // collect volumes of entities that are not masked out and sum them up
263  for (auto&& s : source_entities_) {

265  source_entities_volumes_[i] = (

279  target_entities_volumes_[i] = (
280  onwhat == Entity_kind::CELL ? target_mesh_.cell_volume(t)

283  #if DEBUG_PART_BY_PART
284  std::printf("- target_volume[%02d]: %.3f\n", t, target_entities_volumes_[i]);
285  if (i == target_part_size_ - 1)

296  // matched source cell should be in the source part
297  if (source_lookup_.count(sw.entityID))
298  intersection_volumes_[i] += sw.weights[0];

306  #endif

310  source_entities_volumes_.end(), 0.);

312  target_entities_volumes_.end(), 0.);

328  std::printf("target volume: %.3f\n", global_target_volume_);

339  // source cells as needed from different partitions to cover the
340  // target partition, UNLESS NO SOURCE CELLS COVERING THE TARGET
341  // CELLS EXIST GLOBALLY. So, at this stage, we can just check
342  // on-rank intersections only

344  // Are some source cells not fully covered by target cells?
345  // Are some target cells not fully covered by source cells

365  has_mismatch_ = true;
366 
367  if (rank_ == 0) {

369  std::fprintf(stderr, "are not fully covered by the source mesh\n");

378  // Discrepancy between intersection volume and source mesh volume PLUS
379  // Discrepancy between intersection volume and target mesh volume
380  relative_voldiff_ = relative_voldiff_source + relative_voldiff_target;
381 
382  // Collect the empty target cells in layers starting from the ones
383  // next to partially or fully covered cells. At the end of this
384  // section, partially or fully covered cells will all have a layer
385  // number of 0 and empty cell layers will have positive layer

388  empty_entities.reserve(target_part_size_);

393  auto const& i = target_relative_index_[entity];
394  if (std::abs(intersection_volumes_[i]) < epsilon_) {
395  empty_entities.emplace_back(entity);
396  is_cell_empty_[i] = true;
397  }

403 #ifdef PORTAGE_ENABLE_MPI
404  if (distributed_) {
405  global_nb_empty = 0;
406  MPI_Reduce(&nb_empty, &global_nb_empty, 1, MPI_INT, MPI_SUM, 0, mycomm_);
407  }

414  "Will assign values based on their neighborhood\n"

432  while (nb_tagged < nb_empty and nb_tagged > old_nb_tagged) {

436

438  auto const& i = target_relative_index_[entity];
439  // skip already set entities
440  if (layer_num_[i] == 0) {

443 
444  for (auto&& neigh : neighbors) {

446  // At least one neighbor has some material or will
447  // receive some material (indicated by having a +ve
448  // layer number)

451  break;
452  }
453  }
454  }

460  layer_num_[t] = nb_layers + 1;
461  }

464  empty_layers_.emplace_back(current_layer_entities);

541  bool fix_mismatch_meshvar(std::string const & src_var_name,
542  std::string const & trg_var_name,
543  double global_lower_bound,
544  double global_upper_bound,
545  double conservation_tol,
546  int maxiter,
547  Partial_fixup_type partial_fixup_type,
548  Empty_fixup_type empty_fixup_type) const {
549 
550  // valid only for part-by-part scenario
551  assert(do_part_by_part_);
552  static bool hit_lower_bound = false;
553  static bool hit_higher_bound = false;
554 
555  // Now process remap variables
556  // WARNING: absolute indexing
557  double const* source_data;
558  double* target_data;
559 
560  source_state_.mesh_get_data(onwhat, src_var_name, &source_data);
561  target_state_.mesh_get_data(onwhat, trg_var_name, &target_data);
562 
563  if (partial_fixup_type == LOCALLY_CONSERVATIVE) {
564  // In interpolate step, we divided the accumulated integral (U)
565  // in a target cell by the intersection volume (v_i) instead of
566  // the target cell volume (v_c) to give a target field of u_t =
567  // U/v_i. In partially filled cells, this will preserve a
568  // constant source field but fill the cell with too much material
569  // (this is the equivalent of requesting Partial_fixup_type::CONSTANT).
570  // To restore conservation (as requested by
571  // Partial_fixup_type::LOCALLY_CONSERVATIVE), we undo the division by
572  // the intersection volume and then divide by the cell volume
573  // (u'_t = U/v_c = u_t*v_i/v_c). This does not affect the values
574  // in fully filled cells
575 
576  for (auto&& entity : target_entities_) {
577  auto const& t = target_relative_index_.at(entity);
578  if (not is_cell_empty_[t]) {
579 
580  #if DEBUG_PART_BY_PART
581  std::printf("fixing target_data[%d] with locally conservative fixup\n", entity);
582  std::printf("= before: %.3f", target_data[entity]);
583  #endif
584 
585  auto const relative_voldiff =
586  std::abs(intersection_volumes_[t] - target_entities_volumes_[t])
587  / target_entities_volumes_[t];
588 
589  if (relative_voldiff > tolerance_) {
590  target_data[entity] *= intersection_volumes_[t] / target_entities_volumes_[t];
591  }
592  #if DEBUG_PART_BY_PART
593  std::printf(", after: %.3f\n", target_data[entity]);
594  #endif
595  }
596  }
597  }
598 
599 
600  if (empty_fixup_type != LEAVE_EMPTY) {
601  // Do something here to populate fully uncovered target
602  // cells. We have layers of empty cells starting out from fully
603  // or partially populated cells. We will assign every empty cell
604  // in a layer the average value of all its populated neighbors.
605  // IN A DISTRIBUTED MESH IT _IS_ POSSIBLE THAT AN EMPTY ENTITY
606  // WILL NOT HAVE ANY OWNED NEIGHBOR IN THIS PARTITION THAT HAS
607  // MEANINGFUL DATA TO EXTRAPOLATE FROM (we remap data only to
608  // owned entities)
609  int current_layer_number = 1;
610 
611  for (auto const& current_layer : empty_layers_) {
612  for (auto&& entity : current_layer) {
613 
614  double averaged_value = 0.;
615  int nb_extrapol = 0;
616  auto neighbors =
617  get_target_filtered_neighbors<Entity_type::PARALLEL_OWNED>(entity);
618 
619  for (auto&& neigh : neighbors) {
620  auto const& i = target_relative_index_.at(neigh);
621  if (layer_num_[i] < current_layer_number) {
622  averaged_value += target_data[neigh];
623  nb_extrapol++;
624  }
625  }
626  if (nb_extrapol > 0) {
627  averaged_value /= nb_extrapol;
628  }
629  #if DEBUG_PART_BY_PART
630  else {
631  std::fprintf(stderr,
632  "No owned neighbors of empty entity to extrapolate data from\n"
633  );
634  }
635  #endif
636  target_data[entity] = averaged_value;
637  }
638  current_layer_number++;
639  }
640  }
641 
642  // if the fixup scheme is constant or locally conservative then we're done
643  if (partial_fixup_type == CONSTANT or partial_fixup_type == LOCALLY_CONSERVATIVE) {
644  return true;
645  } else if (partial_fixup_type == SHIFTED_CONSERVATIVE) {
646 
647  // At this point assume that all cells have some value in them
648  // for the variable
649  // Now compute the net discrepancy between integrals over source
650  // and target. Excess comes from target cells not fully covered by
651  // source cells and deficit from source cells not fully covered by
652  // target cells
653  double source_sum = 0.;
654  double target_sum = 0.;
655 
656  for (auto&& s : source_entities_) {
657  auto const& i = source_relative_index_.at(s);
658  source_sum += source_data[s] * source_entities_volumes_[i];
659  }
660 
661  for (auto&& t : target_entities_) {
662  auto const& i = target_relative_index_.at(t);
663  target_sum += target_data[t] * target_entities_volumes_[i];
664  }
665 
666  double global_source_sum = source_sum;
667  double global_target_sum = target_sum;
668 
669 #ifdef PORTAGE_ENABLE_MPI
670  if (distributed_) {
671  MPI_Allreduce(
672  &source_sum, &global_source_sum, 1, MPI_DOUBLE, MPI_SUM, mycomm_
673  );
674 
675  MPI_Allreduce(
676  &target_sum, &global_target_sum, 1, MPI_DOUBLE, MPI_SUM, mycomm_
677  );
678  }
679 #endif
680 
681  double absolute_diff = global_target_sum - global_source_sum;
682  double relative_diff = absolute_diff / global_source_sum;
683 
684  if (std::abs(relative_diff) < conservation_tol) {
685  return true; // discrepancy is too small - nothing to do
686  }
687 
688  // Now redistribute the discrepancy among cells in proportion to
689  // their volume. This will restore conservation and if the
690  // original distribution was a constant it will make the field a
691  // slightly different constant. Do multiple iterations because
692  // we may have some leftover quantity that we were not able to
693  // add/subtract from some cells in any given iteration. We don't
694  // expect that process to take more than two iterations at the
695  // most.
696 
697  double adj_target_volume;
698  double global_adj_target_volume;
699  double global_covered_target_volume;
700 
701  if (empty_fixup_type == Empty_fixup_type::LEAVE_EMPTY) {
702  double covered_target_volume = 0.;
703  for (auto&& entity : target_entities_) {
704  auto const& t = target_relative_index_.at(entity);
705  if (not is_cell_empty_[t]) {
706  covered_target_volume += target_entities_volumes_[t];
707  }
708  }
709  global_covered_target_volume = covered_target_volume;
710 #ifdef PORTAGE_ENABLE_MPI
711  if (distributed_) {
712  MPI_Allreduce(
713  &covered_target_volume, &global_covered_target_volume,
714  1, MPI_DOUBLE, MPI_SUM, mycomm_
715  );
716  }
717 #endif
718  adj_target_volume = covered_target_volume;
719  global_adj_target_volume = global_covered_target_volume;
720  }
721  else {
722  adj_target_volume = std::accumulate(
723  target_entities_volumes_.begin(), target_entities_volumes_.end(), 0
724  );
725  global_adj_target_volume = global_target_volume_;
726  }
727 
728  // get the right entity type
729  auto target_entity_type = [&](int entity) -> Entity_type {
730  return (onwhat == Entity_kind::CELL ? target_mesh_.cell_get_type(entity)
731  : target_mesh_.node_get_type(entity));
732  };
733 
734  // sort of a "unit" discrepancy or difference per unit volume
735  double udiff = absolute_diff / global_adj_target_volume;
736 
737  int iter = 0;
738  while (std::abs(relative_diff) > conservation_tol and iter < maxiter) {
739 
740  for (auto&& entity : target_entities_) {
741  auto const& t = target_relative_index_.at(entity);
742  bool is_owned = target_entity_type(entity) == Entity_type::PARALLEL_OWNED;
743  bool should_fix = (empty_fixup_type != LEAVE_EMPTY or not is_cell_empty_[t]);
744 
745  if (is_owned and should_fix) {
746  if ((target_data[entity] - udiff) < global_lower_bound) {
747  // Subtracting the full excess will make this cell violate the
748  // lower bound. So subtract only as much as will put this cell
749  // exactly at the lower bound
750  target_data[entity] = global_lower_bound;
751 
752  if (not hit_lower_bound) {
753  std::fprintf(stderr,
754  "Hit lower bound for cell %d (and maybe other ones) on rank %d\n",
755  t, rank_
756  );
757  hit_lower_bound = true;
758  }
759  // this cell is no longer in play for adjustment - so remove its
760  // volume from the adjusted target_volume
761  adj_target_volume -= target_entities_volumes_[t];
762 
763  } else if ((target_data[entity] - udiff) > global_upper_bound) { // udiff < 0
764  // Adding the full deficit will make this cell violate the
765  // upper bound. So add only as much as will put this cell
766  // exactly at the upper bound
767  target_data[entity] = global_upper_bound;
768 
769  if (not hit_higher_bound) {
770  std::fprintf(stderr,
771  "Hit upper bound for cell %d (and maybe other ones) on rank %d\n",
772  t, rank_
773  );
774  hit_higher_bound = true;
775  }
776 
777  // this cell is no longer in play for adjustment - so remove its
778  // volume from the adjusted target_volume
779  adj_target_volume -= target_entities_volumes_[t];
780  } else {
781  // This is the equivalent of
782  // [curval*cellvol - diff*cellvol/meshvol]
783  // curval = ---------------------------------------
784  // cellvol
785  target_data[entity] -= udiff;
786  }
787  } // only non-empty cells
788  } // iterate through mesh cells
789 
790  // Compute the new integral over all processors
791  target_sum = 0.;
792  for (auto&& entity : target_entities_) {
793  auto const& t = target_relative_index_.at(entity);
794  target_sum += target_entities_volumes_[t] * target_data[entity];
795  }
796 
797  global_target_sum = target_sum;
798 #ifdef PORTAGE_ENABLE_MPI
799  if (distributed_) {
800  MPI_Allreduce(
801  &target_sum, &global_target_sum, 1, MPI_DOUBLE, MPI_SUM, mycomm_
802  );
803  }
804 #endif
805 
806  // If we did not hit lower or upper bounds, this should be
807  // zero after the first iteration. If we did hit some bounds,
808  // then recalculate the discrepancy and discrepancy per unit
809  // volume, but only taking into account volume of cells that
810  // are not already at the bounds - if we use the entire mesh
811  // volume for the recalculation, the convergence slows down
812  absolute_diff = global_target_sum - global_source_sum;
813  global_adj_target_volume = adj_target_volume;
814 
815 #ifdef PORTAGE_ENABLE_MPI
816  if (distributed_) {
817  MPI_Allreduce(
818  &adj_target_volume, &global_adj_target_volume,
819  1, MPI_DOUBLE, MPI_SUM, mycomm_
820  );
821  }
822 #endif
823 
824  udiff = absolute_diff / global_adj_target_volume;
825  relative_diff = absolute_diff / global_source_sum;
826 
827  // Now reset adjusted target mesh volume to be the full volume
828  // in preparation for the next iteration
829  global_adj_target_volume = (
830  empty_fixup_type == LEAVE_EMPTY ? global_covered_target_volume
831  : global_target_volume_
832  );
833 
834  iter++;
835  } // while leftover is not zero
836 
837  if (std::abs(relative_diff) > conservation_tol) {
838  if (rank_ == 0) {
839  std::fprintf(stderr,
840  "Redistribution not entirely successfully for variable %s\n"
841  "Relative conservation error is %f\n"
842  "Absolute conservation error is %f\n",
843  src_var_name.data(), relative_diff, absolute_diff
844  );
845  return false;
846  }
847  }
848 
849  return true;
850  } else {
851  std::fprintf(stderr, "Unknown Partial fixup type\n");
852  return false;
853  }
854  }
