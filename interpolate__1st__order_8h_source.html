336  double operator() (int const targetCellID,
337  std::vector<Weights_t> const & sources_and_weights) const
338  {
339  int nsrccells = sources_and_weights.size();
340  if (!nsrccells) return 0.0;
341 
342  // contribution of the source cell is its field value weighted by
343  // its "weight" (in this case, its 0th moment/area/volume)
344 
345  double val = 0.0;
346  double wtsum0 = 0.0;
347  double vol = target_mesh_.cell_volume(targetCellID);
348 
349  int nsummed = 0;
350  if (field_type_ == Field_type::MESH_FIELD) {
351  for (auto const& wt : sources_and_weights) {
352  int srccell = wt.entityID;
353  std::vector<double> pair_weights = wt.weights;
354  if (pair_weights[0]/vol < num_tols_.min_relative_volume)
355  continue; // skip small intersections
356  val += source_vals_[srccell] * pair_weights[0];
357  wtsum0 += pair_weights[0];
358  nsummed++;
359  }
360  } else if (field_type_ == Field_type::MULTIMATERIAL_FIELD) {
361  for (auto const& wt : sources_and_weights) {
362  int srccell = wt.entityID;
363  std::vector<double> pair_weights = wt.weights;
364  if (pair_weights[0]/vol < num_tols_.min_relative_volume)
365  continue; // skip small intersections
366  int matcell = source_state_.cell_index_in_material(srccell, matid_);
367  val += source_vals_[matcell] * pair_weights[0]; // 1st order
368  wtsum0 += pair_weights[0];
369  nsummed++;
370  }
371  }
372 
373  // Normalize the value by the volume of the intersection of the
374  // target cells with the source mesh. This will do the right thing
375  // for single-material and multi-material remap (conservative and
376  // constant preserving) if there is NO mismatch between source and
377  // target mesh boundaries. IF THERE IS A MISMATCH, THIS WILL
378  // PRESERVE CONSTANT VALUES BUT NOT BE CONSERVATIVE. THEN WE HAVE
379  // TO DO A SEMI-LOCAL OR GLOBAL REPAIR.
380  if (nsummed)
381  val /= wtsum0;
382  else
383  val = 0.0;
384 
385  return val;
386  } // operator()
