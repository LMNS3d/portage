318  double operator() (int const targetCellID,
319  std::vector<Weights_t> const & sources_and_weights) const
320  {
321  int nsrccells = sources_and_weights.size();
322  if (!nsrccells) return 0.0;
323 
324  // contribution of the source cell is its field value weighted by
325  // its "weight" (in this case, its 0th moment/area/volume)
326 
327  double val = 0.0;
328  double wtsum0 = 0.0;
329  double vol = target_mesh_.cell_volume(targetCellID);
330 
331  int nsummed = 0;
332  if (field_type_ == Field_type::MESH_FIELD) {
333  for (auto const& wt : sources_and_weights) {
334  int srccell = wt.entityID;
335  std::vector<double> pair_weights = wt.weights;
336  if (pair_weights[0]/vol < 1.0e-12) continue; // skip small intersections
337  val += source_vals_[srccell] * pair_weights[0];
338  wtsum0 += pair_weights[0];
339  nsummed++;
340  }
341  } else if (field_type_ == Field_type::MULTIMATERIAL_FIELD) {
342  for (auto const& wt : sources_and_weights) {
343  int srccell = wt.entityID;
344  std::vector<double> pair_weights = wt.weights;
345  if (pair_weights[0]/vol < 1.0e-12) continue; // skip small intersections
346  int matcell = source_state_.cell_index_in_material(srccell, matid_);
347  val += source_vals_[matcell] * pair_weights[0]; // 1st order
348  wtsum0 += pair_weights[0];
349  nsummed++;
350  }
351  }
352 
353  // Normalize the value by the volume of the intersection of the
354  // target cells with the source mesh. This will do the right thing
355  // for single-material and multi-material remap (conservative and
356  // constant preserving) if there is NO mismatch between source and
357  // target mesh boundaries. IF THERE IS A MISMATCH, THIS WILL
358  // PRESERVE CONSTANT VALUES BUT NOT BE CONSERVATIVE. THEN WE HAVE
359  // TO DO A SEMI-LOCAL OR GLOBAL REPAIR.
360  if (nsummed)
361  val /= wtsum0;
362  else
363  val = 0.0;
364 
365  return val;
366  } // operator()
