248 template<typename SourceMeshType, typename TargetMeshType, typename StateType,
249  long D>
250 double Interpolate_1stOrder<SourceMeshType, TargetMeshType, StateType,
251  CELL, D> :: operator()
252  (int const targetCellID,
253  std::vector<Weights_t> const & sources_and_weights) const {
254 
255  int nsrccells = sources_and_weights.size();
256  if (!nsrccells) {
257 #ifdef DEBUG
258  std::cerr << "WARNING: No source cells contribute to target cell." <<
259  std::endl;
260 #endif
261  return 0.0;
262  }
263 
264  // contribution of the source cell is its field value weighted by
265  // its "weight" (in this case, its 0th moment/area/volume)
266  double val = 0.0;
267  double wtsum0 = 0.0;
268  for (auto const& wt : sources_and_weights) {
269  int srccell = wt.entityID;
270  std::vector<double> pair_weights = wt.weights;
271  val += source_vals_[srccell] * pair_weights[0]; // 1st order
272  wtsum0 += pair_weights[0];
273  }
274 
275  // Normalize the value by sum of all the weights
276 
277  double vol = target_mesh_.cell_volume(targetCellID);
278  val /= vol;
279 
280 #ifdef DEBUG
281  static bool first = true;
282  if (first && fabs((vol-wtsum0)/vol) > 1.0e-10) {
283  std::cerr << "WARNING: Meshes may be mismatched in the neighborhood of cell " <<
284  targetCellID << " in the target mesh (and maybe other places too)\n";
285  first = false;
286  }
287 #endif
288 
289  return val;
290 }
