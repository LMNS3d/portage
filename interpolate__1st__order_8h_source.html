268 template<typename SourceMeshType, typename TargetMeshType, typename StateType,
269  long D>
270 double Interpolate_1stOrder<SourceMeshType, TargetMeshType, StateType,
271  CELL, D> :: operator()
272  (int const targetCellID,
273  std::vector<Weights_t> const & sources_and_weights) const {
274 
275  int nsrccells = sources_and_weights.size();
276  if (!nsrccells) {
277 #ifdef DEBUG
278  std::cerr << "WARNING: No source cells contribute to target cell." <<
279  std::endl;
280 #endif
281  return 0.0;
282  }
283 
284  // contribution of the source cell is its field value weighted by
285  // its "weight" (in this case, its 0th moment/area/volume)
286  double val = 0.0;
287  double wtsum0 = 0.0;
288  for (auto const& wt : sources_and_weights) {
289  int srccell = wt.entityID;
290  std::vector<double> pair_weights = wt.weights;
291  val += source_vals_[srccell] * pair_weights[0]; // 1st order
292  wtsum0 += pair_weights[0];
293  }
294 
295  // Normalize the value by sum of all the weights
296 
297  double vol = target_mesh_.cell_volume(targetCellID);
298  val /= vol;
299 
300 #ifdef DEBUG
301  static bool first = true;
302  if (first && fabs((vol-wtsum0)/vol) > 1.0e-10) {
303  std::cerr << "WARNING: Meshes may be mismatched in the neighborhood of cell " <<
304  targetCellID << " in the target mesh (and maybe other places too)\n";
305  first = false;
306  }
307 #endif
308 
309  return val;
310 }

393  NODE, D> :: operator()
394  (int const targetNodeID,

408  // moment/area/volume of its intersection with the target dual cell)
409

413  int srcnode = wt.entityID;

415  val += source_vals_[srcnode] * pair_weights[0]; // 1st order
416  wtsum0 += pair_weights[0];

422  val /= vol;
