311  double operator() (int const targetCellID,
312  std::vector<Weights_t> const & sources_and_weights) const
313  {
314  int nsrccells = sources_and_weights.size();
315  if (!nsrccells) return 0.0;
316 
317  // contribution of the source cell is its field value weighted by
318  // its "weight" (in this case, its 0th moment/area/volume)
319 
320  double val = 0.0;
321  double wtsum0 = 0.0;
322  double vol = target_mesh_.cell_volume(targetCellID);
323 
324  int nsummed = 0;
325  if (field_type_ == Field_type::MESH_FIELD) {
326  for (auto const& wt : sources_and_weights) {
327  int srccell = wt.entityID;
328  std::vector<double> pair_weights = wt.weights;
329  if (pair_weights[0]/vol < 1.0e-12) continue; // skip small intersections
330  val += source_vals_[srccell] * pair_weights[0];
331  wtsum0 += pair_weights[0];
332  nsummed++;
333  }
334  } else if (field_type_ == Field_type::MULTIMATERIAL_FIELD) {
335  for (auto const& wt : sources_and_weights) {
336  int srccell = wt.entityID;
337  std::vector<double> pair_weights = wt.weights;
338  if (pair_weights[0]/vol < 1.0e-12) continue; // skip small intersections
339  int matcell = source_state_.cell_index_in_material(srccell, matid_);
340  val += source_vals_[matcell] * pair_weights[0]; // 1st order
341  wtsum0 += pair_weights[0];
342  nsummed++;
343  }
344  }
345 
346  // Normalize the value by the volume of the intersection of the
347  // target cells with the source mesh. This will do the right thing
348  // for single-material and multi-material remap (conservative and
349  // constant preserving) if there is NO mismatch between source and
350  // target mesh boundaries. IF THERE IS A MISMATCH, THIS WILL
351  // PRESERVE CONSTANT VALUES BUT NOT BE CONSERVATIVE. THEN WE HAVE
352  // TO DO A SEMI-LOCAL OR GLOBAL REPAIR.
353  if (nsummed)
354  val /= wtsum0;
355  else
356  val = 0.0;
357 
358  return val;
359  } // operator()
