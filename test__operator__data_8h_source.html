109 // function to deform reference points
110 
111 template<size_t dim>
112 std::vector<Point<dim>> deform_points(const std::vector<Point<dim>> &points,
113  const std::vector<std::vector<double>> &matrix) {
114  auto result(points);
115  for (int i=0; i<points.size(); i++) {
116  for (int j=0; j<dim; j++) {
117  result[i][j] = 0.;
118  }
119  }
120  for (int i=0; i<points.size(); i++) {
121  for (int j=0; j<dim; j++) {
122  for (int k=0; k<dim; k++) {
123  result[i][j] += matrix[j][k]*points[i][k];
124  }
125  }
126  }
127  return result;
128 }

163 template<Portage::Meshfree::Basis::Type type, size_t dim>
164 typename Portage::Meshfree::Basis::Traits<type,dim>::array_t
165 makeTranslatedExact(
166  typename Portage::Meshfree::Basis::Traits<type,dim>::array_t &values,
167  Point<dim> &point)
168 {
169  auto tf = Portage::Meshfree::Basis::transfactor<dim>(type, point);
170  typename Portage::Meshfree::Basis::Traits<type,dim>::array_t tex;
171  for (int i=0; i<tf.size(); i++) tex[i] = 0.;
172  for (int i=0; i<tf.size(); i++)
173  for (int j=0; j<tf.size(); j++)
174  tex[i] += tf[i][j]*values[j];
175  return tex;
176 }
