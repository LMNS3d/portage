75  template <class State_Wrapper>
76  void initialize(State_Wrapper const & input,
77  std::vector<std::string> var_names)
78  {
79  for (unsigned int i=0; i<var_names.size(); i++)
80  {
81  std::string varname = var_names[i]; // get_data wants const string
82  Entity_kind entity = input.get_entity(varname);
83  int dataSize = input.get_data_size(entity, varname);
84  T const *data;
85 
86  input.get_data(entity, varname, &data);
87  std::shared_ptr<std::vector<T>> field = std::make_shared<std::vector<T>>();
88  field->resize(dataSize);
89  std::copy(data, data+dataSize, field->begin());
90  state_.push_back(field);
91  name_map_[varname] = state_.size() - 1;
92  }
93  }
