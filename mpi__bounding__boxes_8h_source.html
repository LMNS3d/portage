606  void setSendRecvCounts(comm_info_t* info,
607  const int commSize,
608  const std::vector<bool>& sendFlags,
609  const int sourceNum,
610  const int sourceNumOwned)
611  {
612  // Set my counts of all entities and owned entities
613  info->sourceNum = sourceNum;
614  info->sourceNumOwned = sourceNumOwned;
615 
616  // Each rank will tell each other rank how many indexes it is going to send it
617  info->sendCounts.resize(commSize);
618  info->recvCounts.resize(commSize);
619  for (unsigned int i=0; i<commSize; i++)
620  info->sendCounts[i] = sendFlags[i] ? info->sourceNum : 0;
621  MPI_Alltoall(&(info->sendCounts[0]), 1, MPI_INT,
622  &(info->recvCounts[0]), 1, MPI_INT, comm_);
623 
624  // Each rank will tell each other rank how many owned indexes it is going to send it
625  info->sendOwnedCounts.resize(commSize);
626  info->recvOwnedCounts.resize(commSize);
627  for (unsigned int i=0; i<commSize; i++)
628  info->sendOwnedCounts[i] = sendFlags[i] ? info->sourceNumOwned : 0;
629  MPI_Alltoall(&(info->sendOwnedCounts[0]), 1, MPI_INT,
630  &(info->recvOwnedCounts[0]), 1, MPI_INT, comm_);
631 
632  // Compute the total number of indexes this rank will receive from all ranks
633  for (unsigned int i=0; i<commSize; i++)
634  info->newNum += info->recvCounts[i];
635  for (unsigned int i=0; i<commSize; i++)
636  info->newNumOwned += info->recvOwnedCounts[i];
637  } // setSendRecvCounts

651  template<typename T>
652  void sendField(const comm_info_t& info,
653  const int commRank, const int commSize,
654  const MPI_Datatype mpiType, const int stride,
655  const std::vector<T>& sourceData,
656  std::vector<T>* newData)
657  {
658  // Perform two rounds of sends: the first for owned cells, and the second for ghost cells
659  std::vector<int> recvGhostCounts(commSize);
660  std::vector<int> sendGhostCounts(commSize);
661 
662  for (unsigned int i=0; i<commSize; i++)
663  {
664  recvGhostCounts[i] = info.recvCounts[i] - info.recvOwnedCounts[i];
665  sendGhostCounts[i] = info.sendCounts[i] - info.sendOwnedCounts[i];

703  template<typename T>
704  void sendData(const int commRank, const int commSize,
705  const MPI_Datatype mpiType, const int stride,
706  const int sourceStart, const int sourceEnd,
707  const int newStart,
708  const std::vector<int>& curSendCounts,
709  const std::vector<int>& curRecvCounts,
710  const std::vector<T>& sourceData,
711  std::vector<T>* newData)
712  {
713  // Each rank will do a non-blocking receive from each rank from
714  // which it will receive data values
715  int writeOffset = newStart;
716  int myOffset;
717  std::vector<MPI_Request> requests;
718  for (unsigned int i=0; i<commSize; i++)
719  {
720  if ((i != commRank) && (curRecvCounts[i] > 0))
721  {
722  MPI_Request request;
723  MPI_Irecv((void *)&((*newData)[stride*writeOffset]),
724  stride*curRecvCounts[i], mpiType, i,
725  MPI_ANY_TAG, comm_, &request);
726  requests.push_back(request);
727  }
728  else if (i == commRank)
729  {
730  myOffset = writeOffset;
731  }
732  writeOffset += curRecvCounts[i];
733  }
734 
735  // Copy data values that will stay on this rank into the
736  // proper place in the new vector
737  if (curRecvCounts[commRank] > 0)
738  std::copy(sourceData.begin() + stride*sourceStart,
739  sourceData.begin() + stride*sourceEnd,
740  newData->begin() + stride*myOffset);
741 
742  // Each rank will send its data values to appropriate ranks
743  for (unsigned int i=0; i<commSize; i++)
744  {
745  if ((i != commRank) && (curSendCounts[i] > 0))
746  {
747  MPI_Send((void *)&(sourceData[stride*sourceStart]),
748  stride*curSendCounts[i], mpiType, i, 0, comm_);
749  }
750  }
751 
752  // Wait for all receives to complete
753  if (requests.size() > 0)
754  {
755  std::vector<MPI_Status> statuses(requests.size());
756  MPI_Waitall(requests.size(), &(requests[0]), &(statuses[0]));
757  }
758  } // sendData

779  targetBoundingBoxes[2*dim*commRank+i+1] = -std::numeric_limits<double>::max();
780  }
781  for (unsigned int c=0; c<targetNumOwnedCells; ++c)
782  {
783  std::vector<int> nodes;
784  target_mesh.cell_get_nodes(c, &nodes);
785  int cellNumNodes = nodes.size();
786  for (unsigned int j=0; j<cellNumNodes; ++j)
787  {
788  int n = nodes[j];
789  // ugly hack, since dim is not known at compile time
790  if (dim == 3)
791  {
792  Point<3> nodeCoord;
793  target_mesh.node_get_coordinates(n, &nodeCoord);
794  for (unsigned int k=0; k<dim; ++k)
795  {
796  if (nodeCoord[k] < targetBoundingBoxes[2*dim*commRank+2*k])
797  targetBoundingBoxes[2*dim*commRank+2*k] = nodeCoord[k];
798  if (nodeCoord[k] > targetBoundingBoxes[2*dim*commRank+2*k+1])
799  targetBoundingBoxes[2*dim*commRank+2*k+1] = nodeCoord[k];
800  }
801  }
802  else if (dim == 2)
803  {
804  Point<2> nodeCoord;
805  target_mesh.node_get_coordinates(n, &nodeCoord);
806  for (unsigned int k=0; k<dim; ++k)
807  {
808  if (nodeCoord[k] < targetBoundingBoxes[2*dim*commRank+2*k])
809  targetBoundingBoxes[2*dim*commRank+2*k] = nodeCoord[k];
810  if (nodeCoord[k] > targetBoundingBoxes[2*dim*commRank+2*k+1])
811  targetBoundingBoxes[2*dim*commRank+2*k+1] = nodeCoord[k];
812  }
813  }
814  } // for j
815  } // for c
816 
817  // Get the source mesh properties and cordinates
818  int sourceNumOwnedCells = source_mesh_flat.num_owned_cells();
819  std::vector<double>& sourceCoords = source_mesh_flat.get_coords();
820 
821  // Compute the bounding box for the source mesh on this rank
822  std::vector<double> sourceBoundingBox(2*dim);
823  for (unsigned int i=0; i<2*dim; i+=2)
824  {
825  sourceBoundingBox[i+0] = std::numeric_limits<double>::max();
826  sourceBoundingBox[i+1] = -std::numeric_limits<double>::max();
827  }
828  for (unsigned int c=0; c<sourceNumOwnedCells; ++c)
829  {
830  std::vector<int> nodes;
831  source_mesh_flat.cell_get_nodes(c, &nodes);
832  int cellNumNodes = nodes.size();
833  for (unsigned int j=0; j<cellNumNodes; ++j)
834  {
835  int n = nodes[j];
836  for (unsigned int k=0; k<dim; ++k)
837  {
838  if (sourceCoords[n*dim+k] < sourceBoundingBox[2*k])
839  sourceBoundingBox[2*k] = sourceCoords[n*dim+k];
840  if (sourceCoords[n*dim+k] > sourceBoundingBox[2*k+1])
841  sourceBoundingBox[2*k+1] = sourceCoords[n*dim+k];
842  }
843  } // for j
844  } // for c
845 
846  // Broadcast the target bounding boxes so that each rank knows the bounding boxes for all ranks
847  for (unsigned int i=0; i<commSize; i++)
848  MPI_Bcast(&(targetBoundingBoxes[0])+i*2*dim, 2*dim, MPI_DOUBLE, i, comm_);
849 
850 
851  // Offset the source bounding boxes by a fudge factor so that we don't send source data to a rank
852  // with a target bounding box that is incident to but not overlapping the source bounding box
853  const double boxOffset = 2.0*std::numeric_limits<double>::epsilon();
854  double min2[dim], max2[dim];
855  for (unsigned int k=0; k<dim; ++k)
856  {
857  min2[k] = sourceBoundingBox[2*k]+boxOffset;
858  max2[k] = sourceBoundingBox[2*k+1]-boxOffset;
859  }
860 
861  // For each target rank with a bounding box that overlaps the bounding box for this rank's partition
862  // of the source mesh, we will send it all our source cells; otherwise, we will send it nothing
863  for (unsigned int i=0; i<commSize; ++i)
864  {
865  double min1[dim], max1[dim];
866  bool sendThis = true;
867  for (unsigned int k=0; k<dim; ++k)
868  {
869  min1[k] = targetBoundingBoxes[2*dim*i+2*k]+boxOffset;
870  max1[k] = targetBoundingBoxes[2*dim*i+2*k+1]-boxOffset;
871  sendThis = sendThis &&
872  ((min1[k] <= min2[k] && min2[k] <= max1[k]) ||
873  (min2[k] <= min1[k] && min1[k] <= max2[k]));
874  }
875 
876  sendFlags[i] = sendThis;
877  }
