496  void setInfo(comm_info_t* info,
497  const int commSize,
498  const std::vector<bool>& sendFlags,
499  const int sourceNum,
500  const int sourceNumOwned)
501  {
502  // Set my counts of all entities and owned entities
503  info->sourceNum = sourceNum;
504  info->sourceNumOwned = sourceNumOwned;
505 
506  // Each rank will tell each other rank how many indexes it is going to send it
507  info->sendCounts.resize(commSize);
508  info->recvCounts.resize(commSize);
509  for (unsigned int i=0; i<commSize; i++)
510  info->sendCounts[i] = sendFlags[i] ? info->sourceNum : 0;
511  MPI_Alltoall(&(info->sendCounts[0]), 1, MPI_INT,
512  &(info->recvCounts[0]), 1, MPI_INT, MPI_COMM_WORLD);
513 
514  // Each rank will tell each other rank how many owned indexes it is going to send it
515  info->sendOwnedCounts.resize(commSize);
516  info->recvOwnedCounts.resize(commSize);
517  for (unsigned int i=0; i<commSize; i++)
518  info->sendOwnedCounts[i] = sendFlags[i] ? info->sourceNumOwned : 0;
519  MPI_Alltoall(&(info->sendOwnedCounts[0]), 1, MPI_INT,
520  &(info->recvOwnedCounts[0]), 1, MPI_INT, MPI_COMM_WORLD);
521 
522  // Compute the total number of indexes this rank will receive from all ranks
523  for (unsigned int i=0; i<commSize; i++)
524  info->newNum += info->recvCounts[i];
525  for (unsigned int i=0; i<commSize; i++)
526  info->newNumOwned += info->recvOwnedCounts[i];
527  } // setInfo

596  template<typename T>
597  void moveData(const int commRank, const int commSize,
598  const MPI_Datatype mpiType, const int stride,
599  const int sourceStart, const int sourceEnd,
600  const int newStart,
601  const std::vector<int>& curSendCounts,
602  const std::vector<int>& curRecvCounts,
603  const std::vector<T>& sourceData,
604  std::vector<T>* newData)
605  {
606  // Each rank will do a non-blocking receive from each rank from
607  // which it will receive data values
608  int writeOffset = newStart;
609  int myOffset;
610  std::vector<MPI_Request> requests;
611  for (unsigned int i=0; i<commSize; i++)
612  {
613  if ((i != commRank) && (curRecvCounts[i] > 0))
614  {
615  MPI_Request request;
616  MPI_Irecv((void *)&((*newData)[stride*writeOffset]),
617  stride*curRecvCounts[i], mpiType, i,
618  MPI_ANY_TAG, MPI_COMM_WORLD, &request);
619  requests.push_back(request);
620  }
621  else if (i == commRank)
622  {
623  myOffset = writeOffset;
624  }
625  writeOffset += curRecvCounts[i];
626  }
627 
628  // Copy data values that will stay on this rank into the
629  // proper place in the new vector

654  void fixListIndices(const comm_info_t& mapInfo,
655  const comm_info_t& rangeInfo,
656  const int commSize,
657  std::vector<int>* newMap)
658  {
659  // compute corrections for each rank
660  std::vector<int> ownedOffsets(commSize), ghostOffsets(commSize);
661  ownedOffsets[0] = 0;
662  std::partial_sum(rangeInfo.recvOwnedCounts.begin(),
663  rangeInfo.recvOwnedCounts.end()-1,
664  ownedOffsets.begin()+1);
665  std::vector<int> recvGhostCounts(commSize);
666  for (unsigned int i=0; i<commSize; ++i)
667  recvGhostCounts[i] =
668  rangeInfo.recvCounts[i] - rangeInfo.recvOwnedCounts[i];
669  ghostOffsets[0] = 0;
670  std::partial_sum(recvGhostCounts.begin(), recvGhostCounts.end()-1,
671  ghostOffsets.begin()+1);
672  for (unsigned int i=0; i<commSize; ++i)
673  ghostOffsets[i] += rangeInfo.newNumOwned - rangeInfo.recvOwnedCounts[i];
674 
675  // correct owned entities, one rank at a time
676  int base = 0;
677  for (unsigned int i=0; i<commSize; ++i) {
678  int ownedCount = mapInfo.recvOwnedCounts[i];
679  int ownedNodeCount = rangeInfo.recvOwnedCounts[i];
680  for (int j=0; j<ownedCount; ++j) {
681  int n = (*newMap)[base + j];
682  (*newMap)[base + j] +=
683  (n < ownedNodeCount ? ownedOffsets[i] : ghostOffsets[i]);
684  }
685  base += ownedCount;
686  } // for i
687 
688  // correct ghost entities, one rank at a time
689  for (unsigned int i=0; i<commSize; ++i) {
690  int ownedCount = mapInfo.recvOwnedCounts[i];
691  int ghostCount = mapInfo.recvCounts[i] - ownedCount;
692  int ownedNodeCount = rangeInfo.recvOwnedCounts[i];
693  for (int j=0; j<ghostCount; ++j) {
694  int n = (*newMap)[base + j];
695  (*newMap)[base + j] +=
696  (n < ownedNodeCount ? ownedOffsets[i] : ghostOffsets[i]);
697  }
698  base += ghostCount;
699  } // for i
700 
701  } // fixListIndices
