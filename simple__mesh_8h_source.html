279  /*
280  @brief Builds the cell-face-node adjacency information.
281  */
282  void build_cfn_adjacencies() {
283  // downward adjacencies
284  cell_to_node_.resize(nodes_per_cell_*num_cells_);
285  cell_to_face_.resize(faces_per_cell_*num_cells_);
286  cell_face_dirs_.resize(faces_per_cell_*num_cells_);
287  face_to_node_.resize(nodes_per_face_*num_faces_);
288  // upward adjacencies
289  node_to_cell_.resize(cells_per_node_aug_*num_nodes_);
290  face_to_cell_.resize(2*num_faces_);
291 
292  // cell adjacencies
293  for (int iz(0); iz < nz_; ++iz)
294  for (int iy(0); iy < ny_; ++iy)
295  for (int ix(0); ix < nx_; ++ix) {
296  auto thisCell = cell_index_(ix, iy, iz);
297  auto cstart = nodes_per_cell_ * thisCell;
298  auto fstart = faces_per_cell_ * thisCell;
299 
300  /*
301  Node ordering in hex:
302  z
303  7-------6 ^ y
304  /| /| | /
305  / | / | | /
306  4--|----5 | |/
307  | 3----|--2 +-------> x
308  | / | /
309  0-------1
310  */
311  cell_to_node_[cstart ] = node_index_(ix, iy, iz);
312  cell_to_node_[cstart+1] = node_index_(ix+1, iy, iz);
313  cell_to_node_[cstart+2] = node_index_(ix+1, iy+1, iz);
314  cell_to_node_[cstart+3] = node_index_(ix, iy+1, iz);
315  cell_to_node_[cstart+4] = node_index_(ix, iy, iz+1);
316  cell_to_node_[cstart+5] = node_index_(ix+1, iy, iz+1);
317  cell_to_node_[cstart+6] = node_index_(ix+1, iy+1, iz+1);
318  cell_to_node_[cstart+7] = node_index_(ix, iy+1, iz+1);
319 
320  // loop over the 8 nodes attached to this cell, and assign
321  // this cell to its cell connectivity
322  // order shouldn't matter here
323  for (int iiz(iz); iiz <= iz+1; ++iiz)
324  for (int iiy(iy); iiy <= iy+1; ++iiy)
325  for (int iix(ix); iix <= ix+1; ++iix) {
326  auto thisNode = node_index_(iix, iiy, iiz);
327  auto cnstart = cells_per_node_aug_ * thisNode;
328  auto & c_at_n = node_to_cell_[cnstart];
329  node_to_cell_[cnstart+c_at_n+1] = thisCell;
330  c_at_n++;
331  }
332 
333  /*
334  Face ordering in hex:
335  z
336  +-------+ ^ y
337  /| /| | /
338  / | 5 2/ | | /
339  +--|----+ 1| |/
340  |3 +0---|--+ +-------> x
341  | / 4 | /
342  +-------+
343 
344  The "dirs" indicate whether or not the nodes are listed in
345  a ccw (dir=+1) or cw (dir=-1) orientation when look DOWN
346  the cell's OUTWARD normal at that face. Images of each
347  face are below when we construct the face adjacencies.
348  */
349  cell_to_face_[fstart ] = xzface_index_(ix, iy, iz); // front
350  cell_face_dirs_[fstart ] = 1;
351  cell_to_face_[fstart+1] = yzface_index_(ix+1, iy, iz); // right
352  cell_face_dirs_[fstart+1] = 1;
353  cell_to_face_[fstart+2] = xzface_index_(ix, iy+1, iz); // back
354  cell_face_dirs_[fstart+2] = -1;
355  cell_to_face_[fstart+3] = yzface_index_(ix, iy, iz); // left
356  cell_face_dirs_[fstart+3] = -1;
357  cell_to_face_[fstart+4] = xyface_index_(ix, iy, iz); // bottom
358  cell_face_dirs_[fstart+4] = -1;
359  cell_to_face_[fstart+5] = xyface_index_(ix, iy, iz+1); // top
360  cell_face_dirs_[fstart+5] = 1;
361 
362  // go over the 6 faces attached to this cell, and assign
363  // this cell to its cell connectivity
364  auto cfstart = 2*xzface_index_(ix, iy, iz) + 1; // +y side of front
365  face_to_cell_[cfstart] = thisCell;
366  cfstart = 2*yzface_index_(ix+1, iy, iz); // -x side of right
367  face_to_cell_[cfstart] = thisCell;
368  cfstart = 2*xzface_index_(ix, iy+1, iz); // -y side of back
369  face_to_cell_[cfstart] = thisCell;
370  cfstart = 2*yzface_index_(ix, iy, iz) + 1; // +x side of left
371  face_to_cell_[cfstart] = thisCell;
372  cfstart = 2*xyface_index_(ix, iy, iz) + 1; // +z side of bottom
373  face_to_cell_[cfstart] = thisCell;
374  cfstart = 2*xyface_index_(ix, iy, iz+1); // -z side of top
375  face_to_cell_[cfstart] = thisCell;
376  }
377 
378  // face adjacencies
379  /* xy faces
380  3-------2
381  / /
382  / /
383  0-------1
384  */
385  for (int iz(0); iz <= nz_; ++iz)
386  for (int iy(0); iy < ny_; ++iy)
387  for (int ix(0); ix < nx_; ++ix) {
388  auto thisFace = xyface_index_(ix, iy, iz);
389  auto nstart = nodes_per_face_ * thisFace;
390  face_to_node_[nstart ] = node_index_(ix, iy, iz);
391  face_to_node_[nstart+1] = node_index_(ix+1, iy, iz);
392  face_to_node_[nstart+2] = node_index_(ix+1, iy+1, iz);
393  face_to_node_[nstart+3] = node_index_(ix, iy+1, iz);
394  }
395  /* xz faces
396  3-------2
397  | |
398  | |
399  | |
400  0-------1
401  */
402  for (int iz(0); iz < nz_; ++iz)
403  for (int iy(0); iy <= ny_; ++iy)
404  for (int ix(0); ix < nx_; ++ix) {
405  auto thisFace = xzface_index_(ix, iy, iz);
406  auto nstart = nodes_per_face_ * thisFace;
407  face_to_node_[nstart ] = node_index_(ix, iy, iz);
408  face_to_node_[nstart+1] = node_index_(ix+1, iy, iz);
409  face_to_node_[nstart+2] = node_index_(ix+1, iy, iz+1);
410  face_to_node_[nstart+3] = node_index_(ix, iy, iz+1);
411  }
412  /* yz faces
413  2
414  /|
415  / |
416  3 |
417  | 1
418  | /
419  |/
420  0
421  */
422  for (int iz(0); iz < nz_; ++iz)
423  for (int iy(0); iy < ny_; ++iy)
424  for (int ix(0); ix <= nx_; ++ix) {
425  auto thisFace = yzface_index_(ix, iy, iz);
426  auto nstart = nodes_per_face_ * thisFace;
427  face_to_node_[nstart ] = node_index_(ix, iy, iz);
428  face_to_node_[nstart+1] = node_index_(ix, iy+1, iz);
429  face_to_node_[nstart+2] = node_index_(ix, iy+1, iz+1);
430  face_to_node_[nstart+3] = node_index_(ix, iy, iz+1);
431  }
432  }
