334  void build_cfn_adjacencies_3d() {
335  // downward adjacencies
336  cell_to_node_.resize(nodes_per_cell_*num_cells_);
337  cell_to_face_.resize(faces_per_cell_*num_cells_);
338  cell_face_dirs_.resize(faces_per_cell_*num_cells_);
339  face_to_node_.resize(nodes_per_face_*num_faces_);
340  // upward adjacencies
341  node_to_cell_.resize(cells_per_node_aug_*num_nodes_);
342  face_to_cell_.resize(2*num_faces_, -1);
343 
344  // cell adjacencies
345  for (int iz(0); iz < nz_; ++iz)
346  for (int iy(0); iy < ny_; ++iy)
347  for (int ix(0); ix < nx_; ++ix) {
348  auto thisCell = cell_index_(ix, iy, iz);
349  auto cstart = nodes_per_cell_ * thisCell;
350  auto fstart = faces_per_cell_ * thisCell;
351 
352  /*
353  Node ordering in hex:
354  z
355  7-------6 ^ y
356  /| /| | /
357  / | / | | /
358  4--|----5 | |/
359  | 3----|--2 +-------> x
360  | / | /
361  0-------1
362  */
363  cell_to_node_[cstart ] = node_index_(ix, iy, iz);
364  cell_to_node_[cstart+1] = node_index_(ix+1, iy, iz);
365  cell_to_node_[cstart+2] = node_index_(ix+1, iy+1, iz);
366  cell_to_node_[cstart+3] = node_index_(ix, iy+1, iz);
367  cell_to_node_[cstart+4] = node_index_(ix, iy, iz+1);
368  cell_to_node_[cstart+5] = node_index_(ix+1, iy, iz+1);
369  cell_to_node_[cstart+6] = node_index_(ix+1, iy+1, iz+1);
370  cell_to_node_[cstart+7] = node_index_(ix, iy+1, iz+1);
371 
372  // loop over the 8 nodes attached to this cell, and assign
373  // this cell to its cell connectivity
374  // order shouldn't matter here
375  for (int iiz(iz); iiz <= iz+1; ++iiz)
376  for (int iiy(iy); iiy <= iy+1; ++iiy)
377  for (int iix(ix); iix <= ix+1; ++iix) {
378  auto thisNode = node_index_(iix, iiy, iiz);
379  auto cnstart = cells_per_node_aug_ * thisNode;
380  auto & c_at_n = node_to_cell_[cnstart];
381  node_to_cell_[cnstart+c_at_n+1] = thisCell;
382  c_at_n++;
383  }
384 
385  /*
386  Face ordering in hex:
387  z
388  +-------+ ^ y
389  /| /| | /
390  / | 5 2/ | | /
391  +--|----+ 1| |/
392  |3 +0---|--+ +-------> x
393  | / 4 | /
394  +-------+
395 
396  The "dirs" indicate whether or not the nodes are listed in
397  a ccw (dir=+1) or cw (dir=-1) orientation when look DOWN
398  the cell's OUTWARD normal at that face. Images of each
399  face are below when we construct the face adjacencies.
400  */
401  cell_to_face_[fstart ] = xzface_index_(ix, iy, iz); // front
402  cell_face_dirs_[fstart ] = 1;
403  cell_to_face_[fstart+1] = yzface_index_(ix+1, iy, iz); // right
404  cell_face_dirs_[fstart+1] = 1;
405  cell_to_face_[fstart+2] = xzface_index_(ix, iy+1, iz); // back
406  cell_face_dirs_[fstart+2] = -1;
407  cell_to_face_[fstart+3] = yzface_index_(ix, iy, iz); // left
408  cell_face_dirs_[fstart+3] = -1;
409  cell_to_face_[fstart+4] = xyface_index_(ix, iy, iz); // bottom
410  cell_face_dirs_[fstart+4] = -1;
411  cell_to_face_[fstart+5] = xyface_index_(ix, iy, iz+1); // top
412  cell_face_dirs_[fstart+5] = 1;
413 
414  // go over the 6 faces attached to this cell, and assign
415  // this cell to its cell connectivity
416  auto cfstart = 2*xzface_index_(ix, iy, iz) + 1; // +y side of front
417  face_to_cell_[cfstart] = thisCell;
418  cfstart = 2*yzface_index_(ix+1, iy, iz); // -x side of right
419  face_to_cell_[cfstart] = thisCell;
420  cfstart = 2*xzface_index_(ix, iy+1, iz); // -y side of back
421  face_to_cell_[cfstart] = thisCell;
422  cfstart = 2*yzface_index_(ix, iy, iz) + 1; // +x side of left
423  face_to_cell_[cfstart] = thisCell;
424  cfstart = 2*xyface_index_(ix, iy, iz) + 1; // +z side of bottom
425  face_to_cell_[cfstart] = thisCell;
426  cfstart = 2*xyface_index_(ix, iy, iz+1); // -z side of top
427  face_to_cell_[cfstart] = thisCell;
428  }
429 
430  // face adjacencies
431  /* xy faces
432  3-------2
433  / /
434  / /
435  0-------1
436  */
437  for (int iz(0); iz <= nz_; ++iz)
438  for (int iy(0); iy < ny_; ++iy)
439  for (int ix(0); ix < nx_; ++ix) {
440  auto thisFace = xyface_index_(ix, iy, iz);
