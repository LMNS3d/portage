86 void make_mesh_cart2d(const real_t xmin, const real_t xmax,
87  const real_t ymin, const real_t ymax,
88  const int ncellsx, const int ncellsy,
89  mesh_t & mesh) {
90  // grid spacing
91  auto dx = (xmax - xmin) / real_t(ncellsx);
92  auto dy = (ymax - ymin) / real_t(ncellsy);
93 
94  // mesh_t mesh;
95  auto num_verts = (ncellsx+1)*(ncellsy+1);
96  // this initializes storage for the number of vertices
97  mesh.init_parameters(num_verts);
98 
99  // create the verts
100  std::vector<vertex_t*> verts;
101  for (auto j = 0; j < ncellsy+1; ++j) {
102  for (auto i = 0; i < ncellsx+1; ++i) {
103  auto vert = mesh.create_vertex(
104  {xmin + dx*real_t(i), ymin + dy*real_t(j)});
105  verts.push_back(vert);
106  }
107  }
108 
109  // create the cells
110  auto ncellsx1 = ncellsx+1;
111  for (auto j = 0; j < ncellsy; ++j) {
112  for (auto i = 0; i < ncellsx; ++i) {
113  // go over verts in counter-clockwise fashion
114  auto c = mesh.create_cell({verts[i + j*ncellsx1],
115  verts[i + 1 + j*ncellsx1],
116  verts[i + 1 + (j + 1)*ncellsx1],
117  verts[i + (j + 1)*ncellsx1]});
118  }
119  }
120 
121  // this setups up the faces, edges, wedges, etc and connectivity info
122  mesh.init();
123 
124 }

386  void dual_cell_get_coordinates(int const nodeid,
387  std::vector<Portage::Point2 > *pplist) const {
388  assert(space_dimension() == 2);
389  auto thisNode = flecsi_mesh_.vertices()[nodeid];
390  pplist->clear();
391  // Loop over the corners associated with this node
392  for (auto corner : flecsi_mesh_.corners(thisNode)) {
393  std::vector<int> wedgeIds;
394  // The two wedges in this corner
395  for (auto wedge : flecsi_mesh_.wedges(corner))
396  wedgeIds.emplace_back(wedge.global_id());
397  order_wedges_ccw(&wedgeIds);
398  // Get the coordinates of each wedgeId, given the new ordering
399  for (auto wId : wedgeIds) {
400  auto w = flecsi_mesh_.wedges()[wId];
401  // push back centroid and then edge midpoint
402  auto cc = w->cell()->centroid();
403  pplist->emplace_back(toPortagePoint<2>(cc));//cc[0], cc[1]);
404  auto nc = w->edge()->midpoint();
405  pplist->emplace_back(toPortagePoint<2>(nc));//nc[0], nc[1]);
406  }
407  }
408  }

410  void order_wedges_ccw(std::vector<int> *wedgeids) const {
411  assert(wedgeids->size() == 2);
412  auto firstWedge = flecsi_mesh_.wedges()[(*wedgeids)[0]];
413  // NOTE: This mimics the order of Jali in 2d: node, face/edge, cell
414  std::vector<Portage::Point2> wcoords;
415  auto nc = firstWedge->vertex()->coordinates();
416  wcoords.emplace_back(nc[0], nc[1]);
417  auto ec = firstWedge->edge()->midpoint();
418  wcoords.emplace_back(ec[0], ec[1]);
419  auto cc = firstWedge->cell()->centroid();
420  wcoords.emplace_back(cc[0], cc[1]);
421 
422  // Ensure (*wedgeids)[0] is the first wedge in a CCW direction
423  if (not ccw(wcoords[0], wcoords[1], wcoords[2]))
424  std::swap((*wedgeids)[0], (*wedgeids)[1]);
425  }

427  // Returns true if the three 2D points (p1, p2, p3) are a counter-clockwise
428  // turn, otherwise returns false (corresponding to clockwise or collinear)
429  bool ccw(const Portage::Point2 p1,
430  const Portage::Point2 p2,
431  const Portage::Point2 p3) const {
432  return (p2[0]-p1[0])*(p3[1]-p1[1]) - (p2[1]-p1[1])*(p3[0]-p1[0]) > 0;
