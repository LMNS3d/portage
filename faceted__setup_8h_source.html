40  template<int DIM, class Mesh_Wrapper> void faceted_setup_cell
41  (const Mesh_Wrapper &mesh,
42  Portage::vector<std::vector<std::vector<double>>> &smoothing_lengths,
43  Portage::vector<Wonton::Point<DIM>> &extents,
44  double smoothing_factor, double boundary_factor)
45  {
46  if (smoothing_factor<0.) {
47  throw std::runtime_error("smoothing_factor must be non-negative");
48  }
49  if (boundary_factor<0.) {
50  throw std::runtime_error("boundary_factor must be non-negative");
51  }
52 
53  int ncells=mesh.num_owned_cells();
54  smoothing_lengths.resize(ncells);
55  extents.resize(ncells);
56 
57  // get face normal and distance information
58  for (int i=0; i<ncells; i++) {
59  std::vector<int> faces, fdirs;
60  mesh.cell_get_faces_and_dirs(i, &faces, &fdirs);
61 
62  int nfaces = faces.size();
63  std::vector<std::vector<double>> h = smoothing_lengths[i];
64  h.resize(nfaces);
65 
66  Wonton::Point<DIM> fcent, ccent, normal, distance;
67  std::vector<int> fnodes;
68  std::vector<Wonton::Point<DIM>> fncoord;
69  for (int j=0; j<nfaces; j++) {
70  // get face data
71  mesh.face_centroid(faces[j], &fcent);
72  mesh.cell_centroid(i, &ccent);
73  mesh.face_get_nodes(faces[j], &fnodes);
74  fncoord.resize(fnodes.size()+1);
75  for (int k=0; k<fnodes.size(); k++) {
76  mesh.node_get_coordinates(fnodes[k], &(fncoord[k]));
77  }
78  fncoord[fnodes.size()] = fncoord[0];
79 
80  if (DIM==2) {
81  // Get 2d face normal
82  std::vector<double> delta(2);
83  for (int k=0;k<2;k++) delta[k] = fncoord[1][k] - fncoord[0][k];
84  normal[0] = delta[1];
85  normal[1] = -delta[0];
86  if (fdirs[j]<0) normal*=-1.0;
87  double norm = sqrt(normal[0]*normal[0] + normal[1]*normal[1]);
88  normal /= norm;
89  h[j].resize(3);
90  h[j][0] = normal[0];
91  h[j][1] = normal[1];
92  } else if (DIM==3) {
93  // Get 3d face normal using average cross product, in case face is not flat
94  // or has nodes close together.
95  std::vector<std::vector<double>> bivec(2,std::vector<double>(3,0.));
96  for (int k=0;k<3;k++) {
97  bivec[0][k] = fncoord[0][k] - fcent[k];
98  normal[k] = 0.;
99  }
100  for (int k=1; k<=fnodes.size(); k++) {
101  for (int m=0;m<3;m++) bivec[1][m] = fncoord[k][m] - fcent[m];
102  std::vector<double> cross(3,0.);
103  cross[0] = bivec[1][2]*bivec[0][1] - bivec[1][1]*bivec[0][2];
104  cross[1] = -bivec[1][2]*bivec[0][0] + bivec[1][0]*bivec[0][2];
105  cross[2] = bivec[1][1]*bivec[0][0] - bivec[1][0]*bivec[0][1];
106  double norm=sqrt(cross[0]*cross[0]+cross[1]*cross[1]+cross[2]*cross[2]);
107  for (int m=0;m<3;m++) {
108  cross[m]/=norm;
109  }
110  normal += cross;
111  bivec[0] = bivec[1];
112  }
113  if (fdirs[j]<0) normal*=-1.0;
114  double norm=0.;
115  for (int k=0;k<3;k++) norm+=normal[k]*normal[k];
116  norm=sqrt(norm);
117  normal /= norm;
118  h[j].resize(4);
119  for (int k=0;k<3;k++) h[j][k] = normal[k];
120  }
121 
122  distance = Wonton::Point<DIM>(fcent-ccent);
123  double smoothing = 0.0;
124  for (int k=0; k<DIM; k++) smoothing += 2*distance[k]*normal[k];
125  if (smoothing<0.0) { // when nodes are ordered backwards
126  normal = -normal;
127  smoothing = -smoothing;
128  }
129  if (mesh.on_exterior_boundary(Wonton::FACE, faces[j]) and boundary_factor > 0.) {
130  h[j][DIM] = boundary_factor*smoothing;
131  } else {
132  h[j][DIM] = smoothing_factor*smoothing;
133  }
134  }
135  smoothing_lengths[i] = h;
136  }
137 
138  // get extent information
139  for (int i=0; i<ncells; i++) {
140  // calculate the min and max of coordinates
141  std::vector<int> node_indices;
142  mesh.cell_get_nodes(i, &node_indices);
143  Wonton::Point<DIM> cmin, cmax, first_coords;
144  mesh.node_get_coordinates(node_indices[0], &first_coords);
145  for (int k=0; k<DIM; k++) {cmin[k] = first_coords[k]; cmax[k] = first_coords[k];}
146  for (int j=1; j<node_indices.size(); j++) {
147  Wonton::Point<DIM> node_coords;
148  mesh.node_get_coordinates(node_indices[j], &node_coords);
149  for (int k=0; k<DIM; k++) {
150  cmin[k] = std::min(node_coords[k], cmin[k]);
151  cmax[k] = std::max(node_coords[k], cmax[k]);
152  }
153  }
154 
155  // subtract to get extents
156  Wonton::Point<DIM> dx;
157  for (int k=0; k<DIM; k++) dx[k] = cmax[k] - cmin[k];
158 
159  // multiply by smoothing_factor
160  for (int k=0; k<DIM; k++) dx[k] *= 2.*smoothing_factor;
161 
162  extents[i] = dx;
163  }
164  }

184  template<int DIM, class Mesh_Wrapper> void faceted_setup_cell
185  (std::vector<Mesh_Wrapper*> meshes,
186  Portage::vector<std::vector<std::vector<double>>> &smoothing_lengths,
187  Portage::vector<Wonton::Point<DIM>> &extents,
188  double smoothing_factor, double boundary_factor)
189  {
190  size_t total_cells=0;
191  for (auto meshptr=meshes.begin(); meshptr!=meshes.end(); meshptr++) {
192  const Mesh_Wrapper &mesh(**meshptr);
193  total_cells += mesh.num_owned_cells();
194  }
195 
196  smoothing_lengths.resize(total_cells);
197  extents.resize(total_cells);
198 
199  size_t offset = 0;
200  for (auto meshptr=meshes.begin(); meshptr!=meshes.end(); meshptr++) {
201  const Mesh_Wrapper &mesh(**meshptr);
202  size_t ncells = mesh.num_owned_cells();
203  Portage::vector<std::vector<std::vector<double>>> sl;
204  Portage::vector<Wonton::Point<DIM>> ex;
205  faceted_setup_cell<DIM, Mesh_Wrapper>(mesh,sl,ex,smoothing_factor,boundary_factor);
206  for (size_t i=0; i<ncells; i++) {
207  smoothing_lengths[offset+i] = sl[i];
208  extents[offset+i] = ex[i];
209  }
210  offset += ncells;
211  }
212  }

248  template<int DIM, class Mesh_Wrapper, class State_Wrapper> void faceted_setup_cell
249  (const Mesh_Wrapper &mesh,
250  const State_Wrapper &state,
251  std::string field,
252  double tolerance,
253  Portage::vector<std::vector<std::vector<double>>> &smoothing_lengths,
254  Portage::vector<Wonton::Point<DIM>> &extents,
255  double smoothing_factor, double boundary_factor)
256  {
257  faceted_setup_cell
258  (mesh,
259  smoothing_lengths,
260  extents,
261  smoothing_factor, smoothing_factor);
262 
263  double *fval;
264  const_cast<State_Wrapper&>(state).mesh_get_data(Wonton::CELL, field, &fval);
265  for (int i=0; i<mesh.num_owned_cells(); i++) {
266  std::vector<int> faces, dirs, fcells;
267  mesh.cell_get_faces_and_dirs(i, &faces, &dirs);
268  std::vector<std::vector<double>> h = smoothing_lengths[i];
269  for (int j=0; j<faces.size(); j++) {
270  mesh.face_get_cells(faces[j], Wonton::PARALLEL_OWNED, &fcells);
271  for (int k=0; k<fcells.size(); k++) {
272  if (fcells[k] == i) continue;
273  if (fabs(fval[i] - fval[fcells[k]]) > tolerance) {
274  h[j][DIM] *= (boundary_factor / smoothing_factor);
275  }
276  }
277  }
278  smoothing_lengths[i] = h;
279  }
