63 std::vector<double> areaAndMomentPolygon(const std::vector<Portage::Point<2>> poly){
64  double area = 0;
65  double cx = 0;
66  double cy = 0;
67  std::vector <double> ret;
68  for(int i=0;i<poly.size()-1;i++){
69  double a = (poly[i][0]*poly[i+1][1]-poly[i+1][0]*poly[i][1]);
70  area+=a;
71  cx+= (poly[i][0]+poly[i+1][0])*a;
72  cy+= (poly[i][1]+poly[i+1][1])*a;
73  }
74  int lastIndex = poly.size()-1;
75  //close the polygon
76  double a = poly[lastIndex][0]*poly[0][1] - poly[0][0]*poly[lastIndex][1];
77  area+= a;
78  cx+= (poly[lastIndex][0]+poly[0][0])*a;
79  cy+= (poly[lastIndex][1]+poly[0][1])*a;
80  ret.emplace_back(.5*area);
81  ret.emplace_back(cx/6.);
82  ret.emplace_back(cy/6.);
83  return ret;
84 }

115 std::vector<std::vector<double> > operator() (const int cellA, const int cellB) const {
116  Poly polyA, polyB;
117  sourceMeshWrapper.cell_get_coordinates(cellA, &polyA);
118  targetMeshWrapper.cell_get_coordinates(cellB, &polyB);
119  double max_size_poly = 0;
120  max_size_poly = IntersectClipper::updateMaxSize(polyA, max_size_poly);
121  max_size_poly = IntersectClipper::updateMaxSize(polyB, max_size_poly);
122 
123  const ClipperLib::Path intPolyA = IntersectClipper::convertPoly2int(polyA, max_size_poly);
124  const ClipperLib::Path intPolyB = IntersectClipper::convertPoly2int(polyB, max_size_poly);
125 
126  //Make clipper aware of polyA and polyB
127  ClipperLib::Clipper clpr;
128  clpr.AddPath(intPolyA, ClipperLib::ptSubject, true);
129  clpr.AddPath(intPolyB, ClipperLib::ptClip, true);
130 
131  //Compute the intersection of all paths which have been added to clipper (in this case
132  //polyA and polyB; use the Even-Odd winding rule in case of self-intersecting polygons
133  //For more information on the winding rules see:
134  //http://www.angusj.com/delphi/clipper/documentation/Docs/Units/ClipperLib/Types/PolyFillType.htm
135  ClipperLib::Paths solution;
136  clpr.Execute(ClipperLib::ctIntersection, solution,
137  ClipperLib::pftEvenOdd, ClipperLib::pftEvenOdd);
138 
139  std::vector<std::vector<Portage::Point<2>>> intersectionList;
140  for(auto const &i: solution){
141  std::vector<Portage::Point<2>> poly;
142  for(auto const &j: i){
143  //Build list of intersections and convert back to doubles
144  poly.emplace_back(integer2real(j.X, max_size_poly), integer2real(j.Y, max_size_poly));
145  }
146  intersectionList.emplace_back(poly);
147  }
148  std::vector<std::vector<double>> moments;
149  for(auto const &i: intersectionList){
150  moments.emplace_back(areaAndMomentPolygon(i));
151  }
152  return moments;
153 }

167 //We must use the same max size for all the polygons, so the number we are looking for is the maximum value in the set--all the X and Y values will be converted using this value
168 static double updateMaxSize( const std::vector<Portage::Point<2>> poly, double max_size_poly){
169  for(auto const &i: poly){
170  double m = std::max(std::abs(i[0]), std::abs(i[1]));
171  if (m > max_size_poly) max_size_poly = m;
172  }
173  return max_size_poly;
174 }

186 static long real2integer(double a, const double max_size){
187  int exp;
188 
189  // We want to move the decimal point of the floating point number so that the last digit given is before the decimal point.
190  // The alogrithm is to find the exponent (in the floating point representation) of the largest number in the polygons (either X or Y coordinate).
191  //Multiply by the power of two which is number of digits in mantissa (in this case, 53) - exponent
192  //lrint, then rounds to an integer
193 
194  //Compute the exponent
195  frexp(max_size, &exp);
196  //Size of long must be >= size double
197  return lrint(ldexp(a, DBL_MANT_DIG-exp));
198 }
