27 std::vector<double> areaAndMomentPolygon(const std::vector<Wonton::Point<2>> poly){
28  double area = 0;
29  double cx = 0;
30  double cy = 0;
31  std::vector <double> ret;
32  for(int i=0;i<poly.size()-1;i++){
33  double a = (poly[i][0]*poly[i+1][1]-poly[i+1][0]*poly[i][1]);
34  area+=a;
35  cx+= (poly[i][0]+poly[i+1][0])*a;
36  cy+= (poly[i][1]+poly[i+1][1])*a;
37  }
38  int lastIndex = poly.size()-1;
39  //close the polygon
40  double a = poly[lastIndex][0]*poly[0][1] - poly[0][0]*poly[lastIndex][1];
41  area+= a;
42  cx+= (poly[lastIndex][0]+poly[0][0])*a;
43  cy+= (poly[lastIndex][1]+poly[0][1])*a;
44  ret.emplace_back(.5*area);
45  ret.emplace_back(cx/6.);
46  ret.emplace_back(cy/6.);
47  return ret;
48 }

79 std::vector<std::vector<double> > operator() (const int cellA, const int cellB) const {
80  Poly polyA, polyB;
81  sourceMeshWrapper.cell_get_coordinates(cellA, &polyA);
82  targetMeshWrapper.cell_get_coordinates(cellB, &polyB);
83  double max_size_poly = 0;
84  max_size_poly = IntersectClipper::updateMaxSize(polyA, max_size_poly);
85  max_size_poly = IntersectClipper::updateMaxSize(polyB, max_size_poly);
86 
87  const ClipperLib::Path intPolyA = IntersectClipper::convertPoly2int(polyA, max_size_poly);
88  const ClipperLib::Path intPolyB = IntersectClipper::convertPoly2int(polyB, max_size_poly);
89 
90  //Make clipper aware of polyA and polyB
91  ClipperLib::Clipper clpr;
92  clpr.AddPath(intPolyA, ClipperLib::ptSubject, true);
93  clpr.AddPath(intPolyB, ClipperLib::ptClip, true);
94 
95  //Compute the intersection of all paths which have been added to clipper (in this case
96  //polyA and polyB; use the Even-Odd winding rule in case of self-intersecting polygons
97  //For more information on the winding rules see:
98  //http://www.angusj.com/delphi/clipper/documentation/Docs/Units/ClipperLib/Types/PolyFillType.htm
99  ClipperLib::Paths solution;
100  clpr.Execute(ClipperLib::ctIntersection, solution,
101  ClipperLib::pftEvenOdd, ClipperLib::pftEvenOdd);
102 
103  std::vector<std::vector<Wonton::Point<2>>> intersectionList;
104  for(auto const &i: solution){
105  std::vector<Wonton::Point<2>> poly;
106  for(auto const &j: i){
107  //Build list of intersections and convert back to doubles
108  poly.emplace_back(integer2real(j.X, max_size_poly), integer2real(j.Y, max_size_poly));
109  }
110  intersectionList.emplace_back(poly);
111  }
112  std::vector<std::vector<double>> moments;
113  for(auto const &i: intersectionList){
114  moments.emplace_back(areaAndMomentPolygon(i));
115  }
116  return moments;
117 }

150 static long real2integer(double a, const double max_size){
151  int exp;
152 
153  // We want to move the decimal point of the floating point number so that the last digit given is before the decimal point.
154  // The alogrithm is to find the exponent (in the floating point representation) of the largest number in the polygons (either X or Y coordinate).
155  //Multiply by the power of two which is number of digits in mantissa (in this case, 53) - exponent
