846

876  for (int i = 0; i < 4; i++) {

891  // the dual cell around this node in `pplist`. The vertices are ordered CCW.
892  // For boundary node 'nodeid', the first vertex is the node itself, this
893  // uniquely determines the 'pplist' vector. For node 'nodeid' not on a
894  // boundary, the vector 'pplist' starts with a random vertex, but it is still
895  // ordered CCW. Use the 'dual_cell_coordinates_canonical_rotation' to rotate
896  // the 'pplist' into a canonical (unique) form.
897 
898  void dual_cell_get_coordinates(int const nodeid,
899  std::vector<Point<2>> *pplist) const {
900 #ifdef DEBUG
901  assert(nodeid < num_entities(NODE, ALL));
902 #endif
903  assert(basicmesh_ptr_->space_dimension() == 2);
904  assert(wedges_requested_);
905 
906  int cornerid, wedgeid, wedgeid0;
907  std::vector<int> cornerids, wedgeids;
908  std::array<Point<2>, 3> wcoords; // (node, edge midpoint, centroid)
909 
910  // Start with an arbitrary corner
911  node_get_corners(nodeid, ALL, &cornerids);
912  cornerid = cornerids[0];
913 
914  // Process this corner
915  corner_get_wedges(cornerid, &wedgeids);
916  order_wedges_ccw(&wedgeids);
917  wedge_get_coordinates(wedgeids[0], &wcoords);
918  pplist->emplace_back(wcoords[2]); // centroid
919  wedge_get_coordinates(wedgeids[1], &wcoords);
920  pplist->emplace_back(wcoords[1]); // edge midpoint
921 
922  wedgeid0 = wedgeids[0];
923 
924  // Process the rest of the corners in the CCW manner
925  wedgeid = wedge_get_opposite_wedge(wedgeids[1]);
926  // wedgeid == -1 means we are on the boundary, and wedgeid == wedgeid0
927  // means we are not on the boundary and we finished the loop
928  while (wedgeid != -1 && wedgeid != wedgeid0) {
929  cornerid = wedge_get_corner(wedgeid);
930  corner_get_wedges(cornerid, &wedgeids);
931  order_wedges_ccw(&wedgeids);
932  assert(wedgeids[0] == wedgeid);
933  wedge_get_coordinates(wedgeids[0], &wcoords);
934  pplist->push_back(wcoords[2]); // centroid
935  wedge_get_coordinates(wedgeids[1], &wcoords);
936  pplist->push_back(wcoords[1]); // edge midpoint
937  wedgeid = wedge_get_opposite_wedge(wedgeids[1]);
938  }
939 
940  if (wedgeid == -1) {
941  // This is a boundary node, go the other way in a CW manner to get all
942  // the coordinates and include the node (nodeid) itself
943  wedge_get_coordinates(wedgeid0, &wcoords);
944  // edge midpoint
