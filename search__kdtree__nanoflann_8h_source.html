46  // Square of Euclidean distance between query point and a point in the swarm
47  inline double kdtree_distance(double const *p1, size_t const idx, size_t /* size */) const {
48  double dist2 = 0.0;
49  Point<D> const& p2 = swarm_.get_particle_coordinates(idx);
50  for (int i = 0; i < D; i++)
51  dist2 += (p1[i]-p2[i])*(p1[i]-p2[i]);

53  }

90 class Search_KDTree_Nanoflann {

93  // Particular kd-tree type based on nanoflann's offerings
94  typedef
95  nanoflann::KDTreeSingleIndexAdaptor<
96  nanoflann::L2_Simple_Adaptor<double, SwarmNanoflann<D, SourceSwarmType>>,
97  SwarmNanoflann<D, SourceSwarmType>,
98  D
99  > kdtree_t;
100 
102  Search_KDTree_Nanoflann() = delete;
103 
104  // Constructor with swarms and extents
115  Search_KDTree_Nanoflann(SourceSwarmType const& source_swarm,
116  TargetSwarmType const& target_swarm,
117  std::shared_ptr<std::vector<Point<D>>> source_extents,
118  std::shared_ptr<std::vector<Point<D>>> target_extents,
119  Meshfree::WeightCenter center = Meshfree::Gather) :
120  sourceSwarm_(source_swarm), targetSwarm_(target_swarm) {
121 
122  assert(center == Meshfree::Gather); // Our KDTree is built on source points
123 
124  int ntgt = target_swarm.num_owned_particles();
125  search_radii_.resize(ntgt);
126  for (int i = 0; i < ntgt; i++) {
127  double len = 0.0;
128  for (int d = 0; d < D; d++)
129  len += (*target_extents)[i][d]*(*target_extents)[i][d];
130  search_radii_[i] = sqrt(len);
131  }
132 
133  kdtree_ = std::make_shared<kdtree_t>(D, sourceSwarm_,
134  nanoflann::KDTreeSingleIndexAdaptorParams(10 /* maxleaf */));
135  kdtree_->buildIndex();
136 
137  std::cout << "Using Nanoflann KD-Tree code\n";
138  }
139 
149  std::vector<unsigned int> operator() (const size_t pointId) const;
150 
151  private:
152  SwarmNanoflann<D, SourceSwarmType> const sourceSwarm_;
153  SwarmNanoflann<D, TargetSwarmType> const targetSwarm_;
154  std::vector<double> search_radii_;
155  std::shared_ptr<kdtree_t> kdtree_ = nullptr;
