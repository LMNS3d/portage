153  vec& add(std::string const name, Entity_kind const on_what,
154  double const * const data = nullptr) {
155  auto it = find(name, on_what);
156  if (it == end()) {
157  // This is a new variable, so store it.
158  auto num_ent = mesh_->num_entities(on_what, Entity_type::PARALLEL_OWNED);
159  key thisKey(name, on_what);
160  if (data == nullptr) {
161  state_vectors_.insert(std::make_pair(thisKey,
162  vec(num_ent)));
163  } else {
164  state_vectors_.insert(std::make_pair(thisKey,
165  vec(data, data+num_ent)));
166  }
167  names_.emplace_back(name);
168  return state_vectors_[thisKey];
169  } else {
170  // This already exists!
171  std::cerr << "Attempted to add duplicate vectors. Ignoring."
172  << std::endl;
173  return it->second;
174  }
175  }
