25 // intersect one source polygon (possibly non-convex) with a
26 // triangular decomposition of a target polygon
27 
28 std::vector<double>
29 intersect_2Dpolys(std::vector<Point<2>> const & source_poly,
30  std::vector<Point<2>> const & target_poly) {
31 
32  std::vector<double> moments(3, 0);
33  bool src_convex = true;
34  bool trg_convex = true;
35  const double eps = 1e-14;
36 
37  const int POLY_ORDER = 1; // max degree of moments to calculate
38 
39  // Initialize source polygon
40 
41  const int size1 = source_poly.size();
42  std::vector<r2d_rvec2> verts1(size1);
43  for (int i = 0; i < size1; ++i) {
44  verts1[i].xy[0] = source_poly[i][0];
45  verts1[i].xy[1] = source_poly[i][1];
46  }
47 #ifdef DEBUG
48  double volume1 = 0.;
49  for (int i = 1; i < size1-1; ++i)
50  volume1 += r2d_orient(verts1[0], verts1[i], verts1[i+1]);
51  if (volume1 < 0.)
52  throw std::runtime_error("target_poly has negative volume");
53 #endif
54 
55  r2d_poly srcpoly_r2d;
56  r2d_init_poly(&srcpoly_r2d, &verts1[0], size1);
57 
58 #ifdef DEBUG
59  if (r2d_is_good(&srcpoly_r2d) == 0)
60  throw std::runtime_error("source_poly: invalid poly");
61 #endif
62 
63 
64  // Initialize target polygon and check for convexity
65 
66  const int size2 = target_poly.size();
67  std::vector<r2d_plane> faces(size2);
68  std::vector<r2d_rvec2> verts2(size2);
69  for (int i = 0; i < size2; ++i) {
70  verts2[i].xy[0] = target_poly[i][0];
71  verts2[i].xy[1] = target_poly[i][1];
72  }
73 
74  // check for convexity of target polygon
75  for (int i = 0; i < size2; ++i)
76  if (r2d_orient(verts2[i], verts2[(i+1)%size2], verts2[(i+2)%size2]) < eps)
77  trg_convex = false;
78 
79  // case 1: target_poly is convex
80  // can simply use faces of target_poly as clip planes
81  if (trg_convex) {
82  r2d_poly_faces_from_verts(&faces[0], &verts2[0], size2);
83 
84  // clip the first poly against the faces of the second
85  r2d_clip(&srcpoly_r2d, &faces[0], size2);
86 
87  // find the moments (up to quadratic order) of the clipped poly
88  r2d_real om[R2D_NUM_MOMENTS(POLY_ORDER)];
89  r2d_reduce(&srcpoly_r2d, om, POLY_ORDER);
90 
91  // Check that the returned volume is positive (if the volume is zero,
92  // i.e. abs(om[0]) < eps, then it can sometimes be slightly negative,
93  // like om[0] == -1.24811e-16. For this reason we use the condition
94  // om[0] < -eps.
95  if (om[0] < -eps) throw std::runtime_error("Negative volume");
96 
97  // Copy moments:
98  for (int j = 0; j < 3; ++j)
99  moments[j] = om[j];
100 
101  } else { // case 2: target_poly is non-convex
102 
103  // check for convexity of source polygon
104  for (int i = 0; i < size1; ++i)
105  if (r2d_orient(verts1[i], verts1[(i+1)%size1], verts1[(i+2)%size1]) < eps)
106  src_convex = false;
107 
108  // if source polygon is convex while target polygon is non-convex,
109  // call the routine with the polygons reversed
110 
111  if (src_convex)
112  return intersect_2Dpolys(target_poly, source_poly);
113  else {
114 
115  // Must divide target_poly into triangles for clipping. Choice
116  // of the central point is crucial. Try the centroid first -
117  // computed by the area weighted sum of centroids of any
118  // triangulation of the polygon
119  bool center_point_ok = true;
120  Point<2> cen(0.0, 0.0);
121  r2d_rvec2 cenr2d;
122  cenr2d.xy[0] = 0.0; cenr2d.xy[1] = 0.0;
123  double area_sum = 0.0;
124  for (int i = 1; i < size2; ++i) {
125  double area = r2d_orient(verts2[0], verts2[i], verts2[(i+1)%size2]);
126  area_sum += area;
127  Point<2> tricen =
128  (target_poly[0] + target_poly[i] + target_poly[(i+1)%size2])/3.0;
129  cen += area*tricen;
130  }
131  cen /= area_sum;
132  cenr2d.xy[0] = cen[0]; cenr2d.xy[1] = cen[1];
133 
134  for (int i = 0; i < size2; ++i)
135  if (r2d_orient(cenr2d, verts2[i], verts2[(i+1)%size2]) < 0.)
136  center_point_ok = false;
137 
138  if (!center_point_ok) {
139  // If the centroid is not ok, we have to find the center of
140  // the feasible set of the polygon. This means clipping the
141  // target_poly with its own face planes/lines. For a
142  // non-convex polygon, this will give a new polygon whose
143  // interior (See Garimella/Shashkov/Pavel paper on untangling)
144 
145  r2d_poly fspoly;
146  r2d_init_poly(&fspoly, &verts2[0], size2);
147 
148  r2d_poly_faces_from_verts(&faces[0], &verts2[0], size2);
149 
150  r2d_clip(&fspoly, &faces[0], size2);
151 
152  // If the resulting polygon is empty, we are out of luck
153  if (fspoly.nverts == 0)
154  std::runtime_error("Could not find a valid center point to triangulate non-convex polygon");
155 
156  // Have R2D compute first and second moments of polygon and
157  // get its centroid from that
158 
159  r2d_real fspoly_moments[R2D_NUM_MOMENTS(POLY_ORDER)];
160  r2d_reduce(&fspoly, fspoly_moments, POLY_ORDER);
161 
162  cen[0] = cenr2d.xy[0] = fspoly_moments[1]/fspoly_moments[0];
163  cen[1] = cenr2d.xy[1] = fspoly_moments[2]/fspoly_moments[0];
164 
165  // Even if the resulting feasible set polygon has vertices,
166  // maybe its degenerate. So we have to verify that its centroid
167  // indeed is a point that will give valid triangles when
168  // paired with the edges of the target polygon.
169 
170  for (int i = 0; i < size2; ++i)
171  if (r2d_orient(cenr2d, verts2[i], verts2[(i+1)%size2]) < 0.)
172  center_point_ok = false;
173  }
174 
175  // If we still don't have a good center point, we are out of luck
176  if (!center_point_ok)
177  std::runtime_error("Could not find a valid center point to triangulate non-convex polygon");
178 
179 
180  for (int i = 0; i < size2; ++i) {
181  verts2[0].xy[0] = cen[0];
182  verts2[0].xy[1] = cen[1];
183  verts2[1].xy[0] = target_poly[i][0];
184  verts2[1].xy[1] = target_poly[i][1];
185  verts2[2].xy[0] = target_poly[(i+1)%size2][0];
186  verts2[2].xy[1] = target_poly[(i+1)%size2][1];
187 
188  r2d_poly_faces_from_verts(&faces[0], &verts2[0], 3);
189 
190  r2d_poly srcpoly_r2d_copy = srcpoly_r2d;
191 
192  // clip the first poly against the faces of the second
193  r2d_clip(&srcpoly_r2d_copy, &faces[0], 3);
194 
195  // find the moments (up to quadratic order) of the clipped poly
196  // Only do this check in Debug mode:
197 #ifdef DEBUG
198  if (R2D_NUM_MOMENTS(POLY_ORDER) != 3)
199  throw std::runtime_error("Invalid number of moments");
200 #endif
201  r2d_real om[R2D_NUM_MOMENTS(POLY_ORDER)];
202  r2d_reduce(&srcpoly_r2d_copy, om, POLY_ORDER);
203 
204  // Check that the returned volume is positive (if the volume is zero,
205  // i.e. abs(om[0]) < eps, then it can sometimes be slightly negative,
206  // like om[0] == -1.24811e-16. For this reason we use the condition
207  // om[0] < -eps.
208  if (om[0] < -eps)
209  throw std::runtime_error("Negative volume for triangle of polygon");
210 
211  // Accumulate moments:
212  for (int j = 0; j < 3; ++j)
213  moments[j] += om[j];
214  } // for i
215  } // if (src_convex) ... else ...
216  } // if convex {} else {}
217 
218  return moments;
219 }
