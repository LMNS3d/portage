116 template <int D, typename SourceMeshType, typename TargetMeshType>
117 std::vector<int>
118  SearchDirectProduct<D, SourceMeshType, TargetMeshType>::operator() (
119  const int tgt_cell) const {
120 
121  // Tolerance for floating-point round-off
122  const auto EPSILON = 10. * std::numeric_limits<double>::epsilon();
123 
124  // Get the bounding box for the target mesh cell
125  Wonton::Point<D> tlo, thi;
126  targetMesh_.cell_get_bounds(tgt_cell, &tlo, &thi);
127  for (int d = 0; d < D; ++d) {
128  // allow for roundoff error: if the intersection is within epsilon, assume
129  // this is a precision issue rather than true physical overlap
130  auto eps = EPSILON * (thi[d] - tlo[d]);
131  tlo[d] += eps;
132  thi[d] -= eps;
133  }
134 
135  // quick check: does this cell intersect the source mesh at all?
136  // if not, exit early
137  Wonton::Point<D> sglo, sghi;
138  sourceMesh_.get_global_bounds(&sglo, &sghi);
139  for (int d = 0; d < D; ++d) {
140  assert(tlo[d] < thi[d]);
141  assert(sglo[d] < sghi[d]);
142  if (tlo[d] >= sghi[d] || thi[d] <= sglo[d])
143  return std::move(std::vector<int>());
144  }
145 
146  // find which source cells overlap target cell, in each dimension
147  std::array<int,D> ilo, ihi;
148  for (int d = 0; d < D; ++d) {
149  auto saxis_begin = sourceMesh_.axis_point_begin(d);
150  auto saxis_end = sourceMesh_.axis_point_end(d);
151 
152  // find last axis point less than or equal to tlo
153  auto is_point_above = [this,d] (const double x, const int pid) {
154  return (x < this->sourceMesh_.get_axis_point(d, pid));
155  };
156  auto itrlo = std::upper_bound(saxis_begin, saxis_end, tlo[d],
157  is_point_above);
158  ilo[d] = itrlo - saxis_begin - 1;
159  // std::max(ilo[d],0) instead of assert(ilo[d] >= 0) to account for cells
160  // that only partially overlap the target cell
161  ilo[d] = std::max(ilo[d], 0);
162 
163  // find first axis point greater than or equal to thi
164  auto is_point_below = [this,d] (const int pid, const double x) {
165  return (this->sourceMesh_.get_axis_point(d, pid) < x);
166  };
167  auto itrhi = std::lower_bound(itrlo, saxis_end, thi[d],
168  is_point_below);
169  ihi[d] = itrhi - saxis_begin;
170  ihi[d] = std::min(ihi[d], sourceMesh_.axis_num_cells(d));
171  assert(ihi[d] > ilo[d]);
172  } // for d
173 
174  // Generate list of cells from lower and upper bounds, return list
175  return(std::move(list_cells(ilo, ihi)));
