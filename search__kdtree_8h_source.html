81  SearchKDTree(const SourceMeshType & source_mesh,
82  const TargetMeshType & target_mesh)
83  : sourceMesh_(source_mesh), targetMesh_(target_mesh) {
84 
85  const int numCells = sourceMesh_.num_owned_cells();
86  std::vector<gk::IsotheticBBox<D>> bboxes;
87  bboxes.reserve(numCells);
88 
89  // find bounding boxes for all cells
90  for (int c = 0; c < numCells; ++c) {
91  std::vector<Point<D>> cell_coord;
92  sourceMesh_.cell_get_coordinates(c, &cell_coord);
93  gk::IsotheticBBox<D> bb;
94  for (const auto& cc : cell_coord) {
95  bb.add(cc);
96  }
97  bboxes.emplace_back(bb);
98  }
99 
100  // create the k-d tree
101  tree_ = gk::KDTreeCreate(bboxes);
102 
103  } // SearchKDTree::SearchKDTree
104 
106  SearchKDTree(const SearchKDTree &) = delete;
107 
109  SearchKDTree & operator = (const SearchKDTree &) = delete;
110

136 template<int D, typename SourceMeshType, typename TargetMeshType>
137 void SearchKDTree<D, SourceMeshType, TargetMeshType>::
138 operator() (const int cellId, std::vector<int> *candidates)
139 const {
140  // find bounding box for target cell
141  std::vector<Point<D>> cell_coord;
142  targetMesh_.cell_get_coordinates(cellId, &cell_coord);
143  gk::IsotheticBBox<D> bb;
144  for (const auto& cc : cell_coord) {
145  bb.add(cc);
146  }
147 
148  // now see which sourceMesh cells have bounding boxes overlapping
149  // with target cell, using the kdtree
150  std::vector<long> lcandidates;
151  gk::Intersect(bb, tree_, lcandidates);
152  candidates->assign(lcandidates.begin(), lcandidates.end());
153 
154 } // SearchKDTree::operator()
